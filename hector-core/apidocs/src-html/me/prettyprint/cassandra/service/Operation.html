001    package me.prettyprint.cassandra.service;
002    
003    import java.util.Collections;
004    import java.util.Map;
005    
006    import me.prettyprint.cassandra.model.ExecutionResult;
007    import me.prettyprint.cassandra.service.CassandraClientMonitor.Counter;
008    import me.prettyprint.hector.api.ConsistencyLevelPolicy;
009    import me.prettyprint.hector.api.exceptions.HectorException;
010    
011    import org.apache.cassandra.thrift.Cassandra;
012    
013    /**
014     * Defines an operation performed on cassandra
015     *
016     * @param <T>
017     *          The result type of the operation (if it has a result), such as the
018     *          result of get_count or get_column
019     *
020     *          Oh closures, how I wish you were here...
021     */
022    public abstract class Operation<T> {
023      private static final Map<String, String> EMPTY_CREDENTIALS = Collections.emptyMap();
024    
025      /** Counts failed attempts */
026      public final Counter failCounter;
027    
028      /** The stopwatch used to measure operation performance */
029      public final String stopWatchTagName;
030    
031      public FailoverPolicy failoverPolicy;
032      public ConsistencyLevelPolicy consistencyLevelPolicy;
033      
034      public String keyspaceName;
035    
036      public Map<String, String> credentials;
037      
038      protected T result;
039      private HectorException exception;
040      private CassandraHost cassandraHost;
041      private long execTime;
042      public final OperationType operationType;
043      
044      public Operation(OperationType operationType) {
045        this.failCounter = operationType.equals(OperationType.READ) ? Counter.READ_FAIL :
046          Counter.WRITE_FAIL;
047        this.operationType = operationType;
048        this.stopWatchTagName = operationType.name();
049      }
050    
051      public Operation(OperationType operationType, Map<String, String> credentials) {
052        this(operationType, FailoverPolicy.ON_FAIL_TRY_ALL_AVAILABLE, null, credentials);
053      }
054      
055      public Operation(OperationType operationType, FailoverPolicy failoverPolicy, String keyspaceName, Map<String, String> credentials) {
056        this.failCounter = operationType.equals(OperationType.READ) ? Counter.READ_FAIL :
057          Counter.WRITE_FAIL;
058        this.operationType = operationType;
059        this.stopWatchTagName = operationType.name();
060        this.failoverPolicy = failoverPolicy;
061        this.keyspaceName = keyspaceName;
062        this.credentials = Collections.unmodifiableMap(credentials);
063      }
064      
065      
066      public void applyConnectionParams(String keyspace, ConsistencyLevelPolicy consistencyLevelPolicy,
067          FailoverPolicy failoverPolicy, Map<String,String> credentials) {
068        // TODO this is a first step. must be cleaned up.
069        this.keyspaceName = keyspace;
070        this.consistencyLevelPolicy = consistencyLevelPolicy;
071        this.failoverPolicy = failoverPolicy;
072        this.credentials = credentials;
073      }
074    
075      public void setResult(T executionResult) {
076        result = executionResult;
077      }
078    
079      /**
080       *
081       * @return The result of the operation, if this is an operation that has a
082       *         result (such as getColumn etc.
083       */
084      public T getResult() {
085        // TODO remove in favor of getExecutionResult
086        return result;
087      }
088      
089      public ExecutionResult<T> getExecutionResult() {
090        return new ExecutionResult<T>(result, execTime, cassandraHost);
091      }
092    
093      /**
094       * Performs the operation on the given cassandra instance.
095       */
096      public abstract T execute(Cassandra.Client cassandra) throws HectorException;
097    
098      public void executeAndSetResult(Cassandra.Client cassandra, CassandraHost cassandraHost) throws HectorException {
099        this.cassandraHost = cassandraHost;
100        long startTime = System.nanoTime();
101        setResult(execute(cassandra));
102        execTime = System.nanoTime() - startTime;
103      }
104    
105      public void setException(HectorException e) {
106        exception = e;
107      }
108    
109      public boolean hasException() {
110        return exception != null;
111      }
112    
113      public HectorException getException() {
114        return exception;
115      }
116      
117      public CassandraHost getCassandraHost() {
118        return this.cassandraHost;
119      }
120    }
