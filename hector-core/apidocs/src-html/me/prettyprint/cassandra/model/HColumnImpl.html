001    package me.prettyprint.cassandra.model;
002    
003    import static me.prettyprint.cassandra.utils.Assert.notNull;
004    
005    import java.nio.ByteBuffer;
006    
007    import me.prettyprint.cassandra.serializers.SerializerTypeInferer;
008    import me.prettyprint.hector.api.Serializer;
009    import me.prettyprint.hector.api.beans.HColumn;
010    
011    import org.apache.cassandra.thrift.Column;
012    import org.apache.commons.lang.builder.EqualsBuilder;
013    import org.apache.commons.lang.builder.HashCodeBuilder;
014    
015    /**
016     * Hector Column definition.
017     *
018     * @param <N> The type of the column name
019     * @param <V> The type of the column value
020     *
021     * @author Ran Tavory (rantav@gmail.com)
022     *
023     */
024    public final class HColumnImpl<N,V> implements HColumn<N, V> {
025    
026      private N name;
027      private V value;
028      private long clock;
029      private int ttl;
030      private Serializer<N> nameSerializer;
031      private Serializer<V> valueSerializer;
032    
033      public HColumnImpl(N name, V value, long clock, Serializer<N> nameSerializer,
034          Serializer<V> valueSerializer) {
035        this(nameSerializer, valueSerializer);
036        notNull(name, "name is null");
037        notNull(value, "value is null");
038    
039        this.name = name;
040        this.value = value;
041        this.clock = clock;
042      }
043    
044      public HColumnImpl(Column thriftColumn, Serializer<N> nameSerializer,
045          Serializer<V> valueSerializer) {
046        this(nameSerializer, valueSerializer);
047        notNull(thriftColumn, "thriftColumn is null");
048        name = nameSerializer.fromByteBuffer(ByteBuffer.wrap(thriftColumn.getName()));
049        value = valueSerializer.fromByteBuffer(ByteBuffer.wrap(thriftColumn.getValue()));
050        clock = thriftColumn.timestamp;
051      }
052    
053      public HColumnImpl(Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
054        notNull(nameSerializer, "nameSerializer is null");
055        notNull(valueSerializer, "valueSerializer is null");
056        this.nameSerializer = nameSerializer;
057        this.valueSerializer = valueSerializer;
058      }
059    
060      public HColumnImpl(N name, V value, long clock) {
061        this(name, value, clock, SerializerTypeInferer.<N>getSerializer(name),
062            SerializerTypeInferer.<V>getSerializer(value));
063      }
064    
065      @Override
066      public HColumn<N,V> setName(N name) {
067        notNull(name, "name is null");
068        this.name = name;
069        return this;
070      }
071    
072      @Override
073      public HColumn<N, V> setValue(V value) {
074        notNull(value, "value is null");
075        this.value = value;
076        return this;
077      }
078    
079      @Override
080      public HColumn<N,V> setClock(long clock) {
081        this.clock = clock;
082        return this;
083      }
084    
085      @Override
086      public HColumn<N,V> setTtl(int ttl) {
087        this.ttl = ttl;
088        return this;
089      }
090    
091      @Override
092      public int getTtl() {
093        return ttl;
094      }
095    
096      @Override
097      public N getName() {
098        return name;
099      }
100    
101      @Override
102      public V getValue() {
103        return value;
104      }
105    
106      @Override
107      public long getClock() {
108        return clock;
109      }
110    
111      public Column toThrift() {
112        return ttl > 0 ? new Column(nameSerializer.toByteBuffer(name), valueSerializer.toByteBuffer(value), clock).setTtl(ttl) :
113          new Column(nameSerializer.toByteBuffer(name), valueSerializer.toByteBuffer(value), clock);
114      }
115    
116      public HColumn<N, V> fromThrift(Column c) {
117        notNull(c, "column is null");
118        name = nameSerializer.fromByteBuffer(c.name);
119        value = valueSerializer.fromByteBuffer(c.value);
120        clock = c.timestamp;
121        ttl = c.ttl;
122        return this;
123      }
124    
125      @Override
126      public Serializer<N> getNameSerializer() {
127        return nameSerializer;
128      }
129    
130      @Override
131      public Serializer<V> getValueSerializer() {
132        return valueSerializer;
133      }
134    
135    
136      @Override
137      public String toString() {
138        return "HColumn(" + name + "=" + value + ")";
139      }
140    
141      @Override
142      public int hashCode() {
143        return new HashCodeBuilder().append(name).append(value).append(clock).toHashCode();
144      }
145    
146      @Override
147      public boolean equals(Object obj) {
148        if (obj == null) {
149          return false;
150        }
151        if (obj == this) {
152          return true;
153        }
154        if (obj.getClass() != getClass()) {
155          return false;
156        }
157        @SuppressWarnings("unchecked")
158        HColumnImpl<N,V> other = (HColumnImpl<N,V>) obj;
159        return new EqualsBuilder().appendSuper(super.equals(obj)).append(name, other.name).
160            append(value, other.value).append(clock, other.clock).isEquals();
161      }
162    }
