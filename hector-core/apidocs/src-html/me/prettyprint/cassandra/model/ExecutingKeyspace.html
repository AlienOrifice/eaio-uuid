018    /**
019     * Thread Safe
020     * @author Ran Tavory
021     * @author zznate
022     */
023    public class ExecutingKeyspace implements Keyspace {
024      private static final Map<String, String> EMPTY_CREDENTIALS = Collections.emptyMap();
025    
026      private ConsistencyLevelPolicy consistencyLevelPolicy;
027      private FailoverPolicy failoverPolicy;
028    
029      private final HConnectionManager connectionManager;
030      private final String keyspace;
031      private final Map<String, String> credentials;
032      private final ExceptionsTranslator exceptionTranslator;
033    
034      public ExecutingKeyspace(String keyspace, HConnectionManager connectionManager,
035          ConsistencyLevelPolicy consistencyLevelPolicy, FailoverPolicy failoverPolicy) {
036          this(keyspace, connectionManager, consistencyLevelPolicy, failoverPolicy, EMPTY_CREDENTIALS);
037      }
038      
039      public ExecutingKeyspace(String keyspace, HConnectionManager connectionManager,
040          ConsistencyLevelPolicy consistencyLevelPolicy, FailoverPolicy failoverPolicy, 
041          Map<String, String> credentials) {
042        Assert.noneNull(consistencyLevelPolicy, connectionManager);
043        this.keyspace = keyspace;
044        this.connectionManager = connectionManager;
045        this.consistencyLevelPolicy = consistencyLevelPolicy;
046        this.failoverPolicy = failoverPolicy;
047        this.credentials = credentials;
048        // TODO make this plug-able
049        this.exceptionTranslator = new ExceptionsTranslatorImpl();
050      }
051    
052      @Override
053      public void setConsistencyLevelPolicy(ConsistencyLevelPolicy cp) {
054        // TODO remove this method
055        this.consistencyLevelPolicy = cp;
056      }
057    
058      @Override
059      public String toString() {
060        return "ExecutingKeyspace(" + keyspace +"," + connectionManager + ")";
061      }
062    
063      @Override
064      public long createClock() {
065        return connectionManager.createClock();
066      }
067    
068      public <T> ExecutionResult<T> doExecute(KeyspaceOperationCallback<T> koc) throws HectorException {
069        KeyspaceService ks = null;
070        try {
071          ks = new KeyspaceServiceImpl(keyspace, consistencyLevelPolicy, connectionManager, failoverPolicy, credentials);
072          return koc.doInKeyspaceAndMeasure(ks);
073        } finally {
074          if (ks != null) {
075            //connectionManager.releaseClient(ks.getClient());
076          }
077        }
078      }
079     
080      public <T> ExecutionResult<T> doExecuteOperation(Operation<T> operation) throws HectorException {
081        operation.applyConnectionParams(keyspace,consistencyLevelPolicy,failoverPolicy,credentials);
082        connectionManager.operateWithFailover(operation);
083        return operation.getExecutionResult();
084      }
085      
086      public ExceptionsTranslator getExceptionsTranslator() {
087        return exceptionTranslator;
088      }
089    }
