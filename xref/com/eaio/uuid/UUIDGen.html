43  /**
44   * This class contains methods to generate UUID fields. These methods have been
45   * refactored out of {@link com.eaio.uuid.UUID}.
46   * <p>
47   * Starting with version 2, this implementation tries to obtain the MAC address
48   * of the network card. Under Microsoft Windows, the <code>ifconfig</code>
49   * command is used which may pop up a command window in Java Virtual Machines
50   * prior to 1.4 once this class is initialized. The command window is closed
51   * automatically.
52   * <p>
53   * The MAC address code has been tested extensively in Microsoft Windows,
54   * Linux, Solaris 8, HP-UX 11, but should work in MacOS X and BSDs, too.
55   * <p>
56   * If you use JDK 6 or later, the code in {@link InterfaceAddress} will be used.
57   *
58   * @see <a href="http://johannburkard.de/software/uuid/">UUID</a>
59   * @author <a href="mailto:jb@eaio.de">Johann Burkard</a>
60   * @version $Id: UUIDGen.java 2914 2010-04-23 11:35:00Z johann $
61   * @see com.eaio.uuid.UUID
62   */
63  public final class UUIDGen {
64  
65      /**
66       * No instances needed.
67       */
68      private UUIDGen() {
69          super();
70      }
71  
72      /**
73       * The last time value. Used to remove duplicate UUIDs.
74       */
75      private static long lastTime = Long.MIN_VALUE;
76      
77      /**
78       * The cached MAC address.
79       */
80      private static String macAddress = null;
81  
82      /**
83       * The current clock and node value.
84       */
85      private static long clockSeqAndNode = 0x8000000000000000L;
86  
87      static {
88  
89          try {
90              Class.forName("java.net.InterfaceAddress");
91              macAddress = Class.forName(
92                      "com.eaio.uuid.UUIDGen$HardwareAddressLookup").newInstance().toString();
93          }
94          catch (ExceptionInInitializerError err) {
95              // Ignored.
96          }
97          catch (ClassNotFoundException ex) {
98              // Ignored.
99          }
100         catch (LinkageError err) {
101             // Ignored.
102         }
103         catch (IllegalAccessException ex) {
104             // Ignored.
105         }
106         catch (InstantiationException ex) {
107             // Ignored.
108         }
109         catch (SecurityException ex) {
110             // Ignored.
111         }
112 
113         if (macAddress == null) {
114 
115             Process p = null;
116             BufferedReader in = null;
117 
118             try {
119                 String osname = System.getProperty("os.name", "");
120 
121                 if (osname.startsWith("Windows")) {
122                     p = Runtime.getRuntime().exec(
123                             new String[] { "ipconfig", "/all" }, null);
124                 }
125                 // Solaris code must appear before the generic code
126                 else if (osname.startsWith("Solaris")
127                         || osname.startsWith("SunOS")) {
128                     String hostName = getFirstLineOfCommand(
129                             "uname", "-n" );
130                     if (hostName != null) {
131                         p = Runtime.getRuntime().exec(
132                                 new String[] { "/usr/sbin/arp", hostName },
133                                 null);
134                     }
135                 }
136                 else if (new File("/usr/sbin/lanscan").exists()) {
137                     p = Runtime.getRuntime().exec(
138                             new String[] { "/usr/sbin/lanscan" }, null);
139                 }
140                 else if (new File("/sbin/ifconfig").exists()) {
141                     p = Runtime.getRuntime().exec(
142                             new String[] { "/sbin/ifconfig", "-a" }, null);
143                 }
144 
145                 if (p != null) {
146                     in = new BufferedReader(new InputStreamReader(
147                             p.getInputStream()), 128);
148                     String l = null;
149                     while ((l = in.readLine()) != null) {
150                         macAddress = MACAddressParser.parse(l);
151                         if (macAddress != null
152                                 && Hex.parseShort(macAddress) != 0xff) {
153                             break;
154                         }
155                     }
156                 }
157 
158             }
159             catch (SecurityException ex) {
160                 // Ignore it.
161             }
162             catch (IOException ex) {
163                 // Ignore it.
164             }
165             finally {
166                 if (p != null) {
167                     if (in != null) {
168                         try {
169                             in.close();
170                         }
171                         catch (IOException ex) {
172                             // Ignore it.
173                         }
174                     }
175                     try {
176                         p.getErrorStream().close();
177                     }
178                     catch (IOException ex) {
179                         // Ignore it.
180                     }
181                     try {
182                         p.getOutputStream().close();
183                     }
184                     catch (IOException ex) {
185                         // Ignore it.
186                     }
187                     p.destroy();
188                 }
189             }
190 
191         }
192 
193         if (macAddress != null) {
194             clockSeqAndNode |= Hex.parseLong(macAddress);
195         }
196         else {
197             try {
198                 byte[] local = InetAddress.getLocalHost().getAddress();
199                 clockSeqAndNode |= (local[0] << 24) & 0xFF000000L;
200                 clockSeqAndNode |= (local[1] << 16) & 0xFF0000;
201                 clockSeqAndNode |= (local[2] << 8) & 0xFF00;
202                 clockSeqAndNode |= local[3] & 0xFF;
203             }
204             catch (UnknownHostException ex) {
205                 clockSeqAndNode |= (long) (Math.random() * 0x7FFFFFFF);
206             }
207         }
208 
209         // Skip the clock sequence generation process and use random instead.
210 
211         clockSeqAndNode |= (long) (Math.random() * 0x3FFF) << 48;
212 
213     }

225     /**
226      * Generates a new time field. Each time field is unique and larger than the
227      * previously generated time field.
228      *
229      * @return a new time value
230      * @see UUID#getTime()
231      */
232     public static long newTime() {
233         return createTime(System.currentTimeMillis());
234     }
235     
236     /**
237      * Creates a new time field from the given timestamp. Note that even identical
238      * values of <code>currentTimeMillis</code> will produce different time fields.
239      * 
240      * @param currentTimeMillis the timestamp
241      * @return a new time value
242      * @see UUID#getTime()
243      */
244     public static synchronized long createTime(long currentTimeMillis) {
245 
246         long time;
247 
248         // UTC time
249 
250         long timeMillis = (currentTimeMillis * 10000) + 0x01B21DD213814000L;
251 
252         if (timeMillis > lastTime) {
253             lastTime = timeMillis;
254         }
255         else {
256             timeMillis = ++lastTime;
257         }
258 
259         // time low
260 
261         time = timeMillis << 32;
262 
263         // time mid
264 
265         time |= (timeMillis & 0xFFFF00000000L) >> 16;
266 
267         // time hi and version
268 
269         time |= 0x1000 | ((timeMillis >> 48) & 0x0FFF); // version 1
270 
271         return time;
272 
273     }

285      * Returns the first line of the shell command.
286      *
287      * @param commands the commands to run
288      * @return the first line of the command
289      * @throws IOException
290      */
291     static String getFirstLineOfCommand(String... commands) throws IOException {
292 
293         Process p = null;
294         BufferedReader reader = null;
295 
296         try {
297             p = Runtime.getRuntime().exec(commands);
298             reader = new BufferedReader(new InputStreamReader(
299                     p.getInputStream()), 128);
300 
301             return reader.readLine();
302         }
303         finally {
304             if (p != null) {
305                 if (reader != null) {
306                     try {
307                         reader.close();
308                     }
309                     catch (IOException ex) {
310                         // Ignore it.
311                     }
312                 }
313                 try {
314                     p.getErrorStream().close();
315                 }
316                 catch (IOException ex) {
317                     // Ignore it.
318                 }
319                 try {
320                     p.getOutputStream().close();
321                 }
322                 catch (IOException ex) {
323                     // Ignore it.
324                 }
325                 p.destroy();
326             }
327         }

331     /**
332      * Scans MAC addresses for good ones.
333      */
334     static class HardwareAddressLookup {
335 
336         /**
337          * @see java.lang.Object#toString()
338          */
339         @Override
340         public String toString() {
341             String out = null;
342             try {
343                 Enumeration<NetworkInterface> ifs = NetworkInterface.getNetworkInterfaces();
344                 if (ifs != null) {
345                     while (ifs.hasMoreElements()) {
346                         NetworkInterface iface = ifs.nextElement();
347                         byte[] hardware = iface.getHardwareAddress();
348                         if (hardware != null && hardware.length == 6
349                                 && hardware[1] != (byte) 0xff) {
350                             out = Hex.append(new StringBuilder(36), hardware).toString();
351                             break;
352                         }
353                     }
354                 }
355             }
356             catch (SocketException ex) {
357                 // Ignore it.
358             }
359             return out;
360         }
