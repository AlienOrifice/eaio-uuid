001    package me.prettyprint.hector.api;
002    
003    import java.util.List;
004    import java.util.Map;
005    import java.util.Set;
006    
007    import me.prettyprint.cassandra.connection.HConnectionManager;
008    import me.prettyprint.cassandra.service.CassandraHost;
009    import me.prettyprint.hector.api.ddl.ColumnFamilyDefinition;
010    import me.prettyprint.hector.api.ddl.KeyspaceDefinition;
011    import me.prettyprint.hector.api.exceptions.HectorException;
012    
013    public interface Cluster {
014    
015      Set<CassandraHost> getKnownPoolHosts(boolean refresh);
016    
017      HConnectionManager getConnectionManager();
018    
019    
020      /**
021       * Adds the host to this Cluster. Unless skipApplyConfig is set to true, the settings in
022       * the CassandraHostConfigurator will be applied to the provided CassandraHost
023       * @param cassandraHost
024       * @param skipApplyConfig
025       */
026      void addHost(CassandraHost cassandraHost, boolean skipApplyConfig);
027    
028      /**
029       * Descriptive name of the cluster.
030       * This name is used to identify the cluster.
031       * @return
032       */
033      String getName();
034    
035    
036      String describeClusterName() throws HectorException;
037    
038      String describeThriftVersion() throws HectorException;
039    
040      KeyspaceDefinition describeKeyspace(final String keyspace) throws HectorException;
041    
042      String getClusterName() throws HectorException;
043    
044      List<KeyspaceDefinition> describeKeyspaces() throws HectorException;
045    
046      /**
047       * Drops the Keyspace from the cluster. Equivalent of 'drop database' in SQL terms
048       *
049       */
050      String dropKeyspace(final String keyspace) throws HectorException;
051    
052      /**
053       * Updates the Keyspace from the cluster.
054       */
055      String updateKeyspace(final KeyspaceDefinition ksdef) throws HectorException;
056    
057      String describePartitioner() throws HectorException;
058    
059      String addColumnFamily(final ColumnFamilyDefinition cfdef) throws HectorException;
060    
061      String dropColumnFamily(final String keyspaceName, final String columnFamily)
062          throws HectorException;
063      
064      String updateColumnFamily(final ColumnFamilyDefinition cfdef) throws HectorException;
065    
066      String addKeyspace(final KeyspaceDefinition ksdef) throws HectorException;
067    
068      Map<String, String> getCredentials();
069    }
