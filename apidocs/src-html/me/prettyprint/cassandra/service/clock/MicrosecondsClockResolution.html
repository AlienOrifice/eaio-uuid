001    package me.prettyprint.cassandra.service.clock;
002    
003    import me.prettyprint.hector.api.ClockResolution;
004    
005    /**
006     * Milliseconds Resolution used to create clocks.
007     *
008     * @author Patricio Echague (pechague@gmail.com)
009     */
010    public class MicrosecondsClockResolution extends AbstractClockResolution implements ClockResolution {
011    
012      private static final long ONE_THOUSAND = 1000L;
013    
014      @Override
015      public long createClock() {
016        return getSystemMilliseconds() * ONE_THOUSAND;
017      }
018    
019    }
