001    package me.prettyprint.cassandra.connection;
002    
003    import java.util.Collection;
004    import java.util.Set;
005    import java.util.concurrent.atomic.AtomicLong;
006    
007    import org.cliffc.high_scale_lib.Counter;
008    
009    import me.prettyprint.cassandra.service.CassandraHost;
010    
011    public class RoundRobinBalancingPolicy implements LoadBalancingPolicy {
012    
013      private AtomicLong counter;
014      
015      public RoundRobinBalancingPolicy() {
016        counter = new AtomicLong(Long.MIN_VALUE);
017      }
018      
019      @Override
020      public ConcurrentHClientPool getPool(Collection<ConcurrentHClientPool> pools,
021          Set<CassandraHost> excludeHosts) {
022        ConcurrentHClientPool[] pa = pools.toArray(new ConcurrentHClientPool[pools.size()]);
023        int location = getAndIncrement(pa.length);
024        ConcurrentHClientPool pool = pa[location];
025        if ( excludeHosts != null && excludeHosts.size() > 0 ) {
026          while ( excludeHosts.contains(pool.getCassandraHost()) ) {
027            pool = pa[getAndIncrement(pa.length)]; 
028          }
029        }
030        return pool;
031      }
032        
033      private int getAndIncrement(int size) {
034        return (int)counter.getAndIncrement() % size;    
035      }
036    
037    }
