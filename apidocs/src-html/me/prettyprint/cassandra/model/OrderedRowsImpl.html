001    package me.prettyprint.cassandra.model;
002    
003    import java.util.ArrayDeque;
004    import java.util.ArrayList;
005    import java.util.Collections;
006    import java.util.LinkedHashMap;
007    import java.util.List;
008    import java.util.Map;
009    
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.beans.OrderedRows;
012    import me.prettyprint.hector.api.beans.Row;
013    
014    import org.apache.cassandra.thrift.Column;
015    
016    /**
017     * Return type from get_range_slices for simple columns
018     * @author Ran Tavory
019     * @author zznate
020     *
021     * @param <N>
022     * @param <V>
023     */
024    public final class OrderedRowsImpl<K,N,V> extends RowsImpl<K,N,V> implements OrderedRows<K, N, V> {
025    
026      private final ArrayDeque<Row<K,N,V>> rowsList;
027    
028      public OrderedRowsImpl(LinkedHashMap<K, List<Column>> thriftRet, Serializer<N> nameSerializer,
029          Serializer<V> valueSerializer) {
030        super(thriftRet, nameSerializer, valueSerializer);
031        rowsList = new ArrayDeque<Row<K,N,V>>(rows.values());    
032      }
033    
034      /**
035       * Preserves rows order
036       * @return an unmodifiable list of Rows
037       */
038      @Override
039      public List<Row<K,N,V>> getList() {
040        return Collections.unmodifiableList(new ArrayList<Row<K, N, V>>(rowsList));
041      }
042    
043      /**
044       * Returns the last element in this row result. Helpful for passing along for paging situations.
045       * @return
046       */
047      @Override
048      public Row<K, N, V> peekLast() {
049        return rowsList != null && rowsList.size() > 0 ? rowsList.peekLast() : null;
050      }
051    }
