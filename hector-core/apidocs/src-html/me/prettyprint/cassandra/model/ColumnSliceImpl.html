001    package me.prettyprint.cassandra.model;
002    
003    import java.util.ArrayList;
004    import java.util.Collections;
005    import java.util.HashMap;
006    import java.util.List;
007    import java.util.Map;
008    
009    import me.prettyprint.cassandra.utils.Assert;
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.beans.ColumnSlice;
012    import me.prettyprint.hector.api.beans.HColumn;
013    
014    import org.apache.cassandra.thrift.Column;
015    
016    public final class ColumnSliceImpl<N,V> implements ColumnSlice<N, V> {
017    
018      private final Map<N,HColumn<N,V>> columnsMap;
019      private final List<HColumn<N,V>> columnsList;
020    
021      public ColumnSliceImpl(List<Column> tColumns, Serializer<N> nameSerializer,
022          Serializer<V> valueSerializer) {
023        Assert.noneNull(tColumns, nameSerializer, valueSerializer);
024        columnsMap = new HashMap<N,HColumn<N,V>>(tColumns.size());
025        List<HColumn<N,V>> list = new ArrayList<HColumn<N,V>>(tColumns.size());
026        for (Column c: tColumns) {
027          HColumn<N, V> column = new HColumnImpl<N,V>(c, nameSerializer, valueSerializer);
028          columnsMap.put(column.getName(), column);
029          list.add(column);
030        }
031        columnsList = Collections.unmodifiableList(list);
032      }
033    
034      /**
035       *
036       * @return an unmodifiable list of the columns
037       */
038      @Override
039      public List<HColumn<N,V>> getColumns() {
040        return columnsList;
041      }
042    
043      @Override
044      public HColumn<N, V> getColumnByName(N columnName) {
045        return columnsMap.get(columnName);
046      }
047    
048      @Override
049      public String toString() {
050        return "ColumnSlice(" + columnsList + ")";
051      }
052    }
