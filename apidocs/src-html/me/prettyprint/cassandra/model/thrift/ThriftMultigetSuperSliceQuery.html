005    import java.util.Collection;

007    import java.util.Map;

022    import org.apache.cassandra.thrift.ColumnParent;
023    import org.apache.cassandra.thrift.SuperColumn;

025    /**
026     * A query wrapper for the thrift call multiget_slice for a slice of supercolumns
027     *
028     * @author ran
029     */
030    public final class ThriftMultigetSuperSliceQuery<K, SN, N, V> extends
031        AbstractSliceQuery<K, SN, V, SuperRows<K, SN, N, V>> implements MultigetSuperSliceQuery<K, SN, N, V> {
032    
033      private Collection<K> keys;
034      private final Serializer<N> nameSerializer;
035    
036      public ThriftMultigetSuperSliceQuery(Keyspace keyspace,
037          Serializer<K> keySerializer,
038          Serializer<SN> sNameSerializer,
039          Serializer<N> nameSerializer,
040          Serializer<V> valueSerializer) {
041        super(keyspace, keySerializer, sNameSerializer, valueSerializer);
042        Assert.notNull(nameSerializer, "nameSerializer can't be null");
043        this.nameSerializer = nameSerializer;
044      }
045    
046      @Override
047      public MultigetSuperSliceQuery<K, SN, N, V> setKeys(K... keys) {
048        this.keys = Arrays.asList(keys);
049        return this;
050      }
051    
052      @Override
053      public QueryResult<SuperRows<K, SN, N, V>> execute() {
054        return new QueryResultImpl<SuperRows<K, SN, N, V>>(
055            keyspace.doExecute(new KeyspaceOperationCallback<SuperRows<K, SN, N, V>>() {
056              @Override
057              public SuperRows<K, SN, N, V> doInKeyspace(KeyspaceService ks) throws HectorException {
058                List<K> keysList = new ArrayList<K>();
059                keysList.addAll(keys);
060                ColumnParent columnParent = new ColumnParent(columnFamilyName);
061                Map<K, List<SuperColumn>> thriftRet = keySerializer.fromBytesMap(ks.multigetSuperSlice(
062                    keySerializer.toBytesList(keysList), columnParent, getPredicate()));
063                return new SuperRowsImpl<K, SN, N, V>(thriftRet, keySerializer, columnNameSerializer,
064                    nameSerializer, valueSerializer);
065              }
066            }), this);
067      }
068    
069      @Override
070      public String toString() {
071        return "MultigetSuperSliceQuery(" + keys + "," + super.toStringInternal() + ")";
072      }
073    
074      @SuppressWarnings("unchecked")
075      @Override
076      public MultigetSuperSliceQuery<K, SN, N, V> setRange(SN start, SN finish, boolean reversed, int count) {
077        return (MultigetSuperSliceQuery<K, SN, N, V>) super.setRange(start, finish, reversed, count);
078      }
079    
080      @SuppressWarnings("unchecked")
081      @Override
082      public MultigetSuperSliceQuery<K, SN, N, V> setColumnNames(SN... columnNames) {
083        return (MultigetSuperSliceQuery<K, SN, N, V>) super.setColumnNames(columnNames);
084      }
085    
086      @SuppressWarnings("unchecked")
087      @Override
088      public MultigetSuperSliceQuery<K, SN, N, V> setColumnFamily(String cf) {
089        return (MultigetSuperSliceQuery<K, SN, N, V>) super.setColumnFamily(cf);
090      }
091    
092    }
