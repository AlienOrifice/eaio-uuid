016    import org.junit.After;

020    public class IndexedSlicesQueryTest extends BaseEmbededServerSetupTest {
021    
022      private final static String KEYSPACE = "Keyspace1";
023      private static final StringSerializer se = new StringSerializer();
024      private static final LongSerializer le = new LongSerializer();
025      private Cluster cluster;
026      private Keyspace keyspace;
027      private String cf = "Indexed1";
028    
029      @Before
030      public void setupCase() {
031        cluster = getOrCreateCluster("MyCluster", "127.0.0.1:9170");
032        keyspace = createKeyspace(KEYSPACE, cluster);
033        createMutator(keyspace, se)
034        .addInsertion("indexedSlicesTest_key1", cf, createColumn("birthyear", 1974L, se, le))
035        .addInsertion("indexedSlicesTest_key1", cf, createColumn("birthmonth", 4L, se, le))
036        .addInsertion("indexedSlicesTest_key2", cf, createColumn("birthyear", 1975L, se, le))
037        .addInsertion("indexedSlicesTest_key2", cf, createColumn("birthmonth", 4L, se, le))
038        .addInsertion("indexedSlicesTest_key3", cf, createColumn("birthyear", 1975L, se, le))
039        .addInsertion("indexedSlicesTest_key3", cf, createColumn("birthmonth", 5L, se, le))
040        .addInsertion("indexedSlicesTest_key4", cf, createColumn("birthyear", 1975L, se, le))
041        .addInsertion("indexedSlicesTest_key4", cf, createColumn("birthmonth", 6L, se, le))
042        .addInsertion("indexedSlicesTest_key5", cf, createColumn("birthyear", 1975L, se, le))
043        .addInsertion("indexedSlicesTest_key5", cf, createColumn("birthmonth", 7L, se, le))
044        .addInsertion("indexedSlicesTest_key6", cf, createColumn("birthyear", 1976L, se, le))
045        .addInsertion("indexedSlicesTest_key6", cf, createColumn("birthmonth", 6L, se, le))
046        .execute();
047      }
048    
049      @After
050      public void teardownCase() {
051        keyspace = null;
052        cluster = null;
053      }
054    
055      @Test
056      public void testInsertGetRemove() {        
057    
058        IndexedSlicesQuery<String, String, Long> indexedSlicesQuery = new IndexedSlicesQuery<String, String, Long>(keyspace, se, se, le);
059        indexedSlicesQuery.addEqualsExpression("birthyear", 1975L);
060        //indexedSlicesQuery.setColumnNames("birthyear");
061        indexedSlicesQuery.setReturnKeysOnly();
062        indexedSlicesQuery.setColumnFamily(cf);
063        indexedSlicesQuery.setStartKey("");
064        QueryResult<OrderedRows<String, String, Long>> result = indexedSlicesQuery.execute();
065        assertEquals(4, result.get().getList().size());
066    
067    
068      }
069      
070      @Test
071      public void testMultiClause() {        
072    
073        QueryResult<OrderedRows<String, String, Long>> result = 
074          new IndexedSlicesQuery<String, String, Long>(keyspace, se, se, le)
075        .addEqualsExpression("birthyear", 1975L)
076        .addGteExpression("birthmonth", 4L)
077        .addLteExpression("birthmonth", 6L)
078        .setColumnNames("birthyear")
079        .setColumnFamily(cf)
080        .setStartKey("")
081        .execute();
082        assertEquals(3, result.get().getList().size());
083    
084    
085      }
086    
087      @Test
088      public void testEqClauseMiss() {        
089        QueryResult<OrderedRows<String, String, Long>> result = 
090          new IndexedSlicesQuery<String, String, Long>(keyspace, se, se, le)
091        .addEqualsExpression("birthyear", 5L)
092        .addGteExpression("birthmonth", 4L)
093        .addLteExpression("birthmonth", 6L)    
094        .setColumnNames("birthyear")
095        .setColumnFamily(cf)
096        .setStartKey("")
097        .execute();
098        assertEquals(0, result.get().getList().size());
099      }
100    
101      @Test
102      public void testRowCountLimit() {
103        QueryResult<OrderedRows<String, String, Long>> result = 
104          new IndexedSlicesQuery<String, String, Long>(keyspace, se, se, le)
105        .addEqualsExpression("birthyear", 1975L)
106        .addGteExpression("birthmonth", 4L)
107        .addLteExpression("birthmonth", 6L)
108        .setColumnNames("birthyear")
109        .setColumnFamily(cf)
110        .setStartKey("")
111        .setRowCount(2)
112        .execute();
113        assertEquals(2, result.get().getList().size());
114      }
