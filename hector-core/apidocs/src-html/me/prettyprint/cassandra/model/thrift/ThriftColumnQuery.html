001    package me.prettyprint.cassandra.model.thrift;
002    
003    import me.prettyprint.cassandra.model.AbstractColumnQuery;
004    import me.prettyprint.cassandra.model.HColumnImpl;
005    import me.prettyprint.cassandra.model.KeyspaceOperationCallback;
006    import me.prettyprint.cassandra.model.QueryResultImpl;
007    import me.prettyprint.cassandra.serializers.StringSerializer;
008    import me.prettyprint.cassandra.serializers.TypeInferringSerializer;
009    import me.prettyprint.cassandra.service.KeyspaceService;
010    import me.prettyprint.hector.api.Keyspace;
011    import me.prettyprint.hector.api.Serializer;
012    import me.prettyprint.hector.api.beans.HColumn;
013    import me.prettyprint.hector.api.exceptions.HNotFoundException;
014    import me.prettyprint.hector.api.exceptions.HectorException;
015    import me.prettyprint.hector.api.query.ColumnQuery;
016    import me.prettyprint.hector.api.query.QueryResult;
017    
018    import org.apache.cassandra.thrift.Column;
019    
020    /**
021     * Thrift implementation of the ColumnQuery type.
022     *
023     * @author Ran Tavory
024     *
025     * @param <N>
026     *          column name type
027     * @param <V>
028     *          value type
029     */
030    public class ThriftColumnQuery<K, N, V> extends AbstractColumnQuery<K, N, V> implements
031        ColumnQuery<K, N, V> {
032    
033      public ThriftColumnQuery(Keyspace keyspace, Serializer<K> keySerializer,
034          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
035        super(keyspace, keySerializer, nameSerializer, valueSerializer);
036      }
037    
038      public ThriftColumnQuery(Keyspace keyspace, Serializer<V> valueSerializer) {
039        super(keyspace, TypeInferringSerializer.<K> get(), TypeInferringSerializer.<N> get(),
040            valueSerializer);
041      }
042    
043      @SuppressWarnings("unchecked")
044      public ThriftColumnQuery(Keyspace keyspaceOperator) {
045        super(keyspaceOperator, TypeInferringSerializer.<K> get(), TypeInferringSerializer.<N> get(),
046            (Serializer<V>) StringSerializer.get());
047      }
048    
049      @Override
050      public QueryResult<HColumn<N, V>> execute() {
051        return new QueryResultImpl<HColumn<N, V>>(
052            keyspace.doExecute(new KeyspaceOperationCallback<HColumn<N, V>>() {
053    
054              @Override
055              public HColumn<N, V> doInKeyspace(KeyspaceService ks) throws HectorException {
056                try {
057                  Column thriftColumn = ks.getColumn(keySerializer.toByteBuffer(key),
058                      ThriftFactory.createColumnPath(columnFamilyName, name, columnNameSerializer));
059                  return new HColumnImpl<N, V>(thriftColumn, columnNameSerializer, valueSerializer);
060                } catch (HNotFoundException e) {
061                  return null;
062                }
063              }
064            }), this);
065      }
066    }
