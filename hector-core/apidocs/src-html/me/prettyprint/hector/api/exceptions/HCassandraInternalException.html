001    package me.prettyprint.hector.api.exceptions;
002    
003    /**
004     * Designed to loosely wrap TApplicationException which can be generated
005     * by Apache Cassandra under a variety of ambiguous conditions - some of
006     * them transient, some of them not. 
007     *
008     * @author zznate
009     */
010    public class HCassandraInternalException extends HectorException {
011    
012      private static final long serialVersionUID = -266109391311421129L;
013      
014      private int type;
015      
016      private static final String ERR_MSG = 
017        "Cassandra encountered an internal error processing this request: ";
018      
019      public HCassandraInternalException(String msg) {
020        super(ERR_MSG + msg); 
021      }
022    
023      public HCassandraInternalException(int type, String msg) {
024        super(ERR_MSG + "TApplicationError type: " + type + " message:" + msg);
025        this.type = type;
026      }
027    
028      public HCassandraInternalException(String s, Throwable t) {
029        super(ERR_MSG + s, t); 
030      }
031    
032      public HCassandraInternalException(Throwable t) {
033        super(t); 
034      }
035    
036      /**
037       * The underlying 'type' directly from TApplicationException#getType
038       * @return
039       */
040      public int getType() {
041        return type;
042      }
043     
044     
045    
046    }
