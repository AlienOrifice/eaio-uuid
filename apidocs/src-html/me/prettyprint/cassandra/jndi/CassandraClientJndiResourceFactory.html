001    package me.prettyprint.cassandra.jndi;
002    
003    import java.util.Hashtable;
004    
005    import javax.naming.Context;
006    import javax.naming.Name;
007    import javax.naming.RefAddr;
008    import javax.naming.Reference;
009    import javax.naming.spi.ObjectFactory;
010    
011    /**
012     * A factory for JNDI Resource managed objects. Responsible for creating a 
013     * {@link CassandraClientJndiResourcePool}. Relies on a supplied URL and 
014     * port which Cassandra is listening on.  These parameters are defined in 
015     * a web application's context.xml file.  For example:
016     * 
017     * <p>
018     * <pre>
019     *     <Resource name="cassandra/CassandraClientFactory"
020     *               auth="Container"
021     *               type="me.prettyprint.cassandra.jndi.CassandraClientJndiResourcePool"
022     *               factory="me.prettyprint.cassandra.jndi.CassandraClientJndiResourceFactory"
023     *               url="localhost"
024     *               port="9160" />      
025     * </pre>
026     *     
027     * @author Perry Hoekstra (dutchman_mn@charter.net)
028     * 
029     * @since 0.5.1-8
030     */
031    
032    public class CassandraClientJndiResourceFactory implements ObjectFactory 
033    {
034      /**
035       * Creates an object using the location or reference information specified. 
036       * 
037       * @param object       The possibly null object containing location or reference information that 
038       *                     can be used in creating an object.
039       * @param jndiName     The name of this object relative to nameCtx, or null if no name is 
040       *                     specified.
041       * @param context      The context relative to which the name parameter is specified, or null 
042       *                     if name is relative to the default initial context.
043       * @param environment  The possibly null environment that is used in creating the object. 
044       * 
045       * @return Object - The object created; null if an object cannot be created. 
046       * 
047       * @exception Exception - if this object factory encountered an exception while attempting 
048       *                        to create an object, and no other object factories are to be tried.
049       */
050      public Object getObjectInstance(Object object, Name jndiName, Context context,
051          Hashtable<?, ?> environment) throws Exception {  
052        Reference resourceRef = null;
053        
054        if (object instanceof Reference) {
055            resourceRef = (Reference) object;
056        } else {
057          throw new Exception("Object provided is not a javax.naming.Reference type");
058        }
059        
060        RefAddr urlRefAddr = resourceRef.get("url");
061        
062        RefAddr portRefAddr = resourceRef.get("port");
063      
064        if ((urlRefAddr != null) && (portRefAddr != null)) {
065          return new CassandraClientJndiResourcePool((String)urlRefAddr.getContent(), 
066                                                   Integer.parseInt((String)portRefAddr.getContent())); 
067        }  else {
068          throw new Exception("A url and port on which Cassandra is installed and listening " + 
069                          "on must be provided as a ResourceParams in the context.xml");
070        }
071      }
072    }
