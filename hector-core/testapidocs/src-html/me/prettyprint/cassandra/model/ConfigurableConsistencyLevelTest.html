001    package me.prettyprint.cassandra.model;
002    
003    import static org.junit.Assert.assertEquals;
004    
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    import me.prettyprint.cassandra.service.OperationType;
009    import me.prettyprint.hector.api.HConsistencyLevel;
010    
011    import org.junit.Before;
012    import org.junit.Test;
013    
014    public class ConfigurableConsistencyLevelTest {
015    
016      private ConfigurableConsistencyLevel configurableConsistencyLevel;
017      private Map<String, HConsistencyLevel> clmap;
018    
019      @Before
020      public void setup() {
021        configurableConsistencyLevel = new ConfigurableConsistencyLevel();
022        clmap = new HashMap<String, HConsistencyLevel>();
023        clmap.put("MyColumnFamily", HConsistencyLevel.ONE);
024        configurableConsistencyLevel.setReadCfConsistencyLevels(clmap);
025        configurableConsistencyLevel.setWriteCfConsistencyLevels(clmap);
026      }
027    
028      @Test
029      public void testReadWriteSame() {
030        assertEquals(HConsistencyLevel.ONE,
031            configurableConsistencyLevel.get(OperationType.READ, "MyColumnFamily"));
032      }
033    
034      @Test
035      public void testDefaults() {
036        configurableConsistencyLevel.setDefaultWriteConsistencyLevel(HConsistencyLevel.ALL);
037        configurableConsistencyLevel.setWriteCfConsistencyLevels(new HashMap<String, HConsistencyLevel>());
038        assertEquals(HConsistencyLevel.ALL,
039            configurableConsistencyLevel.get(OperationType.WRITE, "MyColumnFamily"));
040      }
041    
042      @Test
043      public void testSetRuntimeCl() {
044        configurableConsistencyLevel.setConsistencyLevelForCfOperation(HConsistencyLevel.ANY,
045            "OtherCf", OperationType.READ);
046        assertEquals(HConsistencyLevel.ANY,
047            configurableConsistencyLevel.get(OperationType.READ, "OtherCf"));
048      }
049    }
