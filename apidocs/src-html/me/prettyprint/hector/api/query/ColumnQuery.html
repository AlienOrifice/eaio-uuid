001    package me.prettyprint.hector.api.query;
002    
003    import me.prettyprint.hector.api.beans.HColumn;
004    
005    /**
006     * A ColumnQuery is used for querying the value of a single and standard column.
007     * <p>
008     * To read a value of a super column use SuperColumnQuery.
009     * To read a value of a sub-column within a super column use SubColumnQuery.
010     *
011     * @author Ran Tavory
012     *
013     * @param <N> Column name type.
014     * @param <V> Column value type.
015     */
016    public interface ColumnQuery<K, N, V> extends Query<HColumn<N, V>>{
017    
018      /**
019       * Set the row key for this query.
020       */
021      ColumnQuery<K, N, V> setKey(K key);
022    
023      /**
024       * Set the column name for this query.
025       */
026      ColumnQuery<K, N, V> setName(N name);
027    
028      ColumnQuery<K, N,V> setColumnFamily(String cf);
029    
030    }
