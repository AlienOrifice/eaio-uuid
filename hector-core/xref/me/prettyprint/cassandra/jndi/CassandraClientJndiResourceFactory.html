11  /**
12   * A factory for JNDI Resource managed objects. Responsible for creating a 
13   * {@link CassandraClientJndiResourcePool}. Relies on a supplied URL and 
14   * port which Cassandra is listening on.  These parameters are defined in 
15   * a web application's context.xml file.  For example:
16   * 
17   * <p>
18   * <pre>
19   *     <Resource name="cassandra/CassandraClientFactory"
20   *               auth="Container"
21   *               type="me.prettyprint.cassandra.jndi.CassandraClientJndiResourcePool"
22   *               factory="me.prettyprint.cassandra.jndi.CassandraClientJndiResourceFactory"
23   *               url="localhost"
24   *               port="9160" />      
25   * </pre>
26   *     
27   * @author Perry Hoekstra (dutchman_mn@charter.net)
28   * 
29   * @since 0.5.1-8
30   */
31  
32  public class CassandraClientJndiResourceFactory implements ObjectFactory 
33  {
34    /**
35     * Creates an object using the location or reference information specified. 
36     * 
37     * @param object       The possibly null object containing location or reference information that 
38     *                     can be used in creating an object.
39     * @param jndiName     The name of this object relative to nameCtx, or null if no name is 
40     *                     specified.
41     * @param context      The context relative to which the name parameter is specified, or null 
42     *                     if name is relative to the default initial context.
43     * @param environment  The possibly null environment that is used in creating the object. 
44     * 
45     * @return Object - The object created; null if an object cannot be created. 
46     * 
47     * @exception Exception - if this object factory encountered an exception while attempting 
48     *                        to create an object, and no other object factories are to be tried.
49     */
50    public Object getObjectInstance(Object object, Name jndiName, Context context,
51        Hashtable<?, ?> environment) throws Exception {  
52      Reference resourceRef = null;
53      
54      if (object instanceof Reference) {
55          resourceRef = (Reference) object;
56      } else {
57        throw new Exception("Object provided is not a javax.naming.Reference type");
58      }
59      
60      RefAddr urlRefAddr = resourceRef.get("url");
61      
62      RefAddr portRefAddr = resourceRef.get("port");
63    
64      if ((urlRefAddr != null) && (portRefAddr != null)) {
65        return new CassandraClientJndiResourcePool((String)urlRefAddr.getContent(), 
66                                                 Integer.parseInt((String)portRefAddr.getContent())); 
67      }  else {
68        throw new Exception("A url and port on which Cassandra is installed and listening " + 
69                        "on must be provided as a ResourceParams in the context.xml");
70      }
71    }
