60    public void registerMonitor(String name, String monitorType, Object monitoringInterface)
61        throws MalformedObjectNameException, InstanceAlreadyExistsException,
62        MBeanRegistrationException, NotCompliantMBeanException {
63  
64      String monitorName = generateMonitorName(name, monitorType);
65      log.info("Registering JMX {}", monitorName);
66  
67      ObjectName oName = new ObjectName(monitorName);
68  
69      // Check if the monitor is already registered
70      if (mbs.isRegistered(oName)) {
71        log.info("Monitor already registered: {}", oName);
72        return;
73      }
74  
75      mbs.registerMBean(monitoringInterface, oName);
76  
77      // Register perf4j monitors
78      if ("true".equalsIgnoreCase(
79          System.getProperty("com.prettyprint.cassandra.load_hector_log4j", "true"))) {
80        registerPerf4J();
81      }
82    }

84    private void registerPerf4J() {
85      URL url = getClass().getClassLoader().getResource("hectorLog4j.xml");
86      if (url == null) {
87        log.warn("Unable to locate hectorLog4j.xml; performance counters will not be exported");
88      } else {
89        try {
90          final Class<?> domConfiguratorClass = getClass().getClassLoader().loadClass("org.apache.log4j.xml.DOMConfigurator");
91          final Method method = domConfiguratorClass.getMethod( "configure", URL.class );
92          method.invoke( null, url );
93        } catch( ClassNotFoundException e ) {
94          log.warn("Unable to load log4j's DOMConfigurator. Performance counters will not be exported. To fix, include the log4j jar in your application's classpath.");
95        } catch( SecurityException e ) {
96          log.error( "Could not access method DOMConfigurator.configure(URL)", e );
97        } catch( NoSuchMethodException e ) {
98          log.error( "Could not find method DOMConfigurator.configure(URL)", e );
99        } catch( IllegalArgumentException e ) {
100         log.error( "Could not invoke method DOMConfigurator.configure(URL)", e );
101       } catch( IllegalAccessException e ) {
102         log.error( "Could not invoke method DOMConfigurator.configure(URL)", e );
103       } catch( InvocationTargetException e ) {
104         throw new RuntimeException(e.getCause());
105       }
106     }
107   }

109   private String generateMonitorName(String className, String monitorType) {
110     StringBuilder sb = new StringBuilder();
111     sb.append(className);
112     sb.append(":ServiceType=");
113     // append the classloader name so we have unique names in web apps.
114     sb.append(getUniqueClassloaderIdentifier());
115     if (null != monitorType && monitorType.length() > 0) {
116       sb.append(",MonitorType=" + monitorType);
117     }
118     return sb.toString();
119   }

121   /**
122    * Generates a unique, but still nice and predictable name representing this classloader so that
123    * even apps operating under a web server such as tomcat with multiple classloaders would bee able
124    * to register each with its own unique mbean.
125    */
126   private String getUniqueClassloaderIdentifier() {
127     String contextPath = getContextPath();
128     if (contextPath != null) {
129       return contextPath;
130     }
131     return "hector";
132   }

134   /**
135    * Tries to guess a context path for the running application.
136    * If this is a web application running under a tomcat server this will work.
137    * If unsuccessful, returns null.
138    * @return A string representing the current context path or null if it cannot be determined.
139    */
140   private String getContextPath() {
141     URL url = getClass().getClassLoader().getResource("/");
142     if (url != null) {
143       String[] elements = url.toString().split("/");
144       for (int i = elements.length - 1; i > 0; --i) {
145         // URLs look like this: file:/.../ImageServer/WEB-INF/classes/
146         // And we want that part that's just before WEB-INF
147         if ("WEB-INF".equals(elements[i])) {
148           return elements[i - 1];
149         }
150       }
151     }
152     return null;
153   }
