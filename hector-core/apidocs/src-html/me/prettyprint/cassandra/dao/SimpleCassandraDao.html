001    package me.prettyprint.cassandra.dao;
002    
003    import static me.prettyprint.hector.api.factory.HFactory.createColumn;
004    import static me.prettyprint.hector.api.factory.HFactory.createMultigetSliceQuery;
005    import static me.prettyprint.hector.api.factory.HFactory.createMutator;
006    
007    import java.util.HashMap;
008    import java.util.Map;
009    
010    import me.prettyprint.cassandra.serializers.StringSerializer;
011    import me.prettyprint.hector.api.Keyspace;
012    import me.prettyprint.hector.api.beans.HColumn;
013    import me.prettyprint.hector.api.beans.Rows;
014    import me.prettyprint.hector.api.exceptions.HectorException;
015    import me.prettyprint.hector.api.factory.HFactory;
016    import me.prettyprint.hector.api.mutation.Mutator;
017    import me.prettyprint.hector.api.query.ColumnQuery;
018    import me.prettyprint.hector.api.query.MultigetSliceQuery;
019    import me.prettyprint.hector.api.query.QueryResult;
020    
021    public class SimpleCassandraDao {
022    
023      private String columnFamilyName;
024      private Keyspace keyspace;
025      private final StringSerializer serializer = StringSerializer.get();
026    
027      /**
028       * Insert a new value keyed by key
029       *
030       * @param key   Key for the value
031       * @param value the String value to insert
032       */
033      public void insert(final String key, final String columnName, final String value) {
034        createMutator(keyspace, serializer).insert(
035            key, columnFamilyName, createColumn(columnName, value, serializer, serializer));
036    
037      }
038    
039      /**
040       * Get a string value.
041       *
042       * @return The string value; null if no value exists for the given key.
043       */
044      public String get(final String key, final String columnName) throws HectorException {
045        ColumnQuery<String, String, String> q = HFactory.createColumnQuery(keyspace,
046            serializer, serializer, serializer);
047        QueryResult<HColumn<String, String>> r = q.setKey(key).
048            setName(columnName).
049            setColumnFamily(columnFamilyName).
050            execute();
051        HColumn<String, String> c = r.get();
052        return c != null ? c.getValue() : null;
053      }
054    
055      /**
056       * Get multiple values
057       * @param keys
058       * @return
059       */
060      public Map<String, String> getMulti(String columnName, String... keys) {
061        MultigetSliceQuery<String, String,String> q = createMultigetSliceQuery(keyspace, serializer, serializer, serializer);
062        q.setColumnFamily(columnFamilyName);
063        q.setKeys(keys);
064        q.setColumnNames(columnName);
065    
066        QueryResult<Rows<String,String,String>> r = q.execute();
067        Rows<String,String,String> rows = r.get();
068        Map<String, String> ret = new HashMap<String, String>(keys.length);
069        for (String k: keys) {
070          HColumn<String, String> c = rows.getByKey(k).getColumnSlice().getColumnByName(columnName);
071          if (c != null && c.getValue() != null) {
072            ret.put(k, c.getValue());
073          }
074        }
075        return ret;
076      }
077    
078      /**
079       * Insert multiple values for a given columnName
080       */
081      public void insertMulti(String columnName, Map<String, String> keyValues) {
082        Mutator<String> m = createMutator(keyspace, serializer);
083        for (Map.Entry<String, String> keyValue: keyValues.entrySet()) {
084          m.addInsertion(keyValue.getKey(), columnFamilyName,
085              createColumn(columnName, keyValue.getValue(), keyspace.createClock(), serializer, serializer));
086        }
087        m.execute();
088      }
089    
090    
091      /**
092       * Delete multiple values
093       */
094      public void delete(String columnName, String... keys) {
095        Mutator<String> m = createMutator(keyspace, serializer);
096        for (String key: keys) {
097          m.addDeletion(key, columnFamilyName,  columnName, serializer);
098        }
099        m.execute();
100      }
101    
102      public void setColumnFamilyName(String columnFamilyName) {
103        this.columnFamilyName = columnFamilyName;
104      }
105    
106    
107      public void setKeyspace(Keyspace keyspace) {
108        this.keyspace = keyspace;
109      }
110    
111    }
