001    package me.prettyprint.cassandra.model;
002    
003    import java.util.HashMap;
004    import java.util.Iterator;
005    import java.util.List;
006    import java.util.Map;
007    
008    import me.prettyprint.cassandra.utils.Assert;
009    import me.prettyprint.hector.api.Serializer;
010    import me.prettyprint.hector.api.beans.SuperRow;
011    import me.prettyprint.hector.api.beans.SuperRows;
012    
013    import org.apache.cassandra.thrift.SuperColumn;
014    
015    /**
016     * Returned by a MultigetSuperSliceQuery (multiget_slice for supercolumns)
017     *
018     * @author Ran Tavory
019     *
020     * @param <N>
021     * @param <V>
022     */
023    public class SuperRowsImpl<K, SN, N, V> implements SuperRows<K, SN, N, V> {
024    
025      private final Map<K, SuperRow<K, SN, N, V>> rows;
026    
027      Serializer<K> keySerializer;
028    
029      public SuperRowsImpl(Map<K, List<SuperColumn>> thriftRet, Serializer<K> keySerializer,
030          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
031        Assert.noneNull(thriftRet, keySerializer, sNameSerializer, nameSerializer, valueSerializer);
032        this.keySerializer = keySerializer;
033        rows = new HashMap<K, SuperRow<K, SN, N, V>>(thriftRet.size());
034        for (Map.Entry<K, List<SuperColumn>> entry : thriftRet.entrySet()) {
035          rows.put(entry.getKey(), new SuperRowImpl<K, SN, N, V>(entry.getKey(), entry.getValue(),
036              sNameSerializer, nameSerializer, valueSerializer));
037        }
038      }
039    
040      @Override
041      public SuperRow<K, SN, N, V> getByKey(K key) {
042        return rows.get(key);
043      }
044    
045      @Override
046      public int getCount() {
047        return rows.size();
048      }
049    
050      @Override
051      public Iterator<SuperRow<K, SN, N, V>> iterator() {
052        return rows.values().iterator();
053      }
054    
055      @Override
056      public String toString() {
057        return "SuperRows(" + rows + ")";
058      }
059    }
