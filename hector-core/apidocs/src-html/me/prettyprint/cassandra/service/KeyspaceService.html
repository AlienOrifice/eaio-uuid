001    package me.prettyprint.cassandra.service;
002    
003    import java.nio.ByteBuffer;
004    import java.util.List;
005    import java.util.Map;
006    
007    import me.prettyprint.hector.api.HConsistencyLevel;
008    import me.prettyprint.hector.api.exceptions.HNotFoundException;
009    import me.prettyprint.hector.api.exceptions.HectorException;
010    
011    import org.apache.cassandra.thrift.Column;
012    import org.apache.cassandra.thrift.ColumnParent;
013    import org.apache.cassandra.thrift.ColumnPath;
014    import org.apache.cassandra.thrift.IndexClause;
015    import org.apache.cassandra.thrift.KeyRange;
016    import org.apache.cassandra.thrift.Mutation;
017    import org.apache.cassandra.thrift.SlicePredicate;
018    import org.apache.cassandra.thrift.SuperColumn;
019    
020    /**
021     * The keyspace is a high level handle to all read/write operations to cassandra.
022     *
023     * A Keyspace object is NOT THREAD SAFE. Use one keyspace per thread please!
024     *
025     * @author rantav
026     */
027    public interface KeyspaceService {
028    
029      public static String CF_TYPE = "Type" ;
030      public static String CF_TYPE_STANDARD = "Standard" ;
031      public static String CF_TYPE_SUPER = "Super" ;
032    
033    //  /**
034    //   * @return The cassandra client object used to obtain this KeySpace.
035    //   */
036    //  CassandraClient getClient();
037    
038      /**
039       * Get the Column at the given columnPath.
040       *
041       * If no value is present, NotFoundException is thrown.
042       *
043       * @throws HNotFoundException
044       *           if no value exists for the column
045       */
046      Column getColumn(ByteBuffer key, ColumnPath columnPath) throws HectorException;
047    
048      Column getColumn(String key, ColumnPath columnPath) throws HectorException;
049    
050      /**
051       * Get the SuperColumn at the given columnPath.
052       *
053       * If no value is present, NotFoundException is thrown.
054       *
055       * by default will return column with native order and the size of the list is
056       * unlimited (so be careful...)
057       *
058       * @throws HNotFoundException
059       *           when a supercolumn is not found
060       */
061      SuperColumn getSuperColumn(ByteBuffer key, ColumnPath columnPath) throws HectorException;
062    
063      SuperColumn getSuperColumn(String key, ColumnPath columnPath) throws HectorException;
064    
065      /**
066       * Get the SuperColumn at the given columnPath.
067       *
068       * If no value is present, NotFoundException is thrown.
069       *
070       * by default will return column with native order and the size of the list is
071       * unlimited (so be careful...)
072       *
073       * @param reversed
074       *          the result Column sort
075       * @param size
076       *          the result column size
077       * @throws HNotFoundException
078       *           when a supercolumn is not found
079       */
080      SuperColumn getSuperColumn(ByteBuffer key, ColumnPath columnPath, boolean reversed, int size)
081          throws HectorException;
082    
083      /**
084       * Get the group of columns contained by columnParent.
085       *
086       * Returns Either a ColumnFamily name or a ColumnFamily/SuperColumn specified
087       * by the given predicate. If no matching values are found, an empty list is
088       * returned.
089       */
090      List<Column> getSlice(ByteBuffer key, ColumnParent columnParent, SlicePredicate predicate)
091          throws HectorException;
092    
093      List<Column> getSlice(String key, ColumnParent columnParent, SlicePredicate predicate)
094      throws HectorException;
095    
096      /**
097       * Get the group of superColumn contained by columnParent.
098       */
099      List<SuperColumn> getSuperSlice(ByteBuffer key, ColumnParent columnParent,
100          SlicePredicate predicate) throws HectorException;
101    
102      List<SuperColumn> getSuperSlice(String key, ColumnParent columnParent,
103              SlicePredicate predicate) throws HectorException;
104    
105      /**
106       * Performs a get for columnPath in parallel on the given list of keys.
107       *
108       * The return value maps keys to the ColumnOrSuperColumn found. If no value
109       * corresponding to a key is present, the key will still be in the map, but
110       * both the column and superColumn references of the ColumnOrSuperColumn
111       * object it maps to will be null.
112       */
113      Map<ByteBuffer, SuperColumn> multigetSuperColumn(List<ByteBuffer> keys, ColumnPath columnPath)
114          throws HectorException;
115    
116      /**
117       * Perform a get for columnPath in parallel on the given list of keys.
118       *
119       * The return value maps keys to the ColumnOrSuperColumn found. If no value
120       * corresponding to a key is present, the key will still be in the map, but
121       * both the column and superColumn references of the ColumnOrSuperColumn
122       * object it maps to will be null.
123       */
124      Map<ByteBuffer, SuperColumn> multigetSuperColumn(List<ByteBuffer> keys, ColumnPath columnPath,
125          boolean reversed, int size) throws HectorException;
126    
127      /**
128       * Performs a get_slice for columnParent and predicate for the given keys in
129       * parallel.
130       */
131      Map<ByteBuffer, List<Column>> multigetSlice(List<ByteBuffer> keys, ColumnParent columnParent,
132          SlicePredicate predicate) throws HectorException;
133    
134      /**
135       * Performs a get_slice for a superColumn columnParent and predicate for the
136       * given keys in parallel.
137       */
138      Map<ByteBuffer, List<SuperColumn>> multigetSuperSlice(List<ByteBuffer> keys,
139          ColumnParent columnParent, SlicePredicate predicate) throws HectorException;
140    
141      /**
142       * Inserts a column.
143       */
144      void insert(ByteBuffer key, ColumnParent columnParent, Column column) throws HectorException;
145    
146      void insert(String key, ColumnPath columnPath, ByteBuffer value) throws HectorException;
147    
148      void insert(String key, ColumnPath columnPath, ByteBuffer value, long timestamp) throws HectorException;
149    
150      /**
151       * Call batch mutate with the assembled mutationMap. This method is a direct pass-through
152       * to the underlying Thrift API
153       */
154      void batchMutate(Map<ByteBuffer,Map<String,List<Mutation>>> mutationMap) throws HectorException;
155    
156      /**
157       * Call batch mutate with the BatchMutation object which encapsulates some of the complexity
158       * of the batch_mutate API signature
159       */
160      void batchMutate(BatchMutation batchMutation) throws HectorException;
161    
162      void remove(ByteBuffer key, ColumnPath columnPath);
163    
164    /**
165       * Same as two argument version, but the caller must specify their own clock
166       */
167      void remove(ByteBuffer key, ColumnPath columnPath, long timestamp) throws HectorException;
168    
169      void remove(String key, ColumnPath columnPath) throws HectorException;
170    
171      void remove(String key, ColumnPath columnPath, long timestamp) throws HectorException;
172    
173    
174      /**
175       * Counts the columns present in columnParent.
176       */
177      int getCount(ByteBuffer key, ColumnParent columnParent, SlicePredicate predicate) throws HectorException;
178    
179      /**
180       * returns a subset of columns for a range of keys.
181       */
182      Map<ByteBuffer, List<Column>> getRangeSlices(ColumnParent columnParent, SlicePredicate predicate,
183          KeyRange keyRange) throws HectorException;
184    
185      /**
186       * returns a subset of super columns for a range of keys.
187       */
188      Map<ByteBuffer, List<SuperColumn>> getSuperRangeSlices(ColumnParent columnParent, SlicePredicate predicate,
189          KeyRange keyRange) throws HectorException;
190    
191      /**
192       * returns a subset of columns for a range of keys.
193       */
194      Map<ByteBuffer, List<Column>> getIndexedSlices(ColumnParent columnParent, IndexClause indexClause,
195          SlicePredicate predicate) throws HectorException;
196    
197      /**
198       * Returns a map of key to column count
199       */
200      Map<ByteBuffer, Integer> multigetCount(List<ByteBuffer> keys, ColumnParent columnParent,
201          SlicePredicate slicePredicate) throws HectorException;
202    
203      /**
204       * @return The consistency level held by this keyspace instance.
205       */
206      HConsistencyLevel getConsistencyLevel(OperationType operationType);
207    
208      String getName();
209    
210      CassandraHost getCassandraHost();
211    }
