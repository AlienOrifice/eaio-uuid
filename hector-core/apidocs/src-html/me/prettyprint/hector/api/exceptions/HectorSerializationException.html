001    package me.prettyprint.hector.api.exceptions;
002    
003    import me.prettyprint.cassandra.serializers.ObjectSerializer;
004    import me.prettyprint.hector.api.Serializer;
005    
006    /**
007     * Describes a serialization exception.
008     * Serialization exceptions may happen when {@link Serializer#fromBytes(byte[])} encounters a value
009     * it's unable to interpret to it's target object. For example an {@link ObjectSerializer} may be
010     * unable to deserialize a java object back to its class if the object it of the wrong format
011     * or the target class is not on the classpath.
012     * @author Ran Tavory
013     *
014     */
015    public class HectorSerializationException extends HectorException {
016    
017      private static final long serialVersionUID = -8704123219107599786L;
018    
019      public HectorSerializationException(String msg) {
020        super(msg);
021      }
022    
023      public HectorSerializationException(Throwable t) {
024        super(t);
025      }
026    
027      public HectorSerializationException(String msg, Throwable t) {
028        super(msg, t);
029      }
030    
031    
032    }
