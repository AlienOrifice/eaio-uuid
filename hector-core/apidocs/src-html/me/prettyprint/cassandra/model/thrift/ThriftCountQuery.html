001    package me.prettyprint.cassandra.model.thrift;
002    
003    import me.prettyprint.hector.api.Keyspace;
004    import me.prettyprint.hector.api.Serializer;
005    import me.prettyprint.hector.api.query.CountQuery;
006    import me.prettyprint.hector.api.query.QueryResult;
007    
008    
009    /**
010     * Counts column for a standard column family
011     *
012     * @author Ran Tavory
013     */
014    public final class ThriftCountQuery<K, N> extends AbstractThriftCountQuery<K, N> implements
015        CountQuery<K, N> {
016    
017      public ThriftCountQuery(Keyspace k, Serializer<K> keySerializer,
018          Serializer<N> nameSerializer) {
019        super(k, keySerializer, nameSerializer);
020      }
021    
022      @Override
023      public QueryResult<Integer> execute() {
024        return countColumns();
025      }
026    
027      @Override
028      public String toString() {
029        return "CountQuery(" + columnFamily + "," + key + ")";
030      }
031    
032      @SuppressWarnings("unchecked")
033      @Override
034      public ThriftCountQuery<K, N> setKey(K key) {
035        return (ThriftCountQuery<K, N>) super.setKey(key);
036      }
037    
038      @SuppressWarnings("unchecked")
039      @Override
040      public ThriftCountQuery<K, N> setColumnFamily(String cf) {
041        return (ThriftCountQuery<K, N>) super.setColumnFamily(cf);
042      }
043    
044      @SuppressWarnings("unchecked")
045      @Override
046      public CountQuery<K, N> setRange(N start, N finish, int count) {
047        return (CountQuery<K, N>) super.setRange(start, finish, count);
048      }
049    
050      @SuppressWarnings("unchecked")
051      @Override
052      public CountQuery<K, N> setColumnNames(N... columnNames) {
053        return (CountQuery<K, N>) super.setColumnNames(columnNames);
054      }
055    }
