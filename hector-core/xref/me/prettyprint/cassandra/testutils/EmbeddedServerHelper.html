38    /**
39     * Set embedded cassandra up and spawn it in a new thread.
40     *
41     * @throws TTransportException
42     * @throws IOException
43     * @throws InterruptedException
44     */
45    public void setup() throws TTransportException, IOException, InterruptedException, ConfigurationException {
46      // delete tmp dir first
47      rmdir(TMP);
48      // make a tmp dir and copy cassandra.yaml and log4j.properties to it
49      copy("/log4j.properties", TMP);
50      copy(yamlFile, TMP);
51      System.setProperty("cassandra.config", "file:"+ TMP + yamlFile);
52  
53      CassandraServiceDataCleaner cleaner = new CassandraServiceDataCleaner();
54      cleaner.prepare();
55  
56      loadYamlTables();
57  
58      cassandra = new EmbeddedCassandraService();
59      cassandra.init();
60      Thread t = new Thread(cassandra);
61      t.setDaemon(true);
62      t.start();
63    }
64  
65    /** Manually load tables from the test configuration file.
66     * @throws ConfigurationException */
67    private void loadYamlTables() throws ConfigurationException {
68      for (KSMetaData table : DatabaseDescriptor.readTablesFromYaml()) {
69        for (CFMetaData cfm : table.cfMetaData().values()) {
70          CFMetaData.map(cfm);
71        }
72        DatabaseDescriptor.setTableDefinition(table, DatabaseDescriptor.getDefsVersion());
73      }
74    }
75  
76    public void teardown() {
77      CassandraServiceDataCleaner cleaner = new CassandraServiceDataCleaner();
78  
79      try {
80        cleaner.cleanupDataDirectories();
81        rmdir(TMP);
82      } catch (IOException e) {
83        // IGNORE
84      }
85    }

87    private static void rmdir(String dir) throws IOException {
88      File dirFile = new File(dir);
89      if (dirFile.exists()) {
90        FileUtils.deleteRecursive(new File(dir));
91      }
92    }
93    /**
94     * Copies a resource from within the jar to a directory.
95     *
96     * @param resource
97     * @param directory
98     * @throws IOException
99     */
100   private static void copy(String resource, String directory) throws IOException {
101     mkdir(directory);
102     InputStream is = EmbeddedServerHelper.class.getResourceAsStream(resource);
103     String fileName = resource.substring(resource.lastIndexOf("/") + 1);
104     File file = new File(directory + System.getProperty("file.separator") + fileName);
105     OutputStream out = new FileOutputStream(file);
106     byte buf[] = new byte[1024];
107     int len;
108     while ((len = is.read(buf)) > 0) {
109       out.write(buf, 0, len);
110     }
111     out.close();
112     is.close();
113   }
