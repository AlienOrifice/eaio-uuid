001    package me.prettyprint.cassandra.model;
002    
003    import me.prettyprint.cassandra.service.KeyspaceService;
004    import me.prettyprint.hector.api.exceptions.HectorException;
005    
006    /**
007     * A callback template used by the package classes. Not for external use.
008     *
009     * @author Ran Tavory
010     *
011     * @param <T>
012     */
013    public abstract class KeyspaceOperationCallback<T> {
014      // TODO remove
015      public abstract T doInKeyspace(final KeyspaceService ks) throws HectorException;
016    
017      public ExecutionResult<T> doInKeyspaceAndMeasure(final KeyspaceService ks) {
018        long start = System.nanoTime();
019        T value = null;
020        value = doInKeyspace(ks);
021    
022        return new ExecutionResult<T>(value, System.nanoTime() - start, ks.getCassandraHost());
023      }
024    
025    }
