21    public ColumnSliceImpl(List<Column> tColumns, Serializer<N> nameSerializer,
22        Serializer<V> valueSerializer) {
23      Assert.noneNull(tColumns, nameSerializer, valueSerializer);
24      columnsMap = new HashMap<N,HColumn<N,V>>(tColumns.size());
25      List<HColumn<N,V>> list = new ArrayList<HColumn<N,V>>(tColumns.size());
26      for (Column c: tColumns) {
27        HColumn<N, V> column = new HColumnImpl<N,V>(c, nameSerializer, valueSerializer);
28        columnsMap.put(column.getName(), column);
29        list.add(column);
30      }
31      columnsList = Collections.unmodifiableList(list);
32    }
33

41    }
