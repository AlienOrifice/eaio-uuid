001    package me.prettyprint.cassandra.service;
002    
003    /**
004     * System properties used by Hector.
005     *
006     * @author Ran Tavory (rantav@gmail.com)
007     *
008     */
009    public enum SystemProperties {
010    
011      /**
012       * Should hector perform name resolution?
013       * Default: no; Do not perform name resolution
014       * For example: -DHECTOR_PERFORM_NAME_RESOLUTION=true
015       */
016      HECTOR_PERFORM_NAME_RESOLUTION,
017    
018      /**
019       * What's the default socket timeout for thrift in miliseconds.
020       * Default: system settings (about a minute or more)
021       * Example: -DCASSANDRA_THRIFT_SOCKET_TIMEOUT=5000
022       */
023      CASSANDRA_THRIFT_SOCKET_TIMEOUT,
024    }
