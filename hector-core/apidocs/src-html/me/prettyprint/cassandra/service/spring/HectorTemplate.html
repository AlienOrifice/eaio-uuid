028    /**
029     * The main interface used to operate with the underlying database.
030     *
031     * The interface is mirroring HFactory.
032     *
033     * @author Bozhidar Bozhanov
034     *
035     */
036    public interface HectorTemplate {
037    
038      <K, N, V> Mutator<K> createMutator(Serializer<K> keySerializer);
039    
040      <K, N, V> Mutator<K> createMutator();
041    
042      <K, N, V> ColumnQuery<K, N, V> createColumnQuery();
043    
044      <K, N, V> ColumnQuery<K, N, V> createColumnQuery(Serializer<V> valueSErializer);
045    
046      <K, N> CountQuery<K, N> createCountQuery(Serializer<K> keySerializer, Serializer<N> nameSerializer);
047    
048      <K, SN> SuperCountQuery<K, SN> createSuperCountQuery(Serializer<K> keySerializer,
049          Serializer<SN> superNameSerializer);
050    
051      <K, SN, N> SubCountQuery<K, SN, N> createSubCountQuery(Serializer<K> keySerializer,
052          Serializer<SN> superNameSerializer, Serializer<N> nameSerializer);
053    
054      <K, SN, N, V> SuperColumnQuery<K, SN, N, V> createSuperColumnQuery(Serializer<K> keySerializer,
055          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer);
056    
057      <K, N, V> MultigetSliceQuery<K, N, V> createMultigetSliceQuery(Serializer<K> keySerializer,
058          Serializer<N> nameSerializer, Serializer<V> valueSerializer);
059    
060      <K, SN, N, V> MultigetSuperSliceQuery<K, SN, N, V> createMultigetSuperSliceQuery(
061          Serializer<K> keySerializer, Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
062          Serializer<V> valueSerializer);
063    
064      <K, SN, N, V> MultigetSubSliceQuery<K, SN, N, V> createMultigetSubSliceQuery(
065          Serializer<K> keySerializer, Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
066          Serializer<V> valueSerializer);
067    
068      <K, N, V> RangeSlicesQuery<K, N, V> createRangeSlicesQuery(Serializer<K> keySerializer,
069          Serializer<N> nameSerializer, Serializer<V> valueSerializer);
070    
071      <K, SN, N, V> RangeSuperSlicesQuery<K, SN, N, V> createRangeSuperSlicesQuery(
072          Serializer<K> keySerializer, Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
073          Serializer<V> valueSerializer);
074    
075      <K, SN, N, V> RangeSubSlicesQuery<K, SN, N, V> createRangeSubSlicesQuery(
076          Serializer<K> keySerializer, Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
077          Serializer<V> valueSerializer);
078    
079      <K, N, V> SliceQuery<K, N, V> createSliceQuery(Serializer<K> keySerializer,
080          Serializer<N> nameSerializer, Serializer<V> valueSerializer);
081    
082      <K> SliceQuery<K, ByteBuffer, ByteBuffer> createSliceQuery();
083    
084      <K, SN, N, V> SubSliceQuery<K, SN, N, V> createSubSliceQuery(Serializer<K> keySerializer,
085          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer);
086    
087      <K, SN, N, V> SuperSliceQuery<K, SN, N, V> createSuperSliceQuery(Serializer<K> keySerializer,
088          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer);
089    
090      <K> SuperSliceQuery<K, ByteBuffer, ByteBuffer, ByteBuffer> createSuperSliceQuery();
091    
092      <K, N, V> IndexedSlicesQuery<K, N, V> createIndexSlicesQuery(Serializer<K> keySerializer,
093          Serializer<N> nameSerializer, Serializer<V> valueSerializer);
094    
095      /**
096       * createSuperColumn accepts a variable number of column arguments
097       *
098       * @param name
099       *          supercolumn name
100       * @param createColumn
101       *          a variable number of column arguments
102       */
103      <SN, N, V> HSuperColumn<SN, N, V> createSuperColumn(SN name, List<HColumn<N, V>> columns,
104          Serializer<SN> superNameSerializer, Serializer<N> nameSerializer,
105          Serializer<V> valueSerializer);
106    
107      <SN, N, V> HSuperColumn<SN, N, V> createSuperColumn(SN name, List<HColumn<N, V>> columns);
108    
109      <SN, N, V> HSuperColumn<SN, N, V> createSuperColumn(SN name, List<HColumn<N, V>> columns,
110          long clock, Serializer<SN> superNameSerializer, Serializer<N> nameSerializer,
111          Serializer<V> valueSerializer);
112    
113      <N, V> HColumn<N, V> createColumn(N name, V value, long clock, Serializer<N> nameSerializer,
114          Serializer<V> valueSerializer);
115    
116      /**
117       * Creates a column with the clock of now.
118       */
119      <N, V> HColumn<N, V> createColumn(N name, V value);
120      
121      /**
122       * Creates a column with the specified <code>name/value</code> and <code>clock</code>.
123       */
124      <N, V> HColumn<N, V> createColumn(N name, V value, long clock);
125    
126      /**
127       * Creates a clock of now with the default clock resolution (micorosec) as defined in
128       * {@link CassandraHost}
129       */
130      long createClock();
131    
132      Cluster getCluster();
133    
134      String getReplicationStrategyClass();
135    
136      int getReplicationFactor();
137    
138      String getKeyspaceName();
139    }
