001    package me.prettyprint.cassandra.model;
002    
003    import java.util.Collection;
004    
005    import me.prettyprint.hector.api.Keyspace;
006    import me.prettyprint.hector.api.Serializer;
007    import me.prettyprint.hector.api.query.Query;
008    
009    import org.apache.cassandra.thrift.SlicePredicate;
010    
011    
012    /**
013     * Base type for SliceQuery, RangeSliceQuery and MultigetSliceQuery
014     * @author Ran Tavory
015     *
016     * @param <N>
017     * @param <T>
018     */
019    public abstract class AbstractSliceQuery<K,N,V,T> extends AbstractQuery<K,N,V,T> implements Query<T> {
020    
021      protected final HSlicePredicate<N> slicePredicate;
022    
023      public AbstractSliceQuery(Keyspace k, Serializer<K> keySerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
024        super(k, keySerializer, nameSerializer, valueSerializer);
025        slicePredicate = new HSlicePredicate<N>(nameSerializer);
026      }
027    
028      /**
029       * Sets the column names to be retrieved by this query
030       * @param columns a varargs of column names
031       */
032      public Query<T> setColumnNames(N... columnNames) {
033        if ( columnNames != null && columnNames.length > 0) {
034          slicePredicate.setColumnNames(columnNames);
035        } else {
036          slicePredicate.setKeysOnlyPredicate();
037        }
038        return this;
039      }
040    
041      /**
042       * Sets the column names to be retrieved by this query
043       * @param columns a Collection of column names
044       */
045      public Query<T> setColumnNames(Collection<N> columnNames) {
046        if ( columnNames != null && columnNames.size() > 0) {
047          slicePredicate.setColumnNames(columnNames);
048        } else {
049          slicePredicate.setKeysOnlyPredicate();
050        }
051        return this;
052      }
053      
054      /**
055       * Wraps the underlying call to {@link HSlicePredicate#setKeysOnlyPredicate()}
056       * Use this for a substantial performance increase when you only need the keys returned
057       */
058      public AbstractSliceQuery<K,N,V,T> setReturnKeysOnly() {
059        slicePredicate.setKeysOnlyPredicate();
060        return this;
061      }
062    
063    
064      public Collection<N> getColumnNames() {
065        return slicePredicate.getColumnNames();
066      }
067    
068      /**
069       * Set a predicate of start/finish to retrieve a list of columns in this range.
070       *
071       * @param start Start key
072       * @param finish End key
073       * @param reversed
074       * @param count
075       * @return
076       */
077      public Query<T> setRange(N start, N finish, boolean reversed, int count) {
078        slicePredicate.setRange(start, finish, reversed, count);
079        return this;
080      }
081    
082      /**
083       *
084       * @return the thrift representation of the predicate
085       */
086      public SlicePredicate getPredicate() {
087        return slicePredicate.toThrift();
088      }
089    
090      protected String toStringInternal() {
091        return slicePredicate.toString();
092      }
093    }
