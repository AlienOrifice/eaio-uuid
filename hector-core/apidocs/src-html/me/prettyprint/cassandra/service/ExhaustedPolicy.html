001    package me.prettyprint.cassandra.service;
002    
003    import me.prettyprint.hector.api.exceptions.PoolExhaustedException;
004    
005    /**
006     * Policy what to do when the connection pool is exhausted.
007     * @author Ran Tavory (rantav@gmail.com)
008     *
009     */
010    public enum ExhaustedPolicy {
011      /**
012       * If the pool is full, fail with the exception {@link PoolExhaustedException}
013       */
014      WHEN_EXHAUSTED_FAIL, 
015      /**
016       * When pool exhausted, grow.
017       */
018      WHEN_EXHAUSTED_GROW, 
019      /**
020       * Block the requesting thread when the pool is exhausted until new connections are available.
021       */
022      WHEN_EXHAUSTED_BLOCK
023    }
