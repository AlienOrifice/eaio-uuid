004    import me.prettyprint.cassandra.model.AbstractSuperColumnQuery;

017    import org.apache.cassandra.thrift.ColumnPath;
018    import org.apache.cassandra.thrift.SuperColumn;

020    /**
021     * Thrift implementation of the SuperColumnQuery
022     *
023     * @author Ran Tavory
024     *
025     * @param <SN>
026     * @param <N>
027     * @param <V>
028     */
029    public final class ThriftSuperColumnQuery<K, SN,N,V> extends AbstractSuperColumnQuery<K, SN, N, V>
030        implements SuperColumnQuery<K, SN, N, V> {
031    
032      /*package*/ public ThriftSuperColumnQuery(Keyspace keyspace,
033          Serializer<K> keySerializer,
034          Serializer<SN> sNameSerializer,
035          Serializer<N> nameSerializer,
036          Serializer<V> valueSerializer) {
037        super(keyspace, keySerializer, sNameSerializer, nameSerializer, valueSerializer);
038      }
039    
040      @Override
041      public QueryResult<HSuperColumn<SN, N, V>> execute() {
042        notNull(columnFamilyName, "columnFamilyName is null");
043        notNull(superName, "superName is null");
044        return new QueryResultImpl<HSuperColumn<SN, N, V>>(keyspace.doExecute(
045            new KeyspaceOperationCallback<HSuperColumn<SN, N, V>>() {
046              @Override
047              public HSuperColumn<SN, N, V> doInKeyspace(KeyspaceService ks) throws HectorException {
048                try {
049                  ColumnPath cpath = ThriftFactory.createSuperColumnPath(columnFamilyName, superName, (N) null,
050                      sNameSerializer, columnNameSerializer);
051                  SuperColumn thriftSuperColumn = ks.getSuperColumn(keySerializer.toByteBuffer(key), cpath);
052                  if (thriftSuperColumn == null) {
053                    return null;
054                  }
055                  return new HSuperColumnImpl<SN, N, V>(thriftSuperColumn, sNameSerializer, columnNameSerializer,
056                      valueSerializer);
057                } catch (HNotFoundException e) {
058                  return null;
059                }
060              }
061            }), this);
062      }
063    }
