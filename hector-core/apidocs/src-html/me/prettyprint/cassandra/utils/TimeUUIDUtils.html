001    package me.prettyprint.cassandra.utils;
002    
003    import java.nio.ByteBuffer;
004    import java.util.UUID;
005    
006    import com.eaio.uuid.UUIDGen;
007    
008    import me.prettyprint.cassandra.service.clock.MicrosecondsSyncClockResolution;
009    import me.prettyprint.hector.api.ClockResolution;
010    
011    /**
012     * Utilitary class to generate TimeUUID (type 1)
013     *
014     * @author Patricio Echague (pechague@gmail.com)
015     *
016     */
017    public final class TimeUUIDUtils {
018    
019      /**
020       * Gets a new and unique time uuid in milliseconds. It is useful to use in a TimeUUIDType sorted column family.
021       *
022       * @return the time uuid
023       */
024      public static java.util.UUID getUniqueTimeUUIDinMillis() {
025        return new java.util.UUID(UUIDGen.newTime(), UUIDGen.getClockSeqAndNode());
026      }
027    
028      /**
029       * Gets a new time uuid using {@link ClockResolution#createClock()} as a time generator.
030       * It is useful to use in a TimeUUIDType sorted column family.
031       *
032       * @param clock a ClockResolution
033       * @return the time uuid
034       */
035      public static java.util.UUID getTimeUUID(ClockResolution clock) {
036        return getTimeUUID(clock.createClock());
037      }
038    
039      /**
040       * Gets a new time uuid based on <code>time<code>.
041       * NOTE: this algorithm does not resolve duplicates. To avoid duplicates use
042       * {@link getTimeUUID(ClockResolution clock)} with an implementaion that provides unique timestamp resolution, like
043       * {@link MicrosecondsSyncClockResolution}
044       * It is useful to use in a TimeUUIDType sorted column family.
045       *
046       * @param clock a ClockResolution
047       * @return the time uuid
048       */
049      public static java.util.UUID getTimeUUID(long time) {
050        return new java.util.UUID(createTime(time), UUIDGen.getClockSeqAndNode());
051      }
052    
053      private static long createTime(long currentTime) {
054        long time;
055    
056        // UTC time
057        long timeToUse = (currentTime * 10000) + 0x01B21DD213814000L;
058    
059        // time low
060        time = timeToUse << 32;
061    
062        // time mid
063        time |= (timeToUse & 0xFFFF00000000L) >> 16;
064    
065        // time hi and version
066        time |= 0x1000 | ((timeToUse >> 48) & 0x0FFF); // version 1
067        return time;
068      }
069    
070    
071      /**
072       * Returns an instance of uuid. Useful for when you read out of cassandra
073       * you are getting a byte[] that needs to be converted into a TimeUUID.
074       *
075       * @param uuid the uuid
076       * @return the java.util.uuid
077       */
078      public static java.util.UUID toUUID(byte[] uuid) {
079        return uuid(uuid, 0);
080      }
081    
082      /**
083       * Retrieves the time as long based on the byte[] representation of a UUID.
084       *
085       * @param uuid byte[] uuid representation
086       * @return a long representing the time
087       */
088      public static long getTimeFromUUID(byte[] uuid) {
089        return TimeUUIDUtils.toUUID(uuid).timestamp();
090      }
091    
092      /**
093       * As byte array.
094       * This method is often used in conjunction with @link {@link #getTimeUUID()}
095       *
096       * @param uuid the uuid
097       *
098       * @return the byte[]
099       */
100      public static byte[] asByteArray(java.util.UUID uuid) {
101        long msb = uuid.getMostSignificantBits();
102        long lsb = uuid.getLeastSignificantBits();
103        byte[] buffer = new byte[16];
104    
105        for (int i = 0; i < 8; i++) {
106          buffer[i] = (byte) (msb >>> 8 * (7 - i));
107        }
108    
109        for (int i = 8; i < 16; i++) {
110          buffer[i] = (byte) (lsb >>> 8 * (7 - i));
111        }
112    
113        return buffer;
114      }
115    
116      /**
117       * Coverts a java.util.UUID into a ByteBuffer.
118       * @param uuid a java.util.UUID
119       * @return a ByteBuffer representaion of the param UUID
120       */
121      public static ByteBuffer asByteBuffer(java.util.UUID uuid) {
122        if (uuid == null) {
123          return null;
124        }
125    
126        return ByteBuffer.wrap(asByteArray(uuid));
127      }
128    
129    
130      public static UUID uuid(byte[] uuid, int offset) {
131        ByteBuffer bb = ByteBuffer.wrap(uuid, offset, 16);
132        return new UUID(bb.getLong(), bb.getLong());
133      }
134    
135      /**
136       * Converts a ByteBuffer containing a UUID into a java.util.UUID
137       * @param bb a ByteBuffer containing a UUID
138       * @return a java.util.UUID
139       */
140      public static UUID uuid(ByteBuffer bb) {
141        bb = bb.slice();
142        return new UUID(bb.getLong(), bb.getLong());
143      }
144    
145    }
