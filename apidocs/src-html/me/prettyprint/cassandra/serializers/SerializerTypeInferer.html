001    package me.prettyprint.cassandra.serializers;
002    
003    import java.util.Date;
004    import java.util.UUID;
005    
006    import me.prettyprint.hector.api.Serializer;
007    
008    /**
009     * Utility class that infers the concrete Serializer needed to turn a value into
010     * its binary representation
011     *
012     * @author Bozhidar Bozhanov
013     *
014     */
015    public class SerializerTypeInferer {
016    
017      @SuppressWarnings({ "rawtypes", "unchecked" })
018      public static <T> Serializer<T> getSerializer(Object value) {
019        Serializer serializer = null;
020        if (value == null) {
021          serializer = ByteBufferSerializer.get();
022        } else if (value instanceof UUID) {
023          serializer = UUIDSerializer.get();
024        } else if (value instanceof String) {
025          serializer = StringSerializer.get();
026        } else if (value instanceof Long) {
027          serializer = LongSerializer.get();
028        } else if (value instanceof Integer) {
029          serializer = IntegerSerializer.get();
030        } else if (value instanceof Boolean) {
031          serializer = BooleanSerializer.get();
032        } else if (value instanceof byte[]) {
033          serializer = BytesArraySerializer.get();
034        } else if (value instanceof Date) {
035          serializer = DateSerializer.get();
036        } else {
037          serializer = ObjectSerializer.get();
038        }
039        // Add other serializers here
040    
041        return serializer;
042      }
043    
044      @SuppressWarnings({ "rawtypes", "unchecked" })
045      public static <T> Serializer<T> getSerializer(Class<?> valueClass) {
046        Serializer serializer = null;
047        if (valueClass.equals(UUID.class)) {
048          serializer = UUIDSerializer.get();
049        } else if (valueClass.equals(String.class)) {
050          serializer = StringSerializer.get();
051        } else if (valueClass.equals(Long.class) || valueClass.equals(long.class)) {
052          serializer = LongSerializer.get();
053        } else if (valueClass.equals(Integer.class) || valueClass.equals(int.class)) {
054          serializer = IntegerSerializer.get();
055        } else if (valueClass.equals(Boolean.class) || valueClass.equals(boolean.class)) {
056          serializer = BooleanSerializer.get();
057        } else if (valueClass.equals(byte[].class)) {
058          serializer = ByteBufferSerializer.get();
059        } else {
060          serializer = ObjectSerializer.get();
061        }
062        // Add other serializers here
063    
064        return serializer;
065      }
066    }
