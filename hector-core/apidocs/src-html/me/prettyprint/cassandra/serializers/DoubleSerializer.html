001    package me.prettyprint.cassandra.serializers;
002    
003    import java.nio.ByteBuffer;
004    
005    import me.prettyprint.cassandra.serializers.AbstractSerializer;
006    import me.prettyprint.cassandra.serializers.LongSerializer;
007    
008    /**
009     * Uses LongSerializer via translating Doubles to and from raw long bytes form.
010     * 
011     * @author Yuri Finkelstein 
012     */
013    public class DoubleSerializer extends AbstractSerializer<Double> {
014    
015      private static final DoubleSerializer instance = new DoubleSerializer();
016    
017      public static DoubleSerializer get() {
018        return instance;
019      }
020      
021      @Override
022      public ByteBuffer toByteBuffer(Double obj) {
023        return LongSerializer.get().toByteBuffer(Double.doubleToRawLongBits(obj));
024      }
025    
026      @Override
027      public Double fromByteBuffer(ByteBuffer bytes) {
028        return Double.longBitsToDouble (LongSerializer.get().fromByteBuffer(bytes));
029      }
030    
031    }
