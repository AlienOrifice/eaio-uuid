028    /**
029     * Uses new secondary indexes. Your CF must be configured for such to use this.
030     * The following creates an Indexed CF with the "birthday" column indexed
031     * (where birthdate represents a timestamp as it is validated by the LongType):
032     * <pre>
033     *         - name: Indexed1
034     *           column_metadata:
035     *             - name: birthdate
036     *               validator_class: LongType
037     *               index_type: KEYS
038     *</pre>
039     *
040     * @author zznate (nate@riptano.com)
041     */
042    public class IndexedSlicesQuery<K,N,V> extends AbstractSliceQuery<K,N,V,OrderedRows<K,N,V>> {
043    
044      private final IndexClause indexClause;
045    
046      public IndexedSlicesQuery(Keyspace k, Serializer<K> keySerializer,
047          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
048        super(k, keySerializer, nameSerializer, valueSerializer);
049        indexClause = new IndexClause();
050      }
051    
052      public IndexedSlicesQuery<K,N,V> addEqualsExpression(N columnName, V columnValue) {
053        indexClause.addToExpressions(new IndexExpression(columnNameSerializer.toByteBuffer(columnName),
054            IndexOperator.EQ,
055            valueSerializer.toByteBuffer(columnValue)));
056        return this;
057      }
058    
059      public IndexedSlicesQuery<K,N,V> addLteExpression(N columnName, V columnValue) {
060        indexClause.addToExpressions(new IndexExpression(columnNameSerializer.toByteBuffer(columnName),
061            IndexOperator.LTE,
062            valueSerializer.toByteBuffer(columnValue)));
063        return this;
064      }
065    
066      public IndexedSlicesQuery<K,N,V> addGteExpression(N columnName, V columnValue) {
067        indexClause.addToExpressions(new IndexExpression(columnNameSerializer.toByteBuffer(columnName),
068            IndexOperator.GTE,
069            valueSerializer.toByteBuffer(columnValue)));
070        return this;
071      }
072    
073      public IndexedSlicesQuery<K,N,V> addLtExpression(N columnName, V columnValue) {
074        indexClause.addToExpressions(new IndexExpression(columnNameSerializer.toByteBuffer(columnName),
075            IndexOperator.LT,
076            valueSerializer.toByteBuffer(columnValue)));
077        return this;
078      }
079    
080      public IndexedSlicesQuery<K,N,V> addGtExpression(N columnName, V columnValue) {
081        indexClause.addToExpressions(new IndexExpression(columnNameSerializer.toByteBuffer(columnName),
082            IndexOperator.GT,
083            valueSerializer.toByteBuffer(columnValue)));
084        return this;
085      }
086    
087      @Override
088      public IndexedSlicesQuery<K, N, V> setColumnNames(Collection<N> columnNames) {
089        super.setColumnNames(columnNames);
090        return this;
091      }
092    
093      @Override
094      public IndexedSlicesQuery<K, N, V> setColumnNames(N... columnNames) {
095        super.setColumnNames(columnNames);
096        return this;
097      }
098    
099      @Override
100      public IndexedSlicesQuery<K, N, V> setRange(N start, N finish,
101          boolean reversed, int count) {
102        super.setRange(start, finish, reversed, count);
103        return this;
104      }
105    
106      @Override
107      public IndexedSlicesQuery<K, N, V> setReturnKeysOnly() {
108        super.setReturnKeysOnly();
109        return this;
110      }
111    
112      public IndexedSlicesQuery<K,N,V> setStartKey(K startKey) {
113        indexClause.setStart_key(keySerializer.toByteBuffer(startKey));
114        return this;
115      }
116    
117      @Override
118      public IndexedSlicesQuery<K,N,V> setColumnFamily(String cf) {
119        super.setColumnFamily(cf);
120        return this;
121      }
122      
123      public IndexedSlicesQuery<K,N,V> setRowCount(int rowCount) {
124        indexClause.setCount(rowCount);
125        return this;
126      }
127    
128      @Override
129      public QueryResult<OrderedRows<K,N, V>> execute() {
130    
131        return new QueryResultImpl<OrderedRows<K,N,V>>(keyspace.doExecuteOperation(
132            new Operation <OrderedRows<K,N,V>>(OperationType.READ) {
133              @Override
134              public OrderedRows<K,N,V> execute(Cassandra.Client client) throws HectorException {
135                LinkedHashMap<ByteBuffer, List<Column>> ret = null;
136                try {
137                  if (!indexClause.isSetStart_key()) {
138                    indexClause.setStart_key(new byte[0]);
139                  }
140                  ColumnParent columnParent = new ColumnParent(columnFamilyName);
141                  List<KeySlice> keySlices = client.get_indexed_slices(columnParent, indexClause,
142                      getPredicate(), ThriftConverter.consistencyLevel(consistencyLevelPolicy.get(operationType)));
143    
144                  ret = (keySlices == null || keySlices.isEmpty() ) ? new LinkedHashMap<ByteBuffer, List<Column>>(0) : 
145                    new LinkedHashMap<ByteBuffer, List<Column>>(
146                        keySlices.size());
147                  for (KeySlice keySlice : keySlices) {
148                    ret.put(ByteBuffer.wrap(keySlice.getKey()), ThriftConverter.getColumnList(keySlice.getColumns()));
149                  }
150                } catch (Exception e) {
151                  HectorException he = keyspace.getExceptionsTranslator().translate(e);
152                  setException(he);
153                  throw he;
154                }
155                Map<K, List<Column>> thriftRet = keySerializer.fromBytesMap(ret);
156                return new OrderedRowsImpl<K,N,V>((LinkedHashMap<K, List<Column>>) thriftRet, columnNameSerializer, valueSerializer);
157              }
158            }), this);
159      }
