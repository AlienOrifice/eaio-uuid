139      @Override
140      public Map<ByteBuffer, List<Column>> getRangeSlices(final ColumnParent columnParent,
141          final SlicePredicate predicate, final KeyRange keyRange) throws HectorException {
142        Operation<Map<ByteBuffer, List<Column>>> op = new Operation<Map<ByteBuffer, List<Column>>>(
143            OperationType.READ, failoverPolicy, keyspaceName, credentials) {
144    
145          @Override
146          public Map<ByteBuffer, List<Column>> execute(Cassandra.Client cassandra)
147              throws HectorException {
148            try {
149              List<KeySlice> keySlices = cassandra.get_range_slices(columnParent,
150                  predicate, keyRange, getThriftCl(OperationType.READ));
151              if (keySlices == null || keySlices.isEmpty()) {
152                return new LinkedHashMap<ByteBuffer, List<Column>>(0);
153              }
154              LinkedHashMap<ByteBuffer, List<Column>> ret = new LinkedHashMap<ByteBuffer, List<Column>>(
155                  keySlices.size());
156              for (KeySlice keySlice : keySlices) {
157                ret.put(ByteBuffer.wrap(keySlice.getKey()), getColumnList(keySlice.getColumns()));
158              }
159              return ret;
160            } catch (Exception e) {
161              throw xtrans.translate(e);
162            }
163          };
164        };
165        operateWithFailover(op);
166        return op.getResult();
167      }

271      @Override
272      public SuperColumn getSuperColumn(final ByteBuffer key, final ColumnPath columnPath,
273          final boolean reversed, final int size) throws HectorException {
274        //valideSuperColumnPath(columnPath);
275        final SliceRange sliceRange = new SliceRange(ByteBuffer.wrap(new byte[0]),
276            ByteBuffer.wrap(new byte[0]), reversed, size);
277        Operation<SuperColumn> op = new Operation<SuperColumn>(OperationType.READ, failoverPolicy,
278            keyspaceName, credentials) {
279    
280          @Override
281          public SuperColumn execute(Cassandra.Client cassandra) throws HectorException {
282            ColumnParent clp = new ColumnParent(columnPath.getColumn_family());
283            clp.setSuper_column(columnPath.getSuper_column());
284    
285            SlicePredicate sp = new SlicePredicate();
286            sp.setSlice_range(sliceRange);
287    
288            try {
289              List<ColumnOrSuperColumn> cosc = cassandra.get_slice(key, clp, sp,
290                  getThriftCl(OperationType.READ));
291              if (cosc == null || cosc.isEmpty()) {
292                return null;
293              }
294              return new SuperColumn(ByteBuffer.wrap(columnPath.getSuper_column()), getColumnList(cosc));
295            } catch (Exception e) {
296              throw xtrans.translate(e);
297            }
298          }
299        };
300        operateWithFailover(op);
301        return op.getResult();
302      }

414      @Override
415      public Map<ByteBuffer, SuperColumn> multigetSuperColumn(List<ByteBuffer> keys, ColumnPath columnPath,
416          boolean reversed, int size) throws HectorException {
417        //valideSuperColumnPath(columnPath);
418    
419        // only can get supercolumn by multigetSuperSlice
420        ColumnParent clp = new ColumnParent(columnPath.getColumn_family());
421        clp.setSuper_column(columnPath.getSuper_column());
422    
423        SliceRange sr = new SliceRange(ByteBuffer.wrap(new byte[0]), ByteBuffer.wrap(new byte[0]), reversed, size);
424        SlicePredicate sp = new SlicePredicate();
425        sp.setSlice_range(sr);
426    
427        Map<ByteBuffer, List<SuperColumn>> sclist = multigetSuperSlice(keys, clp, sp);
428    
429        if (sclist == null || sclist.isEmpty()) {
430          return Collections.emptyMap();
431        }
432    
433        Map<ByteBuffer, SuperColumn> result = new HashMap<ByteBuffer, SuperColumn>(keys.size() * 2);
434        for (Map.Entry<ByteBuffer, List<SuperColumn>> entry : sclist.entrySet()) {
435          List<SuperColumn> sclistByKey = entry.getValue();
436          if (sclistByKey.size() > 0) {
437            result.put(entry.getKey(), sclistByKey.get(0));
438          }
439        }
440        return result;
441      }

444      @Override
445      public Map<ByteBuffer, List<SuperColumn>> multigetSuperSlice(final List<ByteBuffer> keys,
446          final ColumnParent columnParent, final SlicePredicate predicate) throws HectorException {
447        Operation<Map<ByteBuffer, List<SuperColumn>>> getCount = new Operation<Map<ByteBuffer, List<SuperColumn>>>(
448            OperationType.READ, failoverPolicy, keyspaceName, credentials) {
449    
450          @Override
451          public Map<ByteBuffer, List<SuperColumn>> execute(Cassandra.Client cassandra)
452              throws HectorException {
453            try {
454              Map<ByteBuffer, List<ColumnOrSuperColumn>> cfmap = cassandra.multiget_slice(
455                  keys, columnParent, predicate, getThriftCl(OperationType.READ));
456              // if user not given super column name, the multiget_slice will return
457              // List
458              // filled with
459              // super column, if user given a column name, the return List will
460              // filled
461              // with column,
462              // this is a bad interface design.
463              if (!columnParent.isSetSuper_column()) {
464                Map<ByteBuffer, List<SuperColumn>> result = new HashMap<ByteBuffer, List<SuperColumn>>();
465                for (Map.Entry<ByteBuffer, List<ColumnOrSuperColumn>> entry : cfmap.entrySet()) {
466                  result.put(entry.getKey(), getSuperColumnList(entry.getValue()));
467                }
468                return result;
469              } else {
470                Map<ByteBuffer, List<SuperColumn>> result = new HashMap<ByteBuffer, List<SuperColumn>>();
471                for (Map.Entry<ByteBuffer, List<ColumnOrSuperColumn>> entry : cfmap.entrySet()) {
472                  SuperColumn spc = new SuperColumn(ByteBuffer.wrap(columnParent.getSuper_column()),
473                      getColumnList(entry.getValue()));
474                  ArrayList<SuperColumn> spclist = new ArrayList<SuperColumn>(1);
475                  spclist.add(spc);
476                  result.put(entry.getKey(), spclist);
477                }
478                return result;
479              }
480            } catch (Exception e) {
481              throw xtrans.translate(e);
482            }
483          }
484        };
485        operateWithFailover(getCount);
486        return getCount.getResult();
487    
488      }
