001    package me.prettyprint.cassandra.model;
002    
003    import java.nio.ByteBuffer;
004    
005    import me.prettyprint.cassandra.utils.Assert;
006    import me.prettyprint.hector.api.Serializer;
007    
008    import org.apache.cassandra.thrift.KeyRange;
009    
010    /**
011     * A helper class for range queries.
012     * <p>
013     * We allow only keys range, no tokens range since if you want to use tokens and you're not hadoop
014     * then you're (probably) doing something wrong (JE)
015     *
016     * @author Ran Tavory
017     *
018     */
019    public final class HKeyRange<K> {
020    
021      private K startKey;
022      private K endKey;
023    
024      private int rowCount = 100;
025    
026      private final Serializer<K> keySerializer;
027    
028      public HKeyRange(Serializer<K> keySerializer) {
029        Assert.notNull(keySerializer, "keySerializer is null");
030        this.keySerializer = keySerializer;
031      }
032    
033      public HKeyRange<K> setKeys(K start, K end) {
034        this.startKey = start;
035        this.endKey = end;
036        return this;
037      }
038      public HKeyRange<K> setRowCount(int rowCount) {
039        this.rowCount = rowCount;
040        return this;
041      }
042    
043      /**
044       *
045       * @return The thrift representation of this object
046       */
047      public KeyRange toThrift() {
048        KeyRange keyRange = new KeyRange(rowCount);
049        keyRange.setStart_key(startKey == null ? ByteBuffer.wrap(new byte[0]) :
050            keySerializer.toByteBuffer(startKey));
051        keyRange.setEnd_key(endKey == null ? ByteBuffer.wrap(new byte[0]) :
052            keySerializer.toByteBuffer(endKey));
053        return keyRange;
054      }
055    
056      @Override
057      public String toString() {
058        return "HKeyRange(start:" + startKey + ",end:" + endKey + "," + ")";
059      }
060    }
