001    package me.prettyprint.cassandra.model;
002    
003    import java.util.List;
004    
005    import me.prettyprint.cassandra.utils.Assert;
006    import me.prettyprint.hector.api.Serializer;
007    import me.prettyprint.hector.api.beans.ColumnSlice;
008    import me.prettyprint.hector.api.beans.Row;
009    
010    import org.apache.cassandra.thrift.Column;
011    
012    /**
013     * A Row is a touple consisting of a Key and a Column Slice.
014     *
015     * A Row may be used to hold the returned value from queries such as get_range_slices.
016     *
017     * @author Ran Tavory
018     *
019     * @param <N> Column name type
020     * @param <V> Column value type
021     *
022     */
023    public final class RowImpl<K,N,V> implements Row<K, N, V>{
024    
025      private final K rowKey;
026      private final ColumnSlice<N,V> columnSlice;
027    
028      /*package*/ RowImpl(K k, List<Column> columns, Serializer<N> nameSerializer,
029          Serializer<V> valueSerializer) {
030        Assert.noneNull(k, columns, nameSerializer, valueSerializer);
031        this.rowKey = k;
032        columnSlice = new ColumnSliceImpl<N,V>(columns, nameSerializer, valueSerializer);
033      }
034    
035      @Override
036      public K getKey() {
037        return rowKey;
038      }
039    
040      @Override
041      public ColumnSlice<N, V> getColumnSlice() {
042        return columnSlice;
043      }
044    
045      @Override
046      public String toString() {
047        return "Row(" + rowKey + "," + columnSlice + ")";
048      }
049    }
