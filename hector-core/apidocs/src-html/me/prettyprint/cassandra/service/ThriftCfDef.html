019      private final String keyspace;

036      public ThriftCfDef(CfDef d) {
037        Assert.notNull(d, "CfDef is null");
038        keyspace = d.keyspace;
039        name = d.name;
040        columnType = ColumnType.getFromValue(d.column_type);
041        comparatorType = ComparatorType.getByClassName(d.comparator_type);
042        subComparatorType = ComparatorType.getByClassName(d.subcomparator_type);
043        comment = d.comment;
044        rowCacheSize = d.row_cache_size;
045        rowCacheSavePeriodInSeconds = d.row_cache_save_period_in_seconds;
046        keyCacheSize = d.key_cache_size;
047        readRepairChance = d.read_repair_chance;
048        columnMetadata = ThriftColumnDef.fromThriftList(d.column_metadata);
049        gcGraceSeconds = d.gc_grace_seconds;
050        defaultValidationClass = d.default_validation_class;
051        id = d.id;
052        minCompactionThreshold = d.min_compaction_threshold;
053        maxCompactionThreshold = d.max_compaction_threshold;
054    
055      }
056      
057      public ThriftCfDef(ColumnFamilyDefinition columnFamilyDefinition) {
058        keyspace = columnFamilyDefinition.getKeyspaceName();
059        name = columnFamilyDefinition.getName();
060        columnType = columnFamilyDefinition.getColumnType();
061        comparatorType = columnFamilyDefinition.getComparatorType();
062        subComparatorType = columnFamilyDefinition.getSubComparatorType();
063        comment = columnFamilyDefinition.getComment();
064        rowCacheSize = columnFamilyDefinition.getRowCacheSize();
065        rowCacheSavePeriodInSeconds = columnFamilyDefinition.getRowCacheSavePeriodInSeconds();
066        keyCacheSize = columnFamilyDefinition.getKeyCacheSize();
067        readRepairChance = columnFamilyDefinition.getReadRepairChance();
068        columnMetadata = columnFamilyDefinition.getColumnMetadata();
069        gcGraceSeconds = columnFamilyDefinition.getGcGraceSeconds();
070        defaultValidationClass = columnFamilyDefinition.getDefaultValidationClass();
071        id = columnFamilyDefinition.getId();
072        minCompactionThreshold = columnFamilyDefinition.getMinCompactionThreshold();
073        maxCompactionThreshold = columnFamilyDefinition.getMaxCompactionThreshold();
074      }
075    
076      public ThriftCfDef(String keyspace, String columnFamilyName) {
077        this.keyspace = keyspace;
078        name = columnFamilyName;
079        columnMetadata = Collections.emptyList();
080    
081        columnType = ColumnType.STANDARD;
082        comparatorType = ComparatorType.BYTESTYPE;
083      }
084    
085      public ThriftCfDef(String keyspace, String columnFamilyName, ComparatorType comparatorType) {
086        this(keyspace, columnFamilyName);
087        if (comparatorType != null) {
088          this.comparatorType = comparatorType;
089        }
090      }
091    
092      public ThriftCfDef(String keyspace, String columnFamilyName, ComparatorType comparatorType, List<ColumnDefinition> columnMetadata) {
093        this(keyspace, columnFamilyName, comparatorType);
094        if (columnMetadata != null) {
095          this.columnMetadata = columnMetadata;
096        }
097      }

099      public static List<ColumnFamilyDefinition> fromThriftList(List<CfDef> cfDefs) {
100        if ((cfDefs == null) || cfDefs.isEmpty()) {
101          return Collections.emptyList();
102        }
103        List<ColumnFamilyDefinition> l = new ArrayList<ColumnFamilyDefinition>(cfDefs.size());
104        for (CfDef d: cfDefs) {
105          l.add(new ThriftCfDef(d));
106        }
107        return l;
108      }

171      public static List<CfDef> toThriftList(List<ColumnFamilyDefinition> cfDefs) {
172        if ((cfDefs == null) || cfDefs.isEmpty()) {
173          return Collections.emptyList();
174        }
175        List<CfDef> l = new ArrayList<CfDef>(cfDefs.size());
176        for (ColumnFamilyDefinition d: cfDefs) {
177          l.add(((ThriftCfDef) d).toThrift());
178        }
179        return l;
180      }
181    
182      public CfDef toThrift() {
183        CfDef d = new CfDef(keyspace, name);
184        d.setColumn_metadata(ThriftColumnDef.toThriftList(columnMetadata));
185        d.setColumn_type(columnType.getValue());
186        d.setComment(comment);
187        d.setComparator_type(comparatorType.getClassName());
188        d.setDefault_validation_class(defaultValidationClass);
189        d.setGc_grace_seconds(gcGraceSeconds);
190        d.setId(id);
191        d.setKey_cache_size(keyCacheSize);
192        d.setMax_compaction_threshold(maxCompactionThreshold);
193        d.setMin_compaction_threshold(minCompactionThreshold);
194        d.setRead_repair_chance(readRepairChance);
195        d.setRow_cache_size(rowCacheSize);
196        if (subComparatorType != null) {
197          d.setSubcomparator_type(subComparatorType.getClassName());
198        }
199        return d;
200      }
