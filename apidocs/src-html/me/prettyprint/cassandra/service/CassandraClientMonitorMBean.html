001    package me.prettyprint.cassandra.service;
002    
003    import java.util.List;
004    import java.util.Set;
005    
006    import me.prettyprint.hector.api.exceptions.HectorTransportException;
007    
008    /**
009     * Defines the various JMX methods the CassandraClientMonitor exposes.
010     *
011     * @author Ran Tavory (ran@outbain.com)
012     *
013     */
014    public interface CassandraClientMonitorMBean {
015    
016      /**
017       * @return Number of failed (and not-recovered) writes.
018       */
019      long getWriteFail();
020    
021      /**
022       * @return Number of failed (and not recovered) reads.
023       */
024      long getReadFail();
025    
026      /**
027       * @return Number of {@link org.apache.cassandra.thrift.TimedOutException} that the client has been able to recover from by
028       * failing over to a different host in the ring.
029       */
030      long getRecoverableTimedOutCount();
031    
032      /**
033       * @return Number of {@link org.apache.cassandra.thrift.UnavailableException} that the client has been able to recover from by
034       * failing over to a different host in the ring.
035       */
036      long getRecoverableUnavailableCount();
037    
038      /**
039       * @return Number of {@link TTransportException} that the client has been able to recover from by
040       * failing over to a different host in the ring.
041       */
042      long getRecoverableTransportExceptionCount();
043    
044      /**
045       * Returns the total number of recoverable errors which is the sum of getRecoverableTimedOutCount,
046       * getRecoverableTimedOutCount and getRecoverableTransportExceptionCount
047       * @return the total number of recoverable errors by failing over the other hosts.
048       */
049      long getRecoverableErrorCount();
050    
051      /**
052       * @return Number of times a skip-host was performed. Hosts are skipped when there are errors at
053       * the current host.
054       */
055      long getSkipHostSuccess();
056    
057      /**
058       * @return Number of times clients were requested when connection pools were exhausted.
059       */
060      long getNumPoolExhaustedEventCount();
061    
062      /**
063       * Number of existing connection pools.
064       * There may be up to one pool per cassandra host.
065       */
066      int getNumPools();
067    
068    
069      /**
070       * Total number of idle clients in all client pools
071       */
072      int getNumIdleConnections();
073    
074      /**
075       * Total number of active clients in all client pools
076       */
077      int getNumActive();
078    
079      /**
080       * Number of exhausted connection pools
081       * @return
082       */
083      int getNumExhaustedPools();
084    
085      long getRecoverableLoadBalancedConnectErrors();
086    
087      /**
088       * List of exhausted pools.
089       * @return
090       */
091      Set<String> getExhaustedPoolNames();
092    
093      /**
094       * Number of threads that are currently blocked, waiting for a free connection.
095       * This number may be greater than 0 only if the {@link ExhaustedPolicy} is
096       * {@link ExhaustedPolicy#WHEN_EXHAUSTED_BLOCK}
097       */
098      int getNumBlockedThreads();
099    
100      /**
101       * How many times did initial connection failed.
102       * @return
103       */
104      long getNumConnectionErrors();
105    
106      public List<String> getKnownHosts();
107    
108      /**
109       * Tells all pulls to update their list of known hosts.
110       * This is useful when an admin adds/removes a host from the ring and wants the application to
111       * update asap.
112       */
113      public void updateKnownHosts() throws HectorTransportException;
114    
115      /**
116       * Retrieves stats per pool.
117       *
118       */
119      public List<String> getStatisticsPerPool();
120    }
