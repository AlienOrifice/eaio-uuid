001    package me.prettyprint.hector.api.query;
002    
003    import me.prettyprint.cassandra.model.thrift.ThriftColumnQuery;
004    
005    
006    /**
007     * The Query interface defines the common parts of all hector queries, such as {@link ThriftColumnQuery}.
008     * <p>
009     * The common usage pattern is to create a query, set the required query attributes and invoke
010     * {@link Query#execute()} such as in the following example:
011     * <pre>
012        ColumnQuery<String, String> q = createColumnQuery(keyspace, serializer, serializer);
013        Result<HColumn<String, String>> r = q.setKey(key).
014            setName(COLUMN_NAME).
015            setColumnFamily(CF_NAME).
016            execute();
017        HColumn<String, String> c = r.get();
018        return c.getValue();
019     * </pre>
020     *
021     * Note that all query mutators, such as setName or setColumnFamily always return the Query object
022     * so it's easy to write strings such as <code>q.setKey(x).setName(y).setColumnFamily(z).execute();</code>
023     *
024     * @author Ran Tavory
025     *
026     * @param <T> Result type. For example Column or SuperColumn
027     */
028    public interface Query<T> {
029    
030      QueryResult<T> execute();
031    
032    }
