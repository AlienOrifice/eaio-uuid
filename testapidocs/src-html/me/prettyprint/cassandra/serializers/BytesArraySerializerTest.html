014    public class BytesArraySerializerTest {
015    
016      @Test
017      public void testConversions() {
018        test(null);
019        test(new byte[]{});
020        test(new byte[]{1});
021        test(new byte[]{1,2,3,4,5});
022        // and also some string oriented tests:
023        test(bytes(""));
024        test(bytes("123"));
025      }
026    
027      private void test(byte[] bytes) {
028        BytesArraySerializer e = new BytesArraySerializer();
029        assertArrayEquals(bytes, e.fromByteBuffer(e.toByteBuffer(bytes))) ;
030      }
