014    public class RoundRobinBalancingPolicyTest extends BaseBalancingPolicyTest {

016      private RoundRobinBalancingPolicy roundRobinBalancingPolicy;

018      @Test
019      public void testGetPoolOk() {
020        roundRobinBalancingPolicy = new RoundRobinBalancingPolicy();
021        assertEquals(poolWith5Active, roundRobinBalancingPolicy.getPool(pools, null));
022        assertEquals(poolWith7Active, roundRobinBalancingPolicy.getPool(pools, null));
023        assertEquals(poolWith10Active, roundRobinBalancingPolicy.getPool(pools, null));
024        assertEquals(poolWith5Active, roundRobinBalancingPolicy.getPool(pools, null));
025        assertEquals(poolWith7Active, roundRobinBalancingPolicy.getPool(pools, null));

028        assertEquals(poolWith7Active, roundRobinBalancingPolicy.getPool(pools, null));
029        assertEquals(poolWith10Active, roundRobinBalancingPolicy.getPool(pools, null));    
030      }
031      
032      @Test
033      public void testSkipExhausted() {
034        Mockito.when(poolWith5Active.getCassandraHost()).thenReturn(new CassandraHost("127.0.0.1:9160"));
035        Mockito.when(poolWith7Active.getCassandraHost()).thenReturn(new CassandraHost("127.0.0.2:9161"));
036        Mockito.when(poolWith10Active.getCassandraHost()).thenReturn(new CassandraHost("127.0.0.3:9162"));
037        
038        roundRobinBalancingPolicy = new RoundRobinBalancingPolicy();
039        assertEquals(poolWith7Active, roundRobinBalancingPolicy.getPool(pools, new HashSet<CassandraHost>(Arrays.asList(new CassandraHost("127.0.0.1:9160")))));
040        assertEquals(poolWith10Active, roundRobinBalancingPolicy.getPool(pools, new HashSet<CassandraHost>(Arrays.asList(new CassandraHost("127.0.0.1:9160")))));
041        assertEquals(poolWith7Active, roundRobinBalancingPolicy.getPool(pools, new HashSet<CassandraHost>(Arrays.asList(new CassandraHost("127.0.0.1:9160")))));
042        assertEquals(poolWith10Active, roundRobinBalancingPolicy.getPool(pools, new HashSet<CassandraHost>(Arrays.asList(new CassandraHost("127.0.0.1:9160")))));
043      }
