021    public class TypeInferringSerializerTest {
022    
023      @Test
024      public void testConversions() {
025        test(new SampleObject());
026        test("Test string");
027        test(Integer.MAX_VALUE);
028        test(Long.MIN_VALUE);
029        //test(null); nulls can't be tested, because the target type is unknown
030      }
031    
032      @Test
033      public void testByteConversions() {
034         TypeInferringSerializer ext = TypeInferringSerializer.get();
035         ByteBuffer byteBuffer = ext.toByteBuffer("bytes".getBytes());
036         Assert.assertTrue(byteBuffer != null);
037       }
038    
039    
040      @SuppressWarnings("rawtypes")
041      private void test(Object object) {
042        TypeInferringSerializer ext = TypeInferringSerializer.get();
043        ByteBuffer bytes = ext.toByteBuffer(object);
044        Serializer ser = SerializerTypeInferer.getSerializer(object.getClass());
045        Object target = ser.fromByteBuffer(bytes);
046        assertEquals(object, target);
047      }
048    
049      private static class SampleObject implements Serializable {
050        private final String a = "test";
051        private final List<Void> b = new ArrayList<Void>();
052        private final Calendar c = Calendar.getInstance();
053        @Override
054        public int hashCode() {
055          final int prime = 31;
056          int result = 1;
057          result = prime * result + (a == null ? 0 : a.hashCode());
058          result = prime * result + (b == null ? 0 : b.hashCode());
059          result = prime * result + (c == null ? 0 : c.hashCode());
060          return result;
061        }
062        @Override
063        public boolean equals(Object obj) {
064          if (this == obj) {
065            return true;
066          }
067          if (obj == null) {
068            return false;
069          }
070          if (getClass() != obj.getClass()) {
071            return false;
072          }
073          SampleObject other = (SampleObject) obj;
074          if (a == null) {
075            if (other.a != null) {
076              return false;
077            }
078          } else if (!a.equals(other.a)) {
079            return false;
080          }
081          if (b == null) {
082            if (other.b != null) {
083              return false;
084            }
085          } else if (!b.equals(other.b)) {
086            return false;
087          }
088          if (c == null) {
089            if (other.c != null) {
090              return false;
091            }
092          } else if (!c.equals(other.c)) {
093            return false;
094          }
095          return true;
096        }
097      }
098    }
