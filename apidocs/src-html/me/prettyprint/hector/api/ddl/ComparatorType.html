001    package me.prettyprint.hector.api.ddl;
002    
003    /**
004     * @author: peter
005     */
006    public class ComparatorType {
007    
008      public static ComparatorType BYTESTYPE = new ComparatorType("org.apache.cassandra.db.marshal.BytesType");
009      public static ComparatorType ASCIITYPE = new ComparatorType("org.apache.cassandra.db.marshal.AsciiType");
010      public static ComparatorType UTF8TYPE = new ComparatorType("org.apache.cassandra.db.marshal.UTF8Type");
011      public static ComparatorType LEXICALUUIDTYPE = new ComparatorType("org.apache.cassandra.db.marshal.LexicalUUIDType");
012      public static ComparatorType TIMEUUIDTYPE = new ComparatorType("org.apache.cassandra.db.marshal.TimeUUIDType");
013      public static ComparatorType LONGTYPE = new ComparatorType("org.apache.cassandra.db.marshal.LongType");
014      public static ComparatorType INTEGERTYPE = new ComparatorType("org.apache.cassandra.db.marshal.IntegerType");
015    
016      private static ComparatorType[] values = {BYTESTYPE, ASCIITYPE, UTF8TYPE, LEXICALUUIDTYPE, TIMEUUIDTYPE, LONGTYPE,INTEGERTYPE};
017    
018      private final String className;
019    
020      private ComparatorType(String className) {
021        this.className = className;
022      }
023    
024      public String getClassName() {
025        return className;
026      }
027    
028      public static ComparatorType getByClassName(String className) {
029    
030        for (int a = 0; a < values.length; a++) {
031          ComparatorType type = values[a];
032          if (type.getClassName().equals(className)) {
033            return type;
034          }
035          if (type.getClassName().equals("org.apache.cassandra.db.marshal." + className)) {
036            return type;
037          }
038        }
039        return new ComparatorType(className);
040      }
041    }
