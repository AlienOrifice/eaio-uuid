001    package me.prettyprint.cassandra.service;

003    import static org.junit.Assert.*;
004

007    import org.junit.Before;
008    import org.junit.Test;
009    
010    public class CassandraHostConfiguratorTest {
011    
012      @Before
013      public void setup() {
014    
015      }
016    
017      @Test
018      public void testSimpleCassandraHostSetup() {
019        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost:9170");
020        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
021        assertEquals(1,cassandraHosts.length);
022      }
023    
024      @Test
025      public void testCassandraHostSetupSplit() {
026        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost:9170,localhost:9171,localhost:9172");
027        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
028        assertEquals(3,cassandraHosts.length);
029        assertEquals(9172,cassandraHosts[2].getPort());
030      }
031    
032      @Test
033      public void testConfigValuesPropogated() {
034        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost:9170,localhost:9171,localhost:9172");
035        cassandraHostConfigurator.setMaxActive(20);
036        cassandraHostConfigurator.setMaxIdle(5);
037        cassandraHostConfigurator.setCassandraThriftSocketTimeout(3000);
038        cassandraHostConfigurator.setMaxWaitTimeWhenExhausted(4000);
039        cassandraHostConfigurator.setExhaustedPolicy(ExhaustedPolicy.WHEN_EXHAUSTED_GROW);
040        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
041        // no need to test all, just a smattering
042        assertEquals(20, cassandraHosts[1].getMaxActive());
043        assertEquals(20, cassandraHosts[0].getMaxActive());
044        assertEquals(5, cassandraHosts[1].getMaxIdle());
045        assertEquals(ExhaustedPolicy.WHEN_EXHAUSTED_GROW, cassandraHosts[1].getExhaustedPolicy());
046        assertEquals(4000, cassandraHosts[1].getMaxWaitTimeWhenExhausted());
047        assertEquals(3000, cassandraHosts[2].getCassandraThriftSocketTimeout());
048        assertEquals(3000, cassandraHosts[0].getCassandraThriftSocketTimeout());
049      }
050    
051      @Test
052      public void testApplyConfig() {
053        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost:9170");
054        cassandraHostConfigurator.setMaxActive(15);
055        CassandraHost extraHost = new CassandraHost("localhost:9171");
056        cassandraHostConfigurator.applyConfig(extraHost);
057        assertEquals(15, extraHost.getMaxActive());
058      }
059    
060      @Test
061      public void testHostnameOnlyDefaultPort() {
062        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost");
063        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
064        assertEquals(CassandraHost.DEFAULT_PORT,cassandraHosts[0].getPort());
065      }
066    
067      @Test
068      public void testHostnameOnlyDefaultPortMultipleHosts() {
069        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("h1,h2,h3:1111");
070        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
071        assertEquals(CassandraHost.DEFAULT_PORT, cassandraHosts[0].getPort());
072        assertEquals(CassandraHost.DEFAULT_PORT, cassandraHosts[1].getPort());
073        assertEquals(1111, cassandraHosts[2].getPort());
074      }
075    
076      @Test
077      public void testConfiguratorPort() {
078        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost");
079        cassandraHostConfigurator.setPort(9177);
080        CassandraHost[] cassandraHosts = cassandraHostConfigurator.buildCassandraHosts();
081        assertEquals(9177,cassandraHosts[0].getPort());
082      }
083    
084      @Test
085      public void testConfiguratorClockResolution() {
086        // Define my own clock resolution.
087        class SequentialClockResolution implements ClockResolution {
088            @Override
089            public long createClock() {
090                return System.currentTimeMillis() * -1;
091            }
092        }
093    
094        CassandraHostConfigurator cassandraHostConfigurator = new CassandraHostConfigurator("localhost");
095        cassandraHostConfigurator.setClockResolution(new SequentialClockResolution());
096        
097        assertNotSame(CassandraHostConfigurator.DEF_CLOCK_RESOLUTION, cassandraHostConfigurator.getClockResolution());
098      }
099    }
