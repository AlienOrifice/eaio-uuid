001    package me.prettyprint.cassandra.serializers;
002    
003    import java.io.UnsupportedEncodingException;
004    import java.nio.ByteBuffer;
005    
006    /**
007     * A StringSerializer translates the byte[] to and from string using utf-8 encoding.
008     * @author Ran Tavory
009     *
010     */
011    public final class StringSerializer extends AbstractSerializer<String> {
012    
013      private static final String UTF_8 = "UTF-8";
014      private static final StringSerializer instance = new StringSerializer();
015    
016      public static StringSerializer get() {
017        return instance;
018      }
019    
020      @Override
021      public ByteBuffer toByteBuffer(String obj) {
022        if (obj == null) {
023          return null;
024        }
025        try {
026          return ByteBuffer.wrap(obj.getBytes(UTF_8));
027        } catch (UnsupportedEncodingException e) {
028          throw new RuntimeException(e);
029        }
030      }
031    
032      @Override
033      public String fromByteBuffer(ByteBuffer byteBuffer) {
034        if (byteBuffer == null) {
035          return null;
036        }
037        try {
038        return new String(byteBuffer.array(), byteBuffer.arrayOffset() + byteBuffer.position(),
039            byteBuffer.remaining(), UTF_8);
040        } catch (UnsupportedEncodingException e) {
041          // not really possible
042          e.printStackTrace();
043        }
044        return null;
045      }
046    }
