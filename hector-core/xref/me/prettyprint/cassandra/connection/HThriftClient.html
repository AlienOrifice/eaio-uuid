58    public Cassandra.Client getCassandra(String keyspaceNameArg) {
59      getCassandra();
60      if ( !StringUtils.equals(keyspaceName, keyspaceNameArg)) {
61        keyspaceName = keyspaceNameArg;
62        try {
63          cassandraClient.set_keyspace(keyspaceName);
64          if ( log.isDebugEnabled() )
65              log.debug("keyspace reset from {} to {}", keyspaceName, keyspaceNameArg);
66        } catch (InvalidRequestException ire) {
67          throw new HInvalidRequestException(ire);
68        } catch (TException e) {
69          throw new HectorTransportException(e);
70        }
71      }
72      return cassandraClient;
73    }

75    HThriftClient close() {
76      if ( log.isDebugEnabled() ) {
77        log.debug("Closing client {}", this);
78      }
79      if ( isOpen() ) {
80        try {
81          transport.flush();
82        } catch (Exception e) {
83          log.error("Could not flush transport (to be expected if the pool is shutting down) in close for client: " + toString(), e);
84        } finally {
85          try {
86            transport.close();
87          } catch (Exception e) {
88            log.error("Error on transport close for client: " +toString(), e);
89          }
90        }
91      }
92      return this;
93    }

96    HThriftClient open() {
97      if ( isOpen() ) {
98        throw new IllegalStateException("Open called on already open connection. You should not have gotten here.");
99      }
100     if ( log.isDebugEnabled() ) {
101       log.debug("Creating a new thrift connection to {}", cassandraHost);
102     }
103 
104     if (cassandraHost.getUseThriftFramedTransport()) {
105       transport = new TFramedTransport(new TSocket(cassandraHost.getHost(), cassandraHost.getPort(), timeout));
106     } else {
107       transport = new TSocket(cassandraHost.getHost(), cassandraHost.getPort(), timeout);
108     }
109     try {
110       transport.open();
111     } catch (TTransportException e) {
112       // Thrift exceptions aren't very good in reporting, so we have to catch the exception here and
113       // add details to it.
114       log.error("Unable to open transport to " + cassandraHost.getName(), e);
115       //clientMonitor.incCounter(Counter.CONNECT_ERROR);
116       throw new HectorTransportException("Unable to open transport to " + cassandraHost.getName() +" , " +
117           e.getLocalizedMessage(), e);
118     }
119     return this;
120   }

134   /**
135    * If CassandraHost was not null we use {@link CassandraHost#getCassandraThriftSocketTimeout()}
136    * if it was greater than zero. Otherwise look for an environment
137    * variable name CASSANDRA_THRIFT_SOCKET_TIMEOUT value.
138    * If doesn't exist, returns 0.
139    * @param cassandraHost
140    */
141   private int getTimeout(CassandraHost cassandraHost) {
142     int timeoutVar = 0;
143     if ( cassandraHost != null && cassandraHost.getCassandraThriftSocketTimeout() > 0 ) {
144       timeoutVar = cassandraHost.getCassandraThriftSocketTimeout();
145     } else {
146       String timeoutStr = System.getProperty(
147           SystemProperties.CASSANDRA_THRIFT_SOCKET_TIMEOUT.toString());
148       if (timeoutStr != null && timeoutStr.length() > 0) {
149         try {
150           timeoutVar = Integer.valueOf(timeoutStr);
151         } catch (NumberFormatException e) {
152           log.error("Invalid value for CASSANDRA_THRIFT_SOCKET_TIMEOUT", e);
153         }
154       }
155     }
156     return timeoutVar;
157   }
