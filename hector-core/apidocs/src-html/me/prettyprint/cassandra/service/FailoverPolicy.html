001    package me.prettyprint.cassandra.service;
002    
003    /**
004     * What should the client do if a call to cassandra node fails and we suspect that the node is
005     * down. (e.g. it's a communication error, not an application error).
006     *
007     * {@value #FAIL_FAST} will return the error as is to the user and not try anything smart
008     *
009     * {@value #ON_FAIL_TRY_ONE_NEXT_AVAILABLE} will try one more random server before returning to the
010     * user with an error
011     *
012     * {@value #ON_FAIL_TRY_ALL_AVAILABLE} will try all available servers in the cluster before giving
013     * up and returning the communication error to the user.
014     *
015     */
016    public class FailoverPolicy {
017    
018      /** On communication failure, just return the error to the client and don't retry */
019      public static FailoverPolicy FAIL_FAST = new FailoverPolicy(0, 0);
020      /** On communication error try one more server before giving up */
021      public static FailoverPolicy ON_FAIL_TRY_ONE_NEXT_AVAILABLE = new FailoverPolicy(1, 0);
022      /** On communication error try all known servers before giving up */
023      public static FailoverPolicy ON_FAIL_TRY_ALL_AVAILABLE = new FailoverPolicy(Integer.MAX_VALUE - 1, 0);
024    
025      public final int numRetries;
026    
027      public final int sleepBetweenHostsMilli;
028    
029      public FailoverPolicy(int numRetries, int sleepBwHostsMilli) {
030        this.numRetries = numRetries;
031        sleepBetweenHostsMilli = sleepBwHostsMilli;
032      }
033    }
