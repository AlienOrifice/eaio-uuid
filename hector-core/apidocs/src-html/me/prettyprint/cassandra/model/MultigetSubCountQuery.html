001    package me.prettyprint.cassandra.model;
002    
003    import java.util.Map;
004    
005    import me.prettyprint.cassandra.service.KeyspaceService;
006    import me.prettyprint.cassandra.utils.Assert;
007    import me.prettyprint.hector.api.Keyspace;
008    import me.prettyprint.hector.api.Serializer;
009    import me.prettyprint.hector.api.exceptions.HectorException;
010    import me.prettyprint.hector.api.query.QueryResult;
011    
012    import org.apache.cassandra.thrift.ColumnParent;
013    
014    /**
015     * Does a multi_get_count within a given superColumn
016     *
017     * @author zznate
018     */
019    public class MultigetSubCountQuery<K,SN,N> extends MultigetCountQuery<K, N> {
020    
021      private final Serializer<SN> superNameSerializer;
022      private SN superColumnName;
023    
024      public MultigetSubCountQuery(Keyspace ko,
025          Serializer<SN> superNameSerializer,
026          Serializer<K> keySerializer,
027          Serializer<N> nameSerializer) {
028        super(ko, keySerializer, nameSerializer);
029        Assert.notNull(superNameSerializer, "superNameSerializer is null");
030        this.superNameSerializer = superNameSerializer;
031      }
032    
033      public MultigetSubCountQuery<K,SN,N> setSuperColumn(SN sc) {
034        superColumnName = sc;
035        return this;
036      }
037    
038      @Override
039      public QueryResult<Map<K, Integer>> execute() {
040        Assert.notNull(keys, "keys list is null");
041        Assert.notNull(columnFamily, "columnFamily is null");
042        Assert.notNull(superColumnName, "superColumnName is null");
043        return new QueryResultImpl<Map<K,Integer>>(keyspace.doExecute(
044            new KeyspaceOperationCallback<Map<K,Integer>>() {
045              @Override
046              public Map<K,Integer> doInKeyspace(KeyspaceService ks) throws HectorException {
047                ColumnParent columnParent = new ColumnParent(columnFamily);
048                columnParent.setSuper_column(superNameSerializer.toByteBuffer(superColumnName));
049                Map<K,Integer> counts = keySerializer.fromBytesMap(
050                    ks.multigetCount(keySerializer.toBytesList(keys), columnParent, slicePredicate.toThrift()));
051                return counts;
052              }
053            }), this);
054      }
055    }
