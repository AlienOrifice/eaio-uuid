001    package me.prettyprint.cassandra.model.thrift;
002    
003    import java.nio.ByteBuffer;
004    import java.util.ArrayList;
005    import java.util.Arrays;
006    import java.util.List;
007    import java.util.Map;
008    
009    import me.prettyprint.cassandra.model.AbstractSliceQuery;
010    import me.prettyprint.cassandra.model.KeyspaceOperationCallback;
011    import me.prettyprint.cassandra.model.QueryResultImpl;
012    import me.prettyprint.cassandra.model.RowsImpl;
013    import me.prettyprint.cassandra.service.KeyspaceService;
014    import me.prettyprint.cassandra.utils.Assert;
015    import me.prettyprint.hector.api.Keyspace;
016    import me.prettyprint.hector.api.Serializer;
017    import me.prettyprint.hector.api.beans.Rows;
018    import me.prettyprint.hector.api.exceptions.HectorException;
019    import me.prettyprint.hector.api.query.MultigetSliceQuery;
020    import me.prettyprint.hector.api.query.QueryResult;
021    
022    import org.apache.cassandra.thrift.Column;
023    import org.apache.cassandra.thrift.ColumnParent;
024    
025    /**
026     * A query wrapper for the thrift call multiget_slice
027     */
028    public final class ThriftMultigetSliceQuery<K, N, V> extends AbstractSliceQuery<K, N, V, Rows<K, N, V>>
029        implements MultigetSliceQuery<K, N, V> {
030    
031      private Iterable<K> keys;
032    
033      public ThriftMultigetSliceQuery(Keyspace k,
034          Serializer<K> keySerializer,
035          Serializer<N> nameSerializer,
036          Serializer<V> valueSerializer) {
037        super(k, keySerializer, nameSerializer, valueSerializer);
038      }
039    
040      @Override
041      public MultigetSliceQuery<K, N, V> setKeys(K... keys) {
042        this.keys = Arrays.asList(keys);
043        return this;
044      }
045    
046      @Override
047      public MultigetSliceQuery<K, N, V> setKeys(Iterable<K> keys) {
048        this.keys = keys;
049        return this;
050      }
051    
052    
053      @Override
054      public QueryResult<Rows<K, N,V>> execute() {
055        Assert.notNull(columnFamilyName, "columnFamilyName can't be null");
056        Assert.notNull(keys, "keys can't be null");
057    
058        return new QueryResultImpl<Rows<K, N,V>>(keyspace.doExecute(
059            new KeyspaceOperationCallback<Rows<K, N,V>>() {
060              @Override
061              public Rows<K, N,V> doInKeyspace(KeyspaceService ks) throws HectorException {
062                List<ByteBuffer> keysList = new ArrayList<ByteBuffer>();
063                for (K k : keys) {
064                  keysList.add(keySerializer.toByteBuffer(k));
065                }
066                ColumnParent columnParent = new ColumnParent(columnFamilyName);
067                Map<K, List<Column>> thriftRet = keySerializer.fromBytesMap(
068                    ks.multigetSlice(keysList, columnParent, getPredicate()));
069                return new RowsImpl<K, N, V>(thriftRet, columnNameSerializer, valueSerializer);
070              }
071            }), this);
072      }
073    
074      @Override
075      public String toString() {
076        return "MultigetSliceQuery(" + keys + "," + super.toStringInternal() + ")";
077      }
078    
079      @SuppressWarnings("unchecked")
080      @Override
081      public MultigetSliceQuery<K, N, V> setColumnNames(N... columnNames) {
082        return (MultigetSliceQuery<K, N, V>) super.setColumnNames(columnNames);
083      }
084    
085      @SuppressWarnings("unchecked")
086      @Override
087      public MultigetSliceQuery<K, N, V> setRange(N start, N finish, boolean reversed,
088          int count) {
089        return (MultigetSliceQuery<K, N, V>) super.setRange(start, finish, reversed, count);
090      }
091    
092      @SuppressWarnings("unchecked")
093      @Override
094      public MultigetSliceQuery<K, N, V> setColumnFamily(String cf) {
095        return (MultigetSliceQuery<K, N, V>) super.setColumnFamily(cf);
096      }
097    }
