001    package me.prettyprint.cassandra.model;
002    
003    import java.util.ArrayList;
004    import java.util.Collections;
005    import java.util.LinkedHashMap;
006    import java.util.List;
007    import java.util.Map;
008    
009    import me.prettyprint.hector.api.Serializer;
010    import me.prettyprint.hector.api.beans.OrderedSuperRows;
011    import me.prettyprint.hector.api.beans.SuperRow;
012    
013    import org.apache.cassandra.thrift.SuperColumn;
014    
015    /**
016     * Return type from get_range_slices for super columns
017     * @author Ran Tavory
018     *
019     */
020    public final class OrderedSuperRowsImpl<K,SN,N,V> extends SuperRowsImpl<K,SN,N,V>
021        implements OrderedSuperRows<K, SN, N, V>{
022    
023      private final List<SuperRow<K,SN,N,V>> rowsList;
024    
025      public OrderedSuperRowsImpl(LinkedHashMap<K, List<SuperColumn>> thriftRet, Serializer<K> keySerializer,
026          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
027          Serializer<V> valueSerializer) {
028        super(thriftRet, keySerializer, sNameSerializer, nameSerializer, valueSerializer);
029        rowsList = new ArrayList<SuperRow<K,SN,N,V>>(thriftRet.size());
030        for (Map.Entry<K, List<SuperColumn>> entry: thriftRet.entrySet()) {
031          rowsList.add(new SuperRowImpl<K,SN,N,V>(entry.getKey(), entry.getValue(), sNameSerializer,
032              nameSerializer, valueSerializer));
033        }
034      }
035    
036      /**
037       * Preserves rows order
038       * @return an unmodifiable list of Rows
039       */
040      @Override
041      public List<SuperRow<K,SN,N,V>> getList() {
042        return Collections.unmodifiableList(rowsList);
043      }
044    }
