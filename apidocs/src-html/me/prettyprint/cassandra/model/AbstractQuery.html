001    package me.prettyprint.cassandra.model;
002    
003    import me.prettyprint.cassandra.utils.Assert;
004    import me.prettyprint.hector.api.Keyspace;
005    import me.prettyprint.hector.api.Serializer;
006    import me.prettyprint.hector.api.query.Query;
007    
008    public abstract class AbstractQuery<K, N, V, T> implements Query<T> {
009    
010      protected final ExecutingKeyspace keyspace;
011      protected String columnFamilyName;
012      protected Serializer<K> keySerializer;
013      protected Serializer<N> columnNameSerializer;
014      protected Serializer<V> valueSerializer;
015      // add: FailoverPolicy, ConsistencyLevelPolicy, Credentials?
016    
017      /*package*/ AbstractQuery(Keyspace k, Serializer<K> keySerializer,
018          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
019        Assert.noneNull(k, keySerializer, nameSerializer, valueSerializer);
020        keyspace = (ExecutingKeyspace) k;
021        this.keySerializer = keySerializer;
022        this.columnNameSerializer = nameSerializer;
023        this.valueSerializer = valueSerializer;
024      }
025    
026      public Query<T> setColumnFamily(String cf) {
027        this.columnFamilyName = cf;
028        return this;
029      }
030    
031      public Serializer<K> getKeySerializer() {
032        return keySerializer;
033      }
034    
035      public AbstractQuery<K, N, V, T> setKeySerializer(Serializer<K> keySerializer) {
036        this.keySerializer = keySerializer;
037        return this;
038      }
039    
040      public Serializer<N> getColumnNameSerializer() {
041        return columnNameSerializer;
042      }
043    
044      public AbstractQuery<K, N, V, T> setColumnNameSerializer(Serializer<N> columnNameSerializer) {
045        this.columnNameSerializer = columnNameSerializer;
046        return this;
047      }
048    
049      public Serializer<V> getValueSerializer() {
050        return valueSerializer;
051      }
052    
053      public void setValueSerializer(Serializer<V> valueSerializer) {
054        this.valueSerializer = valueSerializer;
055      }
056    }
