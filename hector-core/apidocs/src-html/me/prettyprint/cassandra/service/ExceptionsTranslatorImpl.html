001    package me.prettyprint.cassandra.service;
002    
003    import java.util.NoSuchElementException;
004    
005    import me.prettyprint.hector.api.exceptions.HCassandraInternalException;
006    import me.prettyprint.hector.api.exceptions.HInvalidRequestException;
007    import me.prettyprint.hector.api.exceptions.HNotFoundException;
008    import me.prettyprint.hector.api.exceptions.HTimedOutException;
009    import me.prettyprint.hector.api.exceptions.HUnavailableException;
010    import me.prettyprint.hector.api.exceptions.HectorException;
011    import me.prettyprint.hector.api.exceptions.HectorTransportException;
012    import me.prettyprint.hector.api.exceptions.PoolExhaustedException;
013    import me.prettyprint.hector.api.exceptions.PoolIllegalStateException;
014    
015    import org.apache.thrift.TApplicationException;
016    import org.apache.thrift.TException;
017    import org.apache.thrift.protocol.TProtocolException;
018    import org.apache.thrift.transport.TTransportException;
019    
020    public final class ExceptionsTranslatorImpl implements ExceptionsTranslator {
021    
022    
023      @Override
024      public HectorException translate(Throwable original) {
025        if (original instanceof HectorException) {
026          return (HectorException) original;
027        } else if (original instanceof TApplicationException) {
028          return new HCassandraInternalException(((TApplicationException)original).getType(), original.getMessage());
029        } else if (original instanceof TException || original instanceof TTransportException) {
030          return new HectorTransportException(original);
031        } else if (original instanceof org.apache.cassandra.thrift.TimedOutException) {
032          return new HTimedOutException(original);
033        } else if (original instanceof org.apache.cassandra.thrift.InvalidRequestException) {
034          // See bug https://issues.apache.org/jira/browse/CASSANDRA-1862
035          // If a bootstrapping node is accessed it responds with IRE. It makes more sense to throw
036          // UnavailableException.
037          // Hector wraps this caveat and fixes this.
038          String why = ((org.apache.cassandra.thrift.InvalidRequestException) original).getWhy();
039          if (why != null && why.contains("bootstrap")) {
040            return new HUnavailableException(original);
041          }
042          HInvalidRequestException e = new HInvalidRequestException(original);
043          e.setWhy(why);
044          return e;
045        } else if (original instanceof TProtocolException) {
046          return new HInvalidRequestException(original);
047        } else if (original instanceof org.apache.cassandra.thrift.NotFoundException) {
048          return new HNotFoundException(original);
049        } else if (original instanceof org.apache.cassandra.thrift.UnavailableException) {
050          return new HUnavailableException(original);
051        } else if (original instanceof NoSuchElementException) {
052          return new PoolExhaustedException(original);
053        } else if (original instanceof IllegalStateException) {
054          return new PoolIllegalStateException(original);
055        } else {
056          return new HectorException(original);
057        }
058      }
059    
060    }
