001    package me.prettyprint.cassandra.examples;
002    
003    import static me.prettyprint.hector.api.factory.HFactory.createColumn;
004    import static me.prettyprint.hector.api.factory.HFactory.createColumnQuery;
005    import static me.prettyprint.hector.api.factory.HFactory.createKeyspace;
006    import static me.prettyprint.hector.api.factory.HFactory.createMultigetSliceQuery;
007    import static me.prettyprint.hector.api.factory.HFactory.createMutator;
008    import static me.prettyprint.hector.api.factory.HFactory.getOrCreateCluster;
009    
010    import java.util.HashMap;
011    import java.util.Map;
012    
013    import me.prettyprint.cassandra.serializers.StringSerializer;
014    import me.prettyprint.hector.api.Cluster;
015    import me.prettyprint.hector.api.Keyspace;
016    import me.prettyprint.hector.api.Serializer;
017    import me.prettyprint.hector.api.beans.HColumn;
018    import me.prettyprint.hector.api.beans.Rows;
019    import me.prettyprint.hector.api.exceptions.HectorException;
020    import me.prettyprint.hector.api.mutation.Mutator;
021    import me.prettyprint.hector.api.query.ColumnQuery;
022    import me.prettyprint.hector.api.query.MultigetSliceQuery;
023    import me.prettyprint.hector.api.query.QueryResult;
024    
025    /**
026     * Thread Safe
027     * @author Ran Tavory
028     *
029     */
030    public class ExampleDaoV2 {
031    
032      private final static String KEYSPACE = "Keyspace1";
033      private final static String HOST_PORT = "localhost:9170";
034      private final static String CF_NAME = "Standard1";
035      /** Column name where values are stored */
036      private final static String COLUMN_NAME = "v";
037      private final StringSerializer serializer = StringSerializer.get();
038    
039      private final Keyspace keyspace;
040    
041      public static void main(String[] args) throws HectorException {
042        Cluster c = getOrCreateCluster("MyCluster", HOST_PORT);
043        ExampleDaoV2 ed = new ExampleDaoV2(createKeyspace(KEYSPACE, c));
044        ed.insert("key1", "value1", StringSerializer.get());
045    
046        System.out.println(ed.get("key1", StringSerializer.get()));
047      }
048    
049      public ExampleDaoV2(Keyspace keyspace) {
050        this.keyspace = keyspace;
051      }
052    
053      /**
054       * Insert a new value keyed by key
055       *
056       * @param key   Key for the value
057       * @param value the String value to insert
058       */
059      public <K> void insert(final K key, final String value, Serializer<K> keySerializer) {
060        createMutator(keyspace, keySerializer).insert(
061            key, CF_NAME, createColumn(COLUMN_NAME, value, serializer, serializer));
062      }
063    
064      /**
065       * Get a string value.
066       *
067       * @return The string value; null if no value exists for the given key.
068       */
069      public <K> String get(final K key, Serializer<K> keySerializer) throws HectorException {
070        ColumnQuery<K, String, String> q = createColumnQuery(keyspace, keySerializer, serializer, serializer);
071        QueryResult<HColumn<String, String>> r = q.setKey(key).
072            setName(COLUMN_NAME).
073            setColumnFamily(CF_NAME).
074            execute();
075        HColumn<String, String> c = r.get();
076        return c == null ? null : c.getValue();
077      }
078    
079      /**
080       * Get multiple values
081       * @param keys
082       * @return
083       */
084      public <K> Map<K, String> getMulti(Serializer<K> keySerializer, K... keys) {
085        MultigetSliceQuery<K, String,String> q = createMultigetSliceQuery(keyspace, keySerializer, serializer, serializer);
086        q.setColumnFamily(CF_NAME);
087        q.setKeys(keys);
088        q.setColumnNames(COLUMN_NAME);
089    
090        QueryResult<Rows<K, String,String>> r = q.execute();
091        Rows<K, String,String> rows = r.get();
092        Map<K, String> ret = new HashMap<K, String>(keys.length);
093        for (K k: keys) {
094          HColumn<String,String> c = rows.getByKey(k).getColumnSlice().getColumnByName(COLUMN_NAME);
095          if (c != null && c.getValue() != null) {
096            ret.put(k, c.getValue());
097          }
098        }
099        return ret;
100      }
101    
102      /**
103       * Insert multiple values
104       */
105      public <K> void insertMulti(Map<K, String> keyValues, Serializer<K> keySerializer) {
106        Mutator<K> m = createMutator(keyspace, keySerializer);
107        for (Map.Entry<K, String> keyValue: keyValues.entrySet()) {
108          m.addInsertion(keyValue.getKey(), CF_NAME,
109              createColumn(COLUMN_NAME, keyValue.getValue(), keyspace.createClock(), serializer, serializer));
110        }
111        m.execute();
112      }
113    
114      /**
115       * Delete multiple values
116       */
117      public <K> void delete(Serializer<K> keySerializer, K... keys) {
118        Mutator<K> m = createMutator(keyspace, keySerializer);
119        for (K key: keys) {
120          m.addDeletion(key, CF_NAME,  COLUMN_NAME, serializer);
121        }
122        m.execute();
123      }
124    }
