001    package me.prettyprint.cassandra.serializers;
002    
003    
004    import java.io.UnsupportedEncodingException;
005    import java.util.Arrays;
006    import java.util.Collection;
007    
008    import org.junit.Assert;
009    import org.junit.Test;
010    import org.junit.runner.RunWith;
011    import org.junit.runners.Parameterized;
012    import org.junit.runners.Parameterized.Parameters;
013    
014    @RunWith(value = Parameterized.class)
015    public class StringSerializerTest {
016    
017      private final String str;
018    
019      public StringSerializerTest(String str) {
020        this.str = str;
021      }
022    
023      @Parameters
024      public static Collection<Object[]> data() throws UnsupportedEncodingException {
025        Object[][] data = new Object[][] {
026                                          {"" },
027                                          {null},
028                                          {"123"},
029                                          {"QWER"},
030                                          {"!@#$#$^%&^*fdghdfghdfgh%^&*"},
031                                          {new String("\u05E9".getBytes(), "utf-8")}
032                                          };
033        return Arrays.asList(data);
034      }
035    
036      @Test
037      public void test() {
038        StringSerializer e = new StringSerializer();
039        Assert.assertEquals(str, e.fromByteBuffer(e.toByteBuffer(str))) ;
040      }
041    }
