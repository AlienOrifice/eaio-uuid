15  /**
16   * Returned by a MultigetSuperSliceQuery (multiget_slice for supercolumns)
17   *
18   * @author Ran Tavory
19   *
20   * @param <N>
21   * @param <V>
22   */
23  public class SuperRowsImpl<K, SN, N, V> implements SuperRows<K, SN, N, V> {
24  
25    private final Map<K, SuperRow<K, SN, N, V>> rows;
26  
27    Serializer<K> keySerializer;
28  
29    public SuperRowsImpl(Map<K, List<SuperColumn>> thriftRet, Serializer<K> keySerializer,
30        Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
31      Assert.noneNull(thriftRet, keySerializer, sNameSerializer, nameSerializer, valueSerializer);
32      this.keySerializer = keySerializer;
33      rows = new HashMap<K, SuperRow<K, SN, N, V>>(thriftRet.size());
34      for (Map.Entry<K, List<SuperColumn>> entry : thriftRet.entrySet()) {
35        rows.put(entry.getKey(), new SuperRowImpl<K, SN, N, V>(entry.getKey(), entry.getValue(),
36            sNameSerializer, nameSerializer, valueSerializer));
37      }
38    }
39  
40    @Override
41    public SuperRow<K, SN, N, V> getByKey(K key) {
42      return rows.get(key);
43    }
44  
45    @Override
46    public int getCount() {
47      return rows.size();
48    }
49  
50    @Override
51    public Iterator<SuperRow<K, SN, N, V>> iterator() {
52      return rows.values().iterator();
53    }
54  
55    @Override
56    public String toString() {
57      return "SuperRows(" + rows + ")";
58    }
