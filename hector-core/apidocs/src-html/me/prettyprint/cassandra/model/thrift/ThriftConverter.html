001    package me.prettyprint.cassandra.model.thrift;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import me.prettyprint.hector.api.HConsistencyLevel;
007    
008    import org.apache.cassandra.thrift.Column;
009    import org.apache.cassandra.thrift.ColumnOrSuperColumn;
010    import org.apache.cassandra.thrift.ConsistencyLevel;
011    
012    /**
013     * @author: peter
014     * @author ran
015     */
016    public class ThriftConverter {
017    
018      public static ConsistencyLevel consistencyLevel(HConsistencyLevel c) {
019        switch (c) {
020        case ALL:
021          return ConsistencyLevel.ALL;
022        case ANY:
023          return ConsistencyLevel.ANY;
024        case ONE:
025          return ConsistencyLevel.ONE;
026        case QUORUM:
027          return ConsistencyLevel.QUORUM;
028        case EACH_QUORUM:
029          return ConsistencyLevel.EACH_QUORUM;
030        case LOCAL_QUORUM:
031          return ConsistencyLevel.LOCAL_QUORUM;
032        default:
033          throw new RuntimeException("Unregornized consistency level " + c);
034        }
035      }
036      
037      /**
038       * Converts a list of ColumnOrSuperColumn to Column
039       * @param columns
040       * @return
041       */
042      public static List<Column> getColumnList(List<ColumnOrSuperColumn> columns) {
043        ArrayList<Column> list = new ArrayList<Column>(columns.size());
044        for (ColumnOrSuperColumn col : columns) {
045          list.add(col.getColumn());
046        }
047        return list;
048      }
049      
050    }
