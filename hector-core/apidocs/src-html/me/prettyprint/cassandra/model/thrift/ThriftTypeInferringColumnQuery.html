001    package me.prettyprint.cassandra.model.thrift;
002    
003    import me.prettyprint.cassandra.serializers.StringSerializer;
004    import me.prettyprint.cassandra.serializers.TypeInferringSerializer;
005    import me.prettyprint.hector.api.Keyspace;
006    import me.prettyprint.hector.api.Serializer;
007    
008    /**
009     * The same as ColumnQuery, but dynamically inferring the serializers needed to transform the key
010     * and column name to their binary form.
011     *
012     * @author Bozhidar Bozhanov
013     *
014     * @param <K> key type
015     * @param <N> column name type
016     * @param <V> value type
017     */
018    public class ThriftTypeInferringColumnQuery<K, N, V> extends ThriftColumnQuery<K, N, V> {
019    
020      public ThriftTypeInferringColumnQuery(Keyspace keyspace, Serializer<V> valueSerializer) {
021        super(keyspace, TypeInferringSerializer.<K>get(), TypeInferringSerializer.<N>get(),
022            valueSerializer);
023      }
024    
025      @SuppressWarnings("unchecked")
026      public ThriftTypeInferringColumnQuery(Keyspace keyspace) {
027        super(keyspace, TypeInferringSerializer.<K> get(), TypeInferringSerializer.<N> get(),
028            (Serializer<V>) StringSerializer.get());
029      }
030    }
