001    package me.prettyprint.cassandra;
002    
003    import java.io.IOException;
004    
005    import me.prettyprint.cassandra.connection.HConnectionManager;
006    import me.prettyprint.cassandra.service.CassandraHostConfigurator;
007    import me.prettyprint.cassandra.testutils.EmbeddedServerHelper;
008    
009    import org.apache.cassandra.config.ConfigurationException;
010    import org.apache.thrift.transport.TTransportException;
011    import org.junit.AfterClass;
012    import org.junit.BeforeClass;
013    
014    /**
015     * Base class for test cases that need access to EmbeddedServerHelper
016     *
017     * @author Nate McCall (nate@vervewireless.com)
018     *
019     */
020    public abstract class BaseEmbededServerSetupTest {
021    
022      private static EmbeddedServerHelper embedded;
023    
024      protected HConnectionManager connectionManager;
025      protected CassandraHostConfigurator cassandraHostConfigurator;
026    
027      /**
028       * Set embedded cassandra up and spawn it in a new thread.
029       *
030       * @throws TTransportException
031       * @throws IOException
032       * @throws InterruptedException
033       */
034      @BeforeClass
035      public static void setup() throws TTransportException, IOException, InterruptedException, ConfigurationException {
036        embedded = new EmbeddedServerHelper();
037        embedded.setup();
038      }
039    
040      @AfterClass
041      public static void teardown() throws IOException {
042        embedded.teardown();
043        embedded = null;
044      }
045    
046    
047      protected void setupClient() {
048        cassandraHostConfigurator = new CassandraHostConfigurator("127.0.0.1:9170");
049        connectionManager = new HConnectionManager(cassandraHostConfigurator);
050      }
051    }
