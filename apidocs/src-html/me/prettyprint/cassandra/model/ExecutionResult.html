001    package me.prettyprint.cassandra.model;
002    
003    import me.prettyprint.cassandra.service.CassandraHost;
004    
005    
006    /**
007     * Describes the state of the executed cassandra command.
008     * This is a handy call metadata inspector which reports the call's execution time, status,
009     * which actual host was connected etc.
010     *
011     * @author Ran
012     *
013     */
014    public class ExecutionResult<T> {
015    
016      private final T value;
017      private final long execTime;
018      private final CassandraHost cassandraHost;
019      
020      protected static final String BASE_MSG_FORMAT = "%s took (%dus) for query (%s) on host: %s";
021    
022      public ExecutionResult(T value, long execTime, CassandraHost cassandraHost) {
023        this.value = value;
024        this.execTime = execTime;
025        this.cassandraHost = cassandraHost;
026      }
027    
028      /**
029       * @return the actual value returned from the query (or null if it was a mutation that doesn't
030       * return a value)
031       */
032      public T get() {
033        return value;
034      }
035    
036      public long getExecutionTimeMicro() {
037        return execTime;
038      }
039    
040      @Override
041      public String toString() {
042        return formatMessage("ExecutionResult", "n/a");
043      }
044      
045      protected String formatMessage(String resultName, String query) {
046        return String.format(BASE_MSG_FORMAT, resultName, execTime, query, (cassandraHost != null ? cassandraHost.getName() : "[none]"));
047      }
048    
049      /** The cassandra host that was actually used to execute the operation */
050      public CassandraHost getHostUsed() {
051        return this.cassandraHost;
052      }
053    
054    
055    }
