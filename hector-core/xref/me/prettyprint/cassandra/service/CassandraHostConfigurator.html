46    public CassandraHost[] buildCassandraHosts() {
47      if (this.hosts == null) {
48        return null;
49      }
50      String[] hostVals = hosts.split(",");
51      CassandraHost[] cassandraHosts = new CassandraHost[hostVals.length];
52      for (int x=0; x<hostVals.length; x++) {
53        CassandraHost cassandraHost = this.port == CassandraHost.DEFAULT_PORT ? new CassandraHost(hostVals[x]) : new CassandraHost(hostVals[x], this.port);
54        applyConfig(cassandraHost);
55        cassandraHosts[x] = cassandraHost;
56      }
57      return cassandraHosts;
58    }

66      cassandraHost.setTimeBetweenEvictionRunsMillis(timeBetweenEvictionRunsMillis);
67      cassandraHost.setMaxWaitTimeWhenExhausted(maxWaitTimeWhenExhausted);
68      cassandraHost.setUseThriftFramedTransport(useThriftFramedTransport);
69  
70      // this is special as it can be passed in as a system property
71      if (cassandraThriftSocketTimeout > 0) {
72        cassandraHost.setCassandraThriftSocketTimeout(cassandraThriftSocketTimeout);
73      }
74      if (exhaustedPolicy != null) {
75        cassandraHost.setExhaustedPolicy(exhaustedPolicy);
76      }
77    }
