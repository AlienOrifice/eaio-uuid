061      public HectorTemplateImpl(Cluster cluster, String keyspace, int replicationFactor, String replicationStrategyClass, ConfigurableConsistencyLevel configurableConsistencyLevelPolicy) {
062        this.cluster = cluster;
063        this.keyspaceName = keyspace;
064        this.replicationFactor = replicationFactor;

066        this.configurableConsistencyLevelPolicy = configurableConsistencyLevelPolicy;
067        initKeyspaceOperator();
068      }

070      public void init() {

072      }

074      private void initKeyspaceOperator() {
075        ConsistencyLevelPolicy clPolicy;
076        if (configurableConsistencyLevelPolicy == null) {
077          clPolicy = HFactory.createDefaultConsistencyLevelPolicy();
078        } else {
079          clPolicy = configurableConsistencyLevelPolicy;
080        }
081        keyspace = HFactory.createKeyspace(keyspaceName, cluster, clPolicy);
082      }

231      // probably should be typed for thrift vs. avro
232      <N> ColumnPath createColumnPath(String columnFamilyName, N columnName,
233          Serializer<N> nameSerializer) {
234        return createColumnPath(columnFamilyName, nameSerializer.toByteBuffer(columnName));
235      }
236    
237      private <N> ColumnPath createColumnPath(String columnFamilyName, ByteBuffer columnName) {
238        Validate.notNull(columnFamilyName, "columnFamilyName cannot be null");
239        ColumnPath columnPath = new ColumnPath(columnFamilyName);
240        if (columnName != null) {
241          columnPath.setColumn(columnName);
242        }
243        return columnPath;
244      }
245    
246      <N> ColumnPath createColumnPath(String columnFamilyName) {
247        return createColumnPath(columnFamilyName, null);
248      }
249    
250      <SN, N> ColumnPath createSuperColumnPath(String columnFamilyName, SN superColumnName,
251          N columnName, Serializer<SN> superNameSerializer, Serializer<N> nameSerializer) {
252        noNullElements(columnFamilyName, superColumnName, superNameSerializer, nameSerializer);
253        ColumnPath columnPath = createColumnPath(columnFamilyName, nameSerializer.toByteBuffer(columnName));
254        columnPath.setSuper_column(superNameSerializer.toByteBuffer(superColumnName));
255        return columnPath;
256      }
257    
258      <SN> ColumnPath createSuperColumnPath(String columnFamilyName, SN superColumnName,
259          Serializer<SN> superNameSerializer) {
260        noNullElements(columnFamilyName, superNameSerializer);
261        ColumnPath columnPath = createColumnPath(columnFamilyName, null);
262        if (superColumnName != null) {
263          columnPath.setSuper_column(superNameSerializer.toByteBuffer(superColumnName));
264        }
265        return columnPath;
266      }
267    
268      private void noNullElements(Object... elements) {
269        Validate.noNullElements(elements);
270      }
