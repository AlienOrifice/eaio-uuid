012    package me.prettyprint.hector.api;
013    
014    
015    import static org.junit.Assert.assertTrue;
016    import static org.junit.Assert.fail;
017    
018    import org.junit.Test;
019    
020    import me.prettyprint.cassandra.service.clock.MicrosecondsClockResolution;
021    import me.prettyprint.cassandra.service.clock.MicrosecondsSyncClockResolution;
022    import me.prettyprint.cassandra.service.clock.MillisecondsClockResolution;
023    import me.prettyprint.cassandra.service.clock.SecondsClockResolution;
024    import me.prettyprint.hector.api.factory.HFactory;
025    
026    /**
027     * @author Patricio Echague (patricioe@gmail.com)
028     *
029     */
030    public class HFactoryTest {
031    
032      @Test
033      public void testCreateClockResolution() throws Exception {
034        try {
035          HFactory.createClockResolution("ItDoesNotExist");
036          fail();
037        } catch (RuntimeException e) {
038            // good!
039        }
040    
041        assertTrue(HFactory.createClockResolution(ClockResolution.SECONDS) instanceof SecondsClockResolution);
042        assertTrue(HFactory.createClockResolution(ClockResolution.MILLISECONDS) instanceof MillisecondsClockResolution);
043        assertTrue(HFactory.createClockResolution(ClockResolution.MICROSECONDS) instanceof MicrosecondsClockResolution);
044        assertTrue(HFactory.createClockResolution(ClockResolution.MICROSECONDS_SYNC)
045                instanceof MicrosecondsSyncClockResolution);
046      }
047    }
