137   @Test
138   public void testRowDeletion() {
139     String cf = "Standard1";
140     long initialTime = keyspace.createClock();
141 
142     Mutator<String> m = createMutator(keyspace, se);
143     for (int i = 0; i < 5; i++) {
144       m.addInsertion("key" + i, cf, createColumn("name", "value" + i, se, se));
145     }
146      m.execute();
147 
148     // Try to delete the row with key "k0" with a clock previous to the insertion.
149     // Cassandra will discard this operation.
150     m.addDeletion("key0", cf, null, se, (initialTime - 100));
151     m.execute();
152 
153     // Check that the delete was harmless
154     QueryResult<HColumn<String, String>> columnResult =
155         createColumnQuery(keyspace, se, se, se).setColumnFamily(cf).setKey("key0").
156             setName("name").execute();
157     assertEquals("value0", columnResult.get().getValue());
158 
159     for (int i = 0; i < 5; i++) {
160       m.addDeletion("key" + i, cf, null, se);
161     }
162     m.execute();
163 
164     // Check that the delete took place now
165     columnResult = createColumnQuery(keyspace, se, se, se).
166         setColumnFamily(cf).setKey("key0").setName("name").execute();
167     assertNull(columnResult.get());
168   }
