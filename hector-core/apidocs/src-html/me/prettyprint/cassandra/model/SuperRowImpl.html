001    package me.prettyprint.cassandra.model;
002    
003    import java.util.List;
004    
005    import me.prettyprint.cassandra.utils.Assert;
006    import me.prettyprint.hector.api.Serializer;
007    import me.prettyprint.hector.api.beans.SuperRow;
008    import me.prettyprint.hector.api.beans.SuperSlice;
009    
010    import org.apache.cassandra.thrift.SuperColumn;
011    
012    /**
013     * A SuperRow is a touple consisting of a Key and a SuperSlice.
014     *
015     * A Row may be used to hold the returned value from queries such as get_range_slices.
016     *
017     * @author Ran Tavory
018     *
019     * @param <N>
020     *          Column name type
021     * @param <V>
022     *          Column value type
023     *
024     */
025    public final class SuperRowImpl<K, SN, N, V> implements SuperRow<K, SN, N, V>{
026    
027      private final K rowKey;
028      private final SuperSlice<SN, N, V> slice;
029    
030      /*package*/SuperRowImpl(K bs, List<SuperColumn> thriftSuperColumns,
031          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
032        Assert.noneNull(bs, thriftSuperColumns, nameSerializer, valueSerializer);
033        this.rowKey = bs;
034        slice = new SuperSliceImpl<SN, N, V>(thriftSuperColumns, sNameSerializer, nameSerializer,
035            valueSerializer);
036      }
037    
038      @Override
039      public K getKey() {
040        return rowKey;
041      }
042    
043      @Override
044      public SuperSlice<SN, N, V> getSuperSlice() {
045        return slice;
046      }
047    
048      @Override
049      public String toString() {
050        return "SuperRow(" + rowKey + "," + slice + ")";
051      }
052    }
