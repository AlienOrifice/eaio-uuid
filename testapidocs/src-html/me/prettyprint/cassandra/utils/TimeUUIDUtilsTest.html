024      /**
025       * This test must be placed FIRST. Please don't change the order.
026       * @throws Exception
027       */
028      @Test
029      public void testTimeUUIDWithClockResolution() throws Exception {
030        ClockResolution clock = new MicrosecondsClockResolution();
031        long time = clock.createClock();
032    
033        // Invoke twice with same time. Both generated UUID should be the same.
034        // Test improved algorithm.
035        assertEquals(TimeUUIDUtils.getTimeUUID(time),
036            java.util.UUID.fromString(
037                new com.eaio.uuid.UUID(UUIDGen.createTime(time), UUIDGen.getClockSeqAndNode()).toString()));
038    
039        clock = new MicrosecondsSyncClockResolution();
040        // Invoke twice with a clockResolution that guarantees unique timestamp. The second must be greater
041        // than the first one.
042        java.util.UUID first = TimeUUIDUtils.getTimeUUID(clock);
043        java.util.UUID second = TimeUUIDUtils.getTimeUUID(clock);
044        assertTrue(second.compareTo(first) > 0);
045      }
046    
047      @Test
048      public void testTimeUUIDAsByteArray() {
049        // Generate UUID, convert to array and back to UUID. Then assert.
050        UUID uuid = TimeUUIDUtils.getUniqueTimeUUIDinMillis();
051        UUID uuidAfterConversion = TimeUUIDUtils.toUUID(TimeUUIDUtils.asByteArray(uuid));
052        assertEquals(uuid, uuidAfterConversion);
053    
054        // Used the previously generated UUID, convert to array and back to UUID. Then compare their times.
055        long timeInUUID = TimeUUIDUtils.getTimeFromUUID(TimeUUIDUtils.asByteArray(uuid));
056        assertEquals(uuid.timestamp(), timeInUUID);
057      }
058    
059      @Test
060      public void testTimeUUIDAsByteBuffer() {
061        UUID expectedUuid = TimeUUIDUtils.getUniqueTimeUUIDinMillis();
062        UUID actualUuid = TimeUUIDUtils.uuid(TimeUUIDUtils.asByteBuffer(expectedUuid));
063        assertEquals(expectedUuid, actualUuid);
064      }
