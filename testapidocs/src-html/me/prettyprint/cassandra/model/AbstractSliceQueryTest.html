015    import org.apache.cassandra.thrift.SlicePredicate;
016    import org.apache.cassandra.thrift.SliceRange;
017    import org.junit.Test;
018    import org.mockito.Mockito;
019    
020    public class AbstractSliceQueryTest {
021    
022    
023      private static final StringSerializer se = StringSerializer.get();
024      private static final Keyspace ko = Mockito.mock(ExecutingKeyspace.class);
025    
026      @Test
027      public void testGetSetPredicate_columnNames() {
028        ConcreteSliceQueury<String, String, String, Rows<String, String, String>> q =
029          new ConcreteSliceQueury<String, String, String, Rows<String, String,String>>(ko, se, se, se);
030        q.setColumnNames("1", "2", "3");
031        SlicePredicate p = q.getPredicate();
032        assertEquals(3, p.getColumn_names().size());
033        assertNull(p.getSlice_range());
034      }
035    
036      @Test
037      public void testGetSetPredicate_range() {
038        ConcreteSliceQueury<String, String, String, Rows<String, String, String>> q =
039          new ConcreteSliceQueury<String, String, String, Rows<String, String,String>>(ko, se, se, se);
040        q.setRange("1", "100", false, 10);
041        SlicePredicate p = q.getPredicate();
042        assertNull(p.getColumn_names());
043        SliceRange range = p.getSlice_range();
044        assertNotNull(range);
045        assertArrayEquals(bytes("1"), range.getStart());
046        assertArrayEquals(bytes("100"), range.getFinish());
047        assertEquals(10, range.getCount());
048      }
049    
050      private static class ConcreteSliceQueury<K, N, V, T> extends AbstractSliceQuery<K, N, V, T> {
051        ConcreteSliceQueury(Keyspace k, Serializer<K> keySerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
052          super(k, keySerializer, nameSerializer, valueSerializer);
053        }
054    
055        @Override
056        public QueryResult<T> execute() {
057          return null;
058        }
059      }
060    }
