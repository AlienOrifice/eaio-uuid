53    public HConnectionManager(CassandraHostConfigurator cassandraHostConfigurator) {
54      loadBalancingPolicy = cassandraHostConfigurator.getLoadBalancingPolicy();
55      clock = cassandraHostConfigurator.getClockResolution();
56      hostPools = new NonBlockingHashMap<CassandraHost, ConcurrentHClientPool>();
57      if ( cassandraHostConfigurator.getRetryDownedHosts() ) {
58        cassandraHostRetryService = new CassandraHostRetryService(this, cassandraHostConfigurator);
59      }
60      for ( CassandraHost host : cassandraHostConfigurator.buildCassandraHosts() ) {
61        try {
62          ConcurrentHClientPool chcp = new ConcurrentHClientPool(host);
63          hostPools.put(host,chcp);
64        } catch (HectorTransportException hte) {
65          log.error("Could not start connection pool for host {}", host);
66          if ( cassandraHostRetryService != null ) {
67            cassandraHostRetryService.add(host);
68          }
69        }
70      }
71  
72      if ( cassandraHostConfigurator.getAutoDiscoverHosts() ) {
73        nodeAutoDiscoverService = new NodeAutoDiscoverService(this, cassandraHostConfigurator);
74      }
75      monitor = JmxMonitor.getInstance(this).getCassandraMonitor();
76      exceptionsTranslator = new ExceptionsTranslatorImpl();
77    }

79    /**
80     * Returns true if the host was successfully added. In any sort of failure exceptions are 
81     * caught and logged, returning false.
82     * @param cassandraHost
83     * @return
84     */
85    public boolean addCassandraHost(CassandraHost cassandraHost) {
86      if ( !getHosts().contains(cassandraHost) ) {
87        ConcurrentHClientPool pool = null;
88        try {
89          pool = new ConcurrentHClientPool(cassandraHost);
90          hostPools.put(cassandraHost, pool);
91          log.info("Added host {} to pool", cassandraHost.getName());
92          return true;
93        } catch (HectorTransportException hte) {
94          log.error("Transport exception host to HConnectionManager: " + cassandraHost, hte);
95        } catch (Exception ex) {
96          log.error("General exception host to HConnectionManager: " + cassandraHost, ex);
97        }
98      } else {
99        log.info("Host already existed for pool {}", cassandraHost.getName());
100     }
101     return false;
102   }

104   /**
105    * Remove the {@link CassandraHost} from the pool, bypassing retry service. This
106    * would be called on a host that is known to be going away. Gracefully shuts down
107    * the underlying connections via {@link ConcurrentHClientPool#shutdown()}
108    * @param cassandraHost
109    */
110   public void removeCassandraHost(CassandraHost cassandraHost) {
111     boolean removed = getHosts().contains(cassandraHost);
112     if ( removed ) {
113       ConcurrentHClientPool pool = hostPools.remove(cassandraHost);
114       if ( pool != null ) {
115         pool.shutdown();
116       } else {
117         removed = false;
118         log.info("removeCassandraHost attempt miss for CassandraHost {} May have been beaten by another thread?", cassandraHost);
119       }
120     }
121     log.info("Remove status for CassandraHost pool {} was {}", cassandraHost, removed);
122   }

137   public void operateWithFailover(Operation<?> op) throws HectorException {
138     final StopWatch stopWatch = new Slf4JStopWatch(perf4jLogger);
139     int retries = Math.min(op.failoverPolicy.numRetries, hostPools.size());
140     HThriftClient client = null;
141     boolean success = false;
142     boolean retryable = false;
143     Set<CassandraHost> excludeHosts = new HashSet<CassandraHost>();
144     // TODO start timer for limiting retry time spent
145     while ( !success ) {
146       try {
147         // TODO how to 'timeout' on this op when underlying pool is exhausted
148         client =  getClientFromLBPolicy(excludeHosts);
149         Cassandra.Client c = client.getCassandra(op.keyspaceName);
150         // Keyspace can be null for some system_* api calls
151         if ( !op.credentials.isEmpty() ) {
152           c.login(new AuthenticationRequest(op.credentials));
153         }
154 
155         op.executeAndSetResult(c, client.cassandraHost);
156         success = true;
157         stopWatch.stop(op.stopWatchTagName + ".success_");
158         break;
159 
160       } catch (Exception ex) {
161         HectorException he = exceptionsTranslator.translate(ex);
162         if ( he instanceof HInvalidRequestException || he instanceof HCassandraInternalException ) {
163           throw he;
164         } else if ( he instanceof HUnavailableException || he instanceof HectorTransportException) {
165           --retries;
166           client.close();
167           markHostAsDown(client);
168           excludeHosts.add(client.cassandraHost);
169           retryable = true;
170           if ( retries > 0 ) {
171             monitor.incCounter(Counter.RECOVERABLE_TRANSPORT_EXCEPTIONS);
172           }
173         } else if ( he instanceof HTimedOutException ) {
174           // DO NOT drecrement retries, we will be keep retrying on timeouts until it comes back
175           retryable = true;
176           monitor.incCounter(Counter.RECOVERABLE_TIMED_OUT_EXCEPTIONS);
177           client.close();
178           // TODO timecheck on how long we've been waiting on timeouts here
179           // suggestion per user moores on hector-users
180         } else if ( he instanceof PoolExhaustedException ) {
181           retryable = true;
182           --retries;
183           if ( hostPools.size() == 1 ) {
184             throw he;
185           }
186           monitor.incCounter(Counter.POOL_EXHAUSTED);
187           excludeHosts.add(client.cassandraHost);
188         }
189         if ( retries == 0 || retryable == false) throw he;
190         log.error("Could not fullfill request on this host {}", client);
191         log.error("Exception: ", he);
192         monitor.incCounter(Counter.SKIP_HOST_SUCCESS);
193         sleepBetweenHostSkips(op.failoverPolicy);
194       } finally {
195         if ( !success ) {
196           monitor.incCounter(op.failCounter);
197           stopWatch.stop(op.stopWatchTagName + ".fail_");
198         }
199         releaseClient(client);
200       }
201     }
202   }

204   /**
205   * Sleeps for the specified time as determined by sleepBetweenHostsMilli.
206   * In many cases failing over to other hosts is done b/c the cluster is too busy, so the sleep b/w
207   * hosts may help reduce load on the cluster.
208   */
209     private void sleepBetweenHostSkips(FailoverPolicy failoverPolicy) {
210       if (failoverPolicy.sleepBetweenHostsMilli > 0) {
211         if ( log.isDebugEnabled() ) {
212           log.debug("Will sleep for {} millisec", failoverPolicy.sleepBetweenHostsMilli);
213         }
214         try {
215           Thread.sleep(failoverPolicy.sleepBetweenHostsMilli);
216         } catch (InterruptedException e) {
217           log.warn("Sleep between hosts interrupted", e);
218         }
219       }
220     }

222   private HThriftClient getClientFromLBPolicy(Set<CassandraHost> excludeHosts) {
223     HThriftClient client;
224     if ( hostPools.size() == 0 ) {
225       throw new HectorException("All host pools marked down. Retry burden pushed out to client.");
226     }
227     try {
228       client = loadBalancingPolicy.getPool(hostPools.values(), excludeHosts).borrowClient();
229     } catch (Exception e) {
230       throw new HectorException("General exception in getClientFromLBPolicy",e);
231     }
232     return client;
233   }

235   void releaseClient(HThriftClient client) {
236     if ( client == null ) return;
237     ConcurrentHClientPool pool = hostPools.get(client.cassandraHost);
238     if ( pool != null ) {
239       pool.releaseClient(client);
240     } else {
241       log.info("Client {} released to inactive or dead pool. Closing.", client);
242       client.close();
243     }
244   }

250   void markHostAsDown(HThriftClient client) {
251     log.error("MARK HOST AS DOWN TRIGGERED for host {}", client.cassandraHost.getName());
252     ConcurrentHClientPool pool = hostPools.remove(client.cassandraHost);
253     if ( pool != null ) {
254       log.error("Pool state on shutdown: {}", pool.getStatusAsString());
255       pool.shutdown();
256       cassandraHostRetryService.add(client.cassandraHost);
257     }
258     client.close();
259   }

273   public void shutdown() {
274     log.info("Shutdown called on HConnectionManager");
275     if ( cassandraHostRetryService != null )
276       cassandraHostRetryService.shutdown();
277     if ( nodeAutoDiscoverService != null )
278       nodeAutoDiscoverService.shutdown();
279 
280     for (ConcurrentHClientPool pool : hostPools.values()) {
281       try {
282         pool.shutdown();
283       } catch (IllegalArgumentException iae) {
284         log.error("Out of order in HConnectionManager shutdown()?: {}", iae.getMessage());
285       }
286     }
