001    package me.prettyprint.hector.api;
002    
003    import java.nio.ByteBuffer;
004    import java.util.List;
005    import java.util.Map;
006    import java.util.Set;
007    
008    import me.prettyprint.cassandra.serializers.StringSerializer;
009    
010    
011    /**
012     * Serializes a type T from the given bytes, or vice a versa.
013     *
014     * In cassandra column names and column values (and starting with 0.7.0 row keys) are all byte[].
015     * To allow type safe conversion in java and keep all conversion code in one place we define the
016     * Extractor interface.
017     * Implementors of the interface define type conversion according to their domains. A predefined
018     * set of common extractors can be found in the extractors package, for example
019     * {@link StringSerializer}.
020     *
021     * @author Ran Tavory
022     *
023     * @param <T> The type to which data extraction should work.
024     */
025    public interface Serializer<T> {
026    
027      /**
028       * Extract bytes from the obj of type T
029       * @param obj
030       * @return
031       */  
032      public ByteBuffer toByteBuffer(T obj);  
033    
034      public byte[] toBytes(T obj);
035      
036      public T fromBytes(byte[] bytes);
037      
038      /**
039       * Extract an object of type T from the bytes.
040       * @param bytes
041       * @return
042       */  
043      public T fromByteBuffer(ByteBuffer byteBuffer);
044    
045      public Set<ByteBuffer> toBytesSet(List<T> list);
046    
047      public List<T> fromBytesSet(Set<ByteBuffer> list);
048    
049      public <V> Map<ByteBuffer, V> toBytesMap(Map<T, V> map);
050    
051      public <V> Map<T, V> fromBytesMap(Map<ByteBuffer, V> map);
052    
053      public List<ByteBuffer> toBytesList(List<T> list);
054    
055      public List<T> fromBytesList(List<ByteBuffer> list);
056    
057    }
