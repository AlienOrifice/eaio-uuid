001    package me.prettyprint.hector.api.query;
002    
003    import java.util.Collection;
004    
005    import me.prettyprint.hector.api.beans.Rows;
006    
007    /**
008     * A query wrapper for the multiget_slice call
009     *
010     * @author ran
011     */
012    public interface MultigetSliceQuery<K, N, V> extends Query<Rows<K, N, V>> {
013    
014      MultigetSliceQuery<K, N, V> setKeys(K... keys);
015      
016      MultigetSliceQuery<K, N, V> setKeys(Iterable<K> keys);
017    
018      /**
019       * Sets the column names to be retrieved by this query
020       * @param columns a list of column names
021       */
022      MultigetSliceQuery<K, N, V> setColumnNames(N... columnNames);
023    
024      Collection<N> getColumnNames();
025    
026      MultigetSliceQuery<K, N, V> setColumnFamily(String cf);
027    
028      /**
029       * Set a predicate of start/finish to retrieve a list of columns in this range.
030       *
031       * @param start Start key
032       * @param finish End key
033       * @param reversed
034       * @param count
035       * @return
036       */
037      MultigetSliceQuery<K, N, V> setRange(N start, N finish, boolean reversed, int count);
038    }
