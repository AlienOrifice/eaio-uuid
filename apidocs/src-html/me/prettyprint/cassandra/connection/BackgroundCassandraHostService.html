001    package me.prettyprint.cassandra.connection;
002    
003    import java.util.concurrent.Executors;
004    import java.util.concurrent.ScheduledExecutorService;
005    import java.util.concurrent.ScheduledFuture;
006    
007    import me.prettyprint.cassandra.service.CassandraHostConfigurator;
008    
009    public abstract class BackgroundCassandraHostService {
010    
011      protected ScheduledExecutorService executor = Executors.newSingleThreadScheduledExecutor();
012    
013      protected final HConnectionManager connectionManager;
014      protected final CassandraHostConfigurator cassandraHostConfigurator;
015    
016      protected ScheduledFuture sf;
017      protected int retryDelayInSeconds;
018    
019      public BackgroundCassandraHostService(HConnectionManager connectionManager,
020          CassandraHostConfigurator cassandraHostConfigurator) {
021        this.connectionManager = connectionManager;
022        this.cassandraHostConfigurator = cassandraHostConfigurator;
023        Runtime.getRuntime().addShutdownHook(new Thread() {
024          @Override
025          public void run() {
026            shutdown();
027          }
028        });
029      }
030    
031      abstract void shutdown();
032    
033      abstract void applyRetryDelay();
034    
035    
036    
037      public int getRetryDelayInSeconds() {
038        return retryDelayInSeconds;
039      }
040    
041      public void setRetryDelayInSeconds(int retryDelayInSeconds) {
042        this.retryDelayInSeconds = retryDelayInSeconds;
043      }
044    
045    }
