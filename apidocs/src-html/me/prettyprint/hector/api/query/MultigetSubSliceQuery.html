001    package me.prettyprint.hector.api.query;
002    
003    import java.util.Collection;
004    
005    import me.prettyprint.hector.api.beans.Rows;
006    
007    /**
008     * A query wrapper for the call multiget_slice for subcolumns  of supercolumns
009     *
010     * @author ran
011     */
012    public interface MultigetSubSliceQuery<K, SN, N, V> extends Query<Rows<K, N, V>>{
013    
014      MultigetSubSliceQuery<K, SN, N, V> setKeys(K... keys);
015    
016      /**
017       * Set the supercolumn to run the slice query on
018       */
019      MultigetSubSliceQuery<K, SN, N, V> setSuperColumn(SN superColumn);
020    
021      MultigetSubSliceQuery<K, SN, N, V> setColumnFamily(String cf);
022    
023      MultigetSubSliceQuery<K, SN, N, V> setRange(N start, N finish, boolean reversed, int count);
024    
025      Collection<N> getColumnNames();
026    
027      /**
028       * Sets the column names to be retrieved by this query
029       * @param columns a list of column names
030       */
031      MultigetSubSliceQuery<K, SN, N, V> setColumnNames(N... columnNames);
032    
033    }
