018    public class BasicKeyspaceDefinition implements KeyspaceDefinition {
019    
020      private String name;
021      private String strategyClass;
022      private int replicationFactor;
023      private Map<String, String> strategyOptions;
024      private List<ColumnFamilyDefinition> columnFamilyList;
025    
026      public BasicKeyspaceDefinition(){
027        this.columnFamilyList = new ArrayList<ColumnFamilyDefinition>();
028        this.strategyOptions = new HashMap<String, String>();
029      }
030    
031      @Override
032      public String getName() {
033        return this.name;
034      }
035    
036      @Override
037      public String getStrategyClass() {
038        return this.strategyClass;
039      }
040    
041      @Override
042      public Map<String, String> getStrategyOptions() {
043        return this.strategyOptions;
044      }
045    
046      @Override
047      public int getReplicationFactor() {
048        return this.replicationFactor;
049      }
050    
051      @Override
052      public List<ColumnFamilyDefinition> getCfDefs() {
053        return this.columnFamilyList;
054      }
055    
056      public void setName(String name) {
057        this.name = name;
058      }
059    
060      public void setStrategyClass(String strategyClass) {
061        this.strategyClass = strategyClass;
062      }
063    
064      public void setReplicationFactor(int replicationFactor) {
065        this.replicationFactor = replicationFactor;
066      }
067    
068      public void addColumnFamily( ColumnFamilyDefinition columnFamilyDefinition ){
069        this.columnFamilyList.add(columnFamilyDefinition);
070      }
071    
072      public void removeColumnFamily( ColumnFamilyDefinition columnFamilyDefinition ){
073        this.columnFamilyList.remove(columnFamilyDefinition);
074      }
075    
076      public void setStrategyOption( String field, String value ){
077        this.strategyOptions.put( field, value);
078      }
079    
080      public void removeStrategyOption( String field ){
081        this.strategyOptions.remove( field );
082      }
083    
084    }
