23    @Override
24    public HectorException translate(Throwable original) {
25      if (original instanceof HectorException) {
26        return (HectorException) original;
27      } else if (original instanceof TApplicationException) {
28        return new HCassandraInternalException(((TApplicationException)original).getType(), original.getMessage());
29      } else if (original instanceof TException || original instanceof TTransportException) {
30        return new HectorTransportException(original);
31      } else if (original instanceof org.apache.cassandra.thrift.TimedOutException) {
32        return new HTimedOutException(original);
33      } else if (original instanceof org.apache.cassandra.thrift.InvalidRequestException) {
34        // See bug https://issues.apache.org/jira/browse/CASSANDRA-1862
35        // If a bootstrapping node is accessed it responds with IRE. It makes more sense to throw
36        // UnavailableException.
37        // Hector wraps this caveat and fixes this.
38        String why = ((org.apache.cassandra.thrift.InvalidRequestException) original).getWhy();
39        if (why != null && why.contains("bootstrap")) {
40          return new HUnavailableException(original);
41        }
42        HInvalidRequestException e = new HInvalidRequestException(original);
43        e.setWhy(why);
44        return e;
45      } else if (original instanceof TProtocolException) {
46        return new HInvalidRequestException(original);
47      } else if (original instanceof org.apache.cassandra.thrift.NotFoundException) {
48        return new HNotFoundException(original);
49      } else if (original instanceof org.apache.cassandra.thrift.UnavailableException) {
50        return new HUnavailableException(original);
51      } else if (original instanceof NoSuchElementException) {
52        return new PoolExhaustedException(original);
53      } else if (original instanceof IllegalStateException) {
54        return new PoolIllegalStateException(original);
55      } else {
56        return new HectorException(original);
57      }
58    }
