001    package me.prettyprint.hector.api.beans;
002    
003    import java.nio.ByteBuffer;
004    import java.util.List;
005    
006    import me.prettyprint.hector.api.Serializer;
007    
008    /**
009     * Models a SuperColumn in a protocol independant manner
010     *
011     * @param <SN>
012     *          SuperColumn name type
013     * @param <N>
014     *          Column name type
015     * @param <V>
016     *          Column value type
017     *
018     * @author zznate
019     */
020    public interface HSuperColumn<SN, N, V> {
021    
022      HSuperColumn<SN, N, V> setName(SN name);
023    
024      HSuperColumn<SN, N, V> setSubcolumns(List<HColumn<N, V>> subcolumns);
025    
026      HSuperColumn<SN, N, V> setClock(long clock);
027    
028      public long getClock();
029    
030      int getSize();
031    
032      SN getName();
033    
034      /**
035       *
036       * @return an unmodifiable list of columns
037       */
038      List<HColumn<N, V>> getColumns();
039    
040      HColumn<N, V> get(int i);
041    
042      Serializer<SN> getNameSerializer();
043    
044      byte[] getNameBytes();
045      
046      ByteBuffer getNameByteBuffer();
047    
048      Serializer<SN> getSuperNameSerializer();
049    
050      Serializer<V> getValueSerializer();
051    
052    }
