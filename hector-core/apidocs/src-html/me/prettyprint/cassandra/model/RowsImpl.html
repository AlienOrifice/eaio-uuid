001    package me.prettyprint.cassandra.model;
002    
003    import java.util.HashMap;
004    import java.util.Iterator;
005    import java.util.LinkedHashMap;
006    import java.util.List;
007    import java.util.Map;
008    
009    import me.prettyprint.cassandra.utils.Assert;
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.beans.Row;
012    import me.prettyprint.hector.api.beans.Rows;
013    
014    import org.apache.cassandra.thrift.Column;
015    
016    /**
017     * Returned by a MultigetSliceQuery (multiget_slice thrift call)
018     *
019     * @author Ran Tavory
020     *
021     * @param <N>
022     * @param <V>
023     */
024    public class RowsImpl<K, N, V> implements Rows<K, N, V> {
025    
026      protected final Map<K, Row<K, N, V>> rows;
027    
028      public RowsImpl(Map<K, List<Column>> thriftRet, Serializer<N> nameSerializer,
029          Serializer<V> valueSerializer) {
030        Assert.noneNull(thriftRet, nameSerializer, valueSerializer);
031        rows = new LinkedHashMap<K, Row<K, N, V>>(thriftRet.size());
032        for (Map.Entry<K, List<Column>> entry : thriftRet.entrySet()) {
033          rows.put(entry.getKey(), new RowImpl<K, N, V>(entry.getKey(), entry.getValue(), nameSerializer,
034              valueSerializer));
035        }
036      }
037    
038      @Override
039      public Row<K, N, V> getByKey(K key) {
040        return rows.get(key);
041      }
042    
043      @Override
044      public int getCount() {
045        return rows.size();
046      }
047    
048      @Override
049      public Iterator<Row<K, N, V>> iterator() {
050        return rows.values().iterator();
051      }
052    
053      @Override
054      public String toString() {
055        return "Rows(" + rows + ")";
056      }
057    }
