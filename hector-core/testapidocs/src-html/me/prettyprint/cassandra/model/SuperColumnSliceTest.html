001    package me.prettyprint.cassandra.model;
002    
003    import java.nio.ByteBuffer;
004    import java.util.ArrayList;
005    import java.util.Arrays;
006    import java.util.List;
007    
008    import junit.framework.Assert;
009    import me.prettyprint.cassandra.serializers.ByteBufferSerializer;
010    import me.prettyprint.cassandra.serializers.LongSerializer;
011    import me.prettyprint.cassandra.serializers.StringSerializer;
012    import me.prettyprint.hector.api.beans.SuperSlice;
013    
014    import org.apache.cassandra.thrift.Column;
015    import org.apache.cassandra.thrift.SuperColumn;
016    import org.junit.Test;
017    
018    /**
019     *
020     * @author Ran Tavory
021     *
022     */
023    public class SuperColumnSliceTest {
024    
025      @Test
026      public void testConstruction() {
027        LongSerializer le = LongSerializer.get();
028        StringSerializer se = StringSerializer.get();
029        ByteBufferSerializer be = ByteBufferSerializer.get();
030        // empty one
031        List<SuperColumn> tColumns = new ArrayList<SuperColumn>();
032        SuperSlice<String, Long, ByteBuffer> slice = new SuperSliceImpl<String, Long, ByteBuffer>(tColumns, se, le,
033            be);
034        Assert.assertTrue(slice.getSuperColumns().isEmpty());
035    
036        // non-empty one
037        Column c = new Column(le.toByteBuffer(5L), be.toByteBuffer(ByteBuffer.wrap(new byte[] { 1 })), 2L);
038        tColumns.add(new SuperColumn(se.toByteBuffer("super"), Arrays.asList(c)));
039        slice = new SuperSliceImpl<String, Long, ByteBuffer>(tColumns, se, le, be);
040        Assert.assertEquals(1, slice.getSuperColumns().size());
041        Assert.assertEquals((Long) 5L, slice.getColumnByName("super").get(0).getName());
042      }
043    }
