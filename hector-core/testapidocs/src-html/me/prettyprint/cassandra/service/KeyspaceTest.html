180        Map<String, Map<String, List<Mutation>>> outerMutationMap = new HashMap<String, Map<String,List<Mutation>>>();
181        for (int i = 0; i < 10; i++) {
182    
183          Map<String, List<Mutation>> mutationMap = new HashMap<String, List<Mutation>>();
184    
185          ArrayList<Mutation> mutations = new ArrayList<Mutation>(10);
186          for (int j = 0; j < 10; j++) {
187            Column col = new Column(StringSerializer.get().toByteBuffer("testBatchMutateColumn_" + j),
188                StringSerializer.get().toByteBuffer("testBatchMutateColumn_value_" + j), connectionManager.createClock());
189            //list.add(col);
190            ColumnOrSuperColumn cosc = new ColumnOrSuperColumn();
191            cosc.setColumn(col);
192            Mutation mutation = new Mutation();
193            mutation.setColumn_or_supercolumn(cosc);
194            mutations.add(mutation);
195          }
196          mutationMap.put("Standard1", mutations);
197          outerMutationMap.put("testBatchMutateColumn_" + i, mutationMap);
198        }
199        keyspace.batchMutate(se.toBytesMap(outerMutationMap));
200        // re-use later
201        outerMutationMap.clear();
202    
203        // get value
204        for (int i = 0; i < 10; i++) {
205          for (int j = 0; j < 10; j++) {
206            ColumnPath cp = new ColumnPath("Standard1");
207            cp.setColumn(bytes("testBatchMutateColumn_" + j));
208    
209            Column col = keyspace.getColumn("testBatchMutateColumn_" + i, cp);
210            assertNotNull(col);
211            String value = string(col.getValue());
212            assertEquals("testBatchMutateColumn_value_" + j, value);
213    
214          }
215        }
216    
217        // batch_mutate delete by key
218        for (int i = 0; i < 10; i++) {
219          ArrayList<Mutation> mutations = new ArrayList<Mutation>(10);
220          Map<String, List<Mutation>> mutationMap = new HashMap<String, List<Mutation>>();
221          SlicePredicate slicePredicate = new SlicePredicate();
222          for (int j = 0; j < 10; j++) {
223            slicePredicate.addToColumn_names(StringSerializer.get().toByteBuffer("testBatchMutateColumn_" + j));
224          }
225          Mutation mutation = new Mutation();
226          Deletion deletion = new Deletion(connectionManager.createClock());
227          deletion.setPredicate(slicePredicate);
228          mutation.setDeletion(deletion);
229          mutations.add(mutation);
230    
231          mutationMap.put("Standard1", mutations);
232          outerMutationMap.put("testBatchMutateColumn_"+i, mutationMap);
233        }
234        keyspace.batchMutate(se.toBytesMap(outerMutationMap));
235        // make sure the values are gone
236        for (int i = 0; i < 10; i++) {
237          for (int j = 0; j < 10; j++) {
238            ColumnPath cp = new ColumnPath("Standard1");
239            cp.setColumn(bytes("testBatchMutateColumn_" + j));
240            try {
241              keyspace.getColumn("testBatchMutateColumn_" + i, cp);
242              fail();
243            } catch (HNotFoundException e) {
244            }
245    
246          }
247        }

806            keyspace.insert("testMultigetCount_key_"+j, cp, StringSerializer.get().toByteBuffer("testMultigetCount_value_" + i));
807          }
808          if (j % 2 == 0) {
809            keys.add(StringSerializer.get().toByteBuffer("testMultigetCount_key_"+j));
810          }
811        }
812

819        assertEquals(5,counts.size());
