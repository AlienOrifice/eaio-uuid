001    package me.prettyprint.cassandra.model;
002    
003    import java.util.Arrays;
004    import java.util.List;
005    import java.util.Map;
006    
007    import me.prettyprint.cassandra.service.KeyspaceService;
008    import me.prettyprint.cassandra.utils.Assert;
009    import me.prettyprint.hector.api.Keyspace;
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.exceptions.HectorException;
012    import me.prettyprint.hector.api.query.Query;
013    import me.prettyprint.hector.api.query.QueryResult;
014    
015    import org.apache.cassandra.thrift.ColumnParent;
016    
017    public class MultigetCountQuery<K,N> implements Query<Map<K, Integer>> {
018    
019      protected final ExecutingKeyspace keyspace;
020      protected final Serializer<K> keySerializer;
021      protected String columnFamily;
022      protected List<K> keys;
023    
024      /** The slice predicate for which the count it performed*/
025      protected final HSlicePredicate<N> slicePredicate;
026    
027      public MultigetCountQuery(Keyspace k, Serializer<K> keySerializer,
028          Serializer<N> nameSerializer) {
029        Assert.notNull(k, "keyspace can't be null");
030        Assert.notNull(keySerializer, "keySerializer can't be null");
031        Assert.notNull(nameSerializer, "columnNameSerializer is null");
032        this.keyspace = (ExecutingKeyspace) k;
033        this.keySerializer = keySerializer;
034        this.slicePredicate = new HSlicePredicate<N>(nameSerializer);
035      }
036    
037      public MultigetCountQuery<K, N> setKeys(K... keys) {
038        this.keys = Arrays.asList(keys);
039        return this;
040      }
041    
042      public MultigetCountQuery<K,N> setColumnFamily(String cf) {
043        this.columnFamily = cf;
044        return this;
045      }
046    
047      public MultigetCountQuery<K,N> setColumnNames(N... columnNames) {
048        slicePredicate.setColumnNames(columnNames);
049        return this;
050      }
051    
052      public MultigetCountQuery<K,N> setRange(N start, N finish, int count) {
053        slicePredicate.setRange(start, finish, false, count);
054        return this;
055      }
056    
057      @Override
058      public QueryResult<Map<K, Integer>> execute() {
059        Assert.notNull(keys, "keys list is null");
060        Assert.notNull(columnFamily, "columnFamily is null");
061        return new QueryResultImpl<Map<K,Integer>>(keyspace.doExecute(
062            new KeyspaceOperationCallback<Map<K,Integer>>() {
063              @Override
064              public Map<K,Integer> doInKeyspace(KeyspaceService ks) throws HectorException {
065                ColumnParent columnParent = new ColumnParent(columnFamily);
066                Map<K,Integer> counts = keySerializer.fromBytesMap(
067                    ks.multigetCount(keySerializer.toBytesList(keys), columnParent, slicePredicate.toThrift()));
068                return counts;
069              }
070            }), this);
071      }
072    
073      @Override
074      public String toString() {
075       return String.format("MultigetCountQuery(%s) on cf: %s with keys: %s",slicePredicate.toString(),columnFamily,keys.toString());
076      }
077    
078    }
