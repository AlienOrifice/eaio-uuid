001    package me.prettyprint.cassandra.model;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.nio.ByteBuffer;
006    
007    import junit.framework.Assert;
008    import me.prettyprint.cassandra.serializers.LongSerializer;
009    import me.prettyprint.cassandra.serializers.StringSerializer;
010    import me.prettyprint.hector.api.beans.ColumnSlice;
011    
012    import org.apache.cassandra.thrift.Column;
013    import org.junit.Test;
014    
015    
016    /**
017     *
018     * @author Ran Tavory
019     *
020     */
021    public class ColumnSliceTest {
022    
023    
024      @Test
025      public void testConstruction() {
026        StringSerializer se = StringSerializer.get();
027        LongSerializer le = LongSerializer.get();
028        List<Column> tColumns = new ArrayList<Column>();
029        ColumnSlice<String, Long> slice = new ColumnSliceImpl<String, Long>(tColumns, se, le);
030        Assert.assertTrue(slice.getColumns().isEmpty());
031    
032        tColumns.add(new Column(ByteBuffer.wrap(new byte[]{}), ByteBuffer.wrap(new byte[]{}), 0L));
033        slice = new ColumnSliceImpl<String, Long>(tColumns, se, le);
034        Assert.assertEquals(1, slice.getColumns().size());
035    
036        tColumns = new ArrayList<Column>();
037        tColumns.add(new Column(se.toByteBuffer("1"), le.toByteBuffer(1L), 0L));
038        slice = new ColumnSliceImpl<String, Long>(tColumns, se, le);
039        Assert.assertEquals((Long) 1L, slice.getColumnByName("1").getValue());
040      }
041    }
