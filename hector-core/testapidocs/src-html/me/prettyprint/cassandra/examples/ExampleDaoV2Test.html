001    package me.prettyprint.cassandra.examples;
002    
003    import static me.prettyprint.hector.api.factory.HFactory.createKeyspace;
004    import static me.prettyprint.hector.api.factory.HFactory.getOrCreateCluster;
005    import static org.junit.Assert.assertEquals;
006    import static org.junit.Assert.assertNotNull;
007    import static org.junit.Assert.assertNull;
008    
009    import java.io.IOException;
010    import java.util.HashMap;
011    import java.util.Map;
012    
013    import me.prettyprint.cassandra.serializers.StringSerializer;
014    import me.prettyprint.cassandra.testutils.EmbeddedServerHelper;
015    import me.prettyprint.hector.api.Cluster;
016    import me.prettyprint.hector.api.exceptions.HectorException;
017    
018    import org.apache.cassandra.config.ConfigurationException;
019    import org.apache.thrift.transport.TTransportException;
020    import org.junit.AfterClass;
021    import org.junit.BeforeClass;
022    import org.junit.Test;
023    
024    public class ExampleDaoV2Test {
025    
026      private static EmbeddedServerHelper embedded;
027    
028      /**
029       * Set embedded cassandra up and spawn it in a new thread.
030       *
031       * @throws TTransportException
032       * @throws IOException
033       * @throws InterruptedException
034       */
035      @BeforeClass
036      public static void setup() throws TTransportException, IOException, InterruptedException, ConfigurationException {
037        embedded = new EmbeddedServerHelper();
038        embedded.setup();
039      }
040    
041      @AfterClass
042      public static void teardown() throws IOException {
043        embedded.teardown();
044      }
045    
046      @Test
047      public void testInsertGetDelete() throws HectorException {
048        Cluster c = getOrCreateCluster("MyCluster", "localhost:9170");
049        ExampleDaoV2 dao = new ExampleDaoV2(createKeyspace("Keyspace1", c));
050        assertNull(dao.get("key", StringSerializer.get()));
051        dao.insert("key", "value", StringSerializer.get());
052        assertEquals("value", dao.get("key", StringSerializer.get()));
053        dao.delete(StringSerializer.get(), "key");
054        assertNull(dao.get("key", StringSerializer.get()));
055      }
056    
057      @Test
058      public void testMultiInsertGetDelete() throws HectorException {
059        Cluster c = getOrCreateCluster("MyCluster", "localhost:9170");
060        ExampleDaoV2 dao = new ExampleDaoV2(createKeyspace("Keyspace1", c));
061    
062        // Get non-existing values
063        Map<String, String> ret = dao.getMulti(StringSerializer.get(), "key1", "key2");
064        assertNotNull(ret);
065        assertNull("value1", ret.get("key1"));
066    
067        // Insert values
068        Map<String, String> keyValues = new HashMap<String, String>();
069        keyValues.put("key1", "value1");
070        keyValues.put("key2", "value2");
071        dao.insertMulti(keyValues, StringSerializer.get());
072    
073        // Simple get test
074        ret = dao.getMulti(StringSerializer.get(), "key1", "key2");
075        assertNotNull(ret);
076        assertEquals("value1", ret.get("key1"));
077        assertEquals("value2", ret.get("key2"));
078    
079        // Get some values that don't exist
080        ret = dao.getMulti(StringSerializer.get(), "key2", "key3");
081        assertNotNull(ret);
082        assertEquals("value2", ret.get("key2"));
083        assertNull(ret.get("key3"));
084        assertNull(ret.get("key1"));
085    
086        // delete
087        dao.delete(StringSerializer.get(), "key1", "key2");
088    
089        // validate deletion
090        ret = dao.getMulti(StringSerializer.get(), "key1", "key2");
091        assertNotNull(ret);
092        assertNull(ret.get("key1"));
093        assertNull(ret.get("key2"));
094    
095      }
096    }
