004    import java.net.UnknownHostException;

009    import org.apache.commons.pool.impl.GenericObjectPool;

013    /**
014     * Encapsulates the information required for connecting to a Cassandra host.
015     * Also exposes pool configuration parameters for that host.
016     *
017     * @author zznate(nate@riptano.com)
018     *
019     */
020    public final class CassandraHost {
021      private static Logger log = LoggerFactory.getLogger(CassandraHost.class);
022    
023      /**
024       * The default port number to which we will connect
025       */
026      public static final int DEFAULT_PORT = 9160;
027    
028      public static final int DEFAULT_MAX_ACTIVE = 50;
029    
030      /**
031       * By default, we will use TSocket transport on thrift (matches default Cassandra configs)
032       */
033      public static final boolean DEFAULT_USE_FRAMED_THRIFT_TRANSPORT = true;
034    
035      /**
036       * The default max wait time when exhausted happens, default value is negative, which means
037       * it'll block indefinitely.
038       */
039      public static final long DEFAULT_MAX_WAITTIME_WHEN_EXHAUSTED = -1;
040    
041      /**
042       * The default max idle number determines how many idle connections may reside in the pool.
043       * If -1 then it's infinity.
044       */
045      public static final int DEFAULT_MAX_IDLE = -1;
046    
047      public static final boolean DEFAULT_LIFO = GenericObjectPool.DEFAULT_LIFO;
048      public static final long DEFAULT_MIN_EVICTABLE_IDLE_TIME_MILLIS = GenericObjectPool.DEFAULT_MIN_EVICTABLE_IDLE_TIME_MILLIS;
049      public static final long DEFAULT_TIME_BETWEEN_EVICTION_RUNS_MILLIS = GenericObjectPool.DEFAULT_TIME_BETWEEN_EVICTION_RUNS_MILLIS;
050    
051      private final String host, ip, url;
052      private final int port;
053      private final String name;
054    
055      private int maxActive = DEFAULT_MAX_ACTIVE;
056      private int maxIdle = DEFAULT_MAX_IDLE;
057    
058      private boolean lifo = DEFAULT_LIFO;
059      private long minEvictableIdleTimeMillis = DEFAULT_MIN_EVICTABLE_IDLE_TIME_MILLIS;
060      private long timeBetweenEvictionRunsMillis = DEFAULT_TIME_BETWEEN_EVICTION_RUNS_MILLIS;
061    
062      private long maxWaitTimeWhenExhausted = DEFAULT_MAX_WAITTIME_WHEN_EXHAUSTED;
063      private int cassandraThriftSocketTimeout;
064      private ExhaustedPolicy exhaustedPolicy = ExhaustedPolicy.WHEN_EXHAUSTED_BLOCK;
065      private boolean useThriftFramedTransport = DEFAULT_USE_FRAMED_THRIFT_TRANSPORT;
066      //TODO(ran): private FailoverPolicy failoverPolicy = DEFAULT_FAILOVER_POLICY;
067    
068      public CassandraHost(String url) {
069        this(url, parsePortFromUrl(url));
070      }
071    
072      public CassandraHost(String url2, int port) {
073        url2 = parseHostFromUrl(url2);
074        this.port = port;
075        StringBuilder b = new StringBuilder();
076        InetAddress address;
077        String turl, tip;
078        try {
079          address = InetAddress.getByName(url2);
080          turl = isPerformNameResolution() ? address.getHostName() : url2;
081          tip = address.getHostAddress();
082        } catch (UnknownHostException e) {
083          log.error("Unable to resolve host {}", url2);
084          turl = url2;
085          tip = url2;
086        }
087        this.host = turl;
088        ip = tip;
089        b.append(url2);
090        b.append("(");
091        b.append(ip);
092        b.append("):");
093        b.append(port);
094        name = b.toString();
095        url = String.format("%s:%d",host,port);
096      }
097    
098      public String getUrl() {
099        return url;
100      }
101    
102      /**
103       * Checks whether name resolution should occur.
104       *
105       * @return
106       */
107      public boolean isPerformNameResolution() {
108        String sysprop = System.getProperty(
109            SystemProperties.HECTOR_PERFORM_NAME_RESOLUTION.toString());
110        return sysprop != null && Boolean.valueOf(sysprop);
111    
112      }
113    
114      public String getName() {
115        return name;
116      }
117    
118      public String getHost() {
119        return host;
120      }
121    
122      public String getIp() {
123        return ip;
124      }
125    
126      public int getPort() {
127        return port;
128      }
129    
130      @Override
131      public String toString() {
132        return name;
133      }
134    
135      @Override
136      public boolean equals(Object obj) {
137        if (! (obj instanceof CassandraHost)) {
138          return false;
139        }
140        CassandraHost other = (CassandraHost) obj;
141        return other.ip.equals(ip) && other.port == port;
142      }
143    
144      @Override
145      public int hashCode() {
146        return ip.hashCode();
147      }
148    
149      public int getMaxActive() {
150        return maxActive;
151      }
152    
153      public void setMaxActive(int maxActive) {
154        this.maxActive = maxActive;
155      }
156    
157      public int getMaxIdle() {
158        return maxIdle;
159      }
160    
161      public void setMaxIdle(int maxIdle) {
162        this.maxIdle = maxIdle;
163      }
164    
165      public long getMaxWaitTimeWhenExhausted() {
166        return maxWaitTimeWhenExhausted;
167      }
168    
169      public void setMaxWaitTimeWhenExhausted(long maxWaitTimeWhenExhausted) {
170        this.maxWaitTimeWhenExhausted = maxWaitTimeWhenExhausted;
171      }
172    
173      public ExhaustedPolicy getExhaustedPolicy() {
174        return exhaustedPolicy;
175      }
176    
177      public void setExhaustedPolicy(ExhaustedPolicy exhaustedPolicy) {
178        this.exhaustedPolicy = exhaustedPolicy;
179      }
180    
181      public int getCassandraThriftSocketTimeout() {
182        return cassandraThriftSocketTimeout;
183      }
184    
185      public void setCassandraThriftSocketTimeout(int cassandraThriftSocketTimeout) {
186        this.cassandraThriftSocketTimeout = cassandraThriftSocketTimeout;
187      }
188    
189      public boolean getUseThriftFramedTransport() {
190        return useThriftFramedTransport;
191      }
192    
193      public void setUseThriftFramedTransport(boolean useThriftFramedTransport) {
194        this.useThriftFramedTransport = useThriftFramedTransport;
195      }
196    
197      public static String parseHostFromUrl(String urlPort) {
198        return urlPort.lastIndexOf(':') > 0 ? urlPort.substring(0, urlPort.lastIndexOf(':')) : urlPort;
199      }
200    
201      public static int parsePortFromUrl(String urlPort) {
202        return urlPort.lastIndexOf(':') > 0 ? Integer.valueOf(urlPort.substring(urlPort.lastIndexOf(':')+1, urlPort.length())) : DEFAULT_PORT;
203      }
204    
205      public boolean getLifo() {
206        return lifo;
207      }
208    
209      public void setLifo(boolean lifo) {
210        this.lifo = lifo;
211      }
212    
213      public long getMinEvictableIdleTimeMillis() {
214        return minEvictableIdleTimeMillis;
215      }
216    
217      public void setMinEvictableIdleTimeMillis(long minEvictableIdleTimeMillis) {
218        this.minEvictableIdleTimeMillis = minEvictableIdleTimeMillis;
219      }
220    
221      public long getTimeBetweenEvictionRunsMillis() {
222        return timeBetweenEvictionRunsMillis;
223      }
224    
225      public void setTimeBetweenEvictionRunsMillis(long timeBetweenEvictionRunsMillis) {
226        this.timeBetweenEvictionRunsMillis = timeBetweenEvictionRunsMillis;
227      }
228
