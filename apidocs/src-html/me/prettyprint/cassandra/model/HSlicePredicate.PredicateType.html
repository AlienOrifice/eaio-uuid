001    package me.prettyprint.cassandra.model;
002    
003    import java.nio.ByteBuffer;
004    import java.util.ArrayList;
005    import java.util.Arrays;
006    import java.util.Collection;
007    import java.util.Collections;
008    import java.util.List;
009    
010    import me.prettyprint.cassandra.utils.Assert;
011    import me.prettyprint.hector.api.Serializer;
012    import me.prettyprint.hector.api.exceptions.HectorException;
013    
014    import org.apache.cassandra.thrift.SlicePredicate;
015    import org.apache.cassandra.thrift.SliceRange;
016    
017    /**
018     * Hector's version of cassandra SlicePredicate
019     *
020     * @author Ran Tavory
021     *
022     */
023    public final class HSlicePredicate<N> {
024    
025      protected Collection<N> columnNames;
026      protected N start;
027      protected N finish;
028      protected boolean reversed;
029      protected int count;
030      /** Is count already set? */
031      private boolean countSet = false;
032      protected final Serializer<N> columnNameSerializer;
033      protected enum PredicateType {Unknown, ColumnNames, Range};
034      protected PredicateType predicateType = PredicateType.Unknown;
035    
036      public HSlicePredicate(Serializer<N> columnNameSerializer) {
037        Assert.notNull(columnNameSerializer, "columnNameSerializer can't be null");
038        this.columnNameSerializer = columnNameSerializer;
039      }
040    
041      /**
042       * Sets the column names to be retrieved by this query
043       *
044       * @param columns
045       *          a list of column names
046       */
047      public HSlicePredicate<N> setColumnNames(N... columnNames) {
048        return setColumnNames(Arrays.asList(columnNames));
049      }
050      
051      /**
052       * Same as varargs signature, except we take a collection
053       *
054       * @param columns
055       *          a list of column names
056       */
057      public HSlicePredicate<N> setColumnNames(Collection<N> columnNames) {
058        this.columnNames = columnNames;
059        predicateType = PredicateType.ColumnNames;
060        return this;
061      }
062    
063      /**
064       * Allows the use of returning just the keys. This avoids de-serialization of row data
065       * and can be a huge optimization in some use cases
066       *
067       */
068      public HSlicePredicate<N> setKeysOnlyPredicate() {
069        this.columnNames = new ArrayList<N>();
070        predicateType = PredicateType.ColumnNames;
071        return this;
072      }
073    
074      /**
075       * Set a predicate of start/finish to retrieve a list of columns in this range.
076       * Either start and or finish can be null which will toggle the underlying predicate to
077       * use an empty byte[]
078       * @param start
079       *          Start key
080       * @param finish
081       *          End key
082       * @param reversed
083       * @param count
084       * @return
085       */
086      public HSlicePredicate<N> setRange(N start, N finish, boolean reversed, int count) {
087        this.start = start;
088        this.finish = finish;
089        this.reversed = reversed;
090        this.count = count;
091        countSet = true;
092        predicateType = PredicateType.Range;
093        return this;
094      }
095    
096      public Collection<N> getColumnNames() {
097        return Collections.unmodifiableCollection(columnNames);
098      }
099    
100      /**
101       * Will throw a runtime exception if neither columnsNames nor count were set.
102       * @return
103       */
104      public SlicePredicate toThrift() {
105        SlicePredicate pred = new SlicePredicate();
106    
107        switch (predicateType) {
108        case ColumnNames:
109          if (columnNames == null ) {
110            return null;
111          }
112          pred.setColumn_names(toThriftColumnNames(columnNames));
113          break;
114        case Range:
115          Assert.isTrue(countSet, "Count was not set, neither were column-names set, can't execute");
116          SliceRange range = new SliceRange(findBytes(start), findBytes(finish), reversed, count);
117          pred.setSlice_range(range);
118          break;
119        case Unknown:
120        default:
121          throw new HectorException(
122              "Neither column names nor range were set, this is an invalid slice predicate");
123        }
124        return pred;
125      }
126    
127      private ByteBuffer findBytes(N val) {
128        ByteBuffer valBytes;
129        if (val == null) {
130          valBytes =  ByteBuffer.wrap(new byte[0]);
131        } else {
132          valBytes = columnNameSerializer.toByteBuffer(val);
133        }
134        return valBytes;
135      }
136    
137      private List<ByteBuffer> toThriftColumnNames(Collection<N> clms) {
138        List<ByteBuffer> ret = new ArrayList<ByteBuffer>(clms.size());
139        for (N name : clms) {
140          ret.add(columnNameSerializer.toByteBuffer(name));
141        }
142        return ret;
143      }
144    
145      @Override
146      public String toString() {
147        return "HSlicePredicate("
148            + (predicateType == PredicateType.ColumnNames ? columnNames :
149              "cStart:" + start + ",cFinish:" + finish) + ")";
150      }
151    }
