075      @Test
076      public void testSubDelete() {
077        Mutator<String> m = createMutator(keyspace, se);
078        List<HColumn<String, String>> columnList = new ArrayList<HColumn<String,String>>();
079        columnList.add(createColumn("col_1","val_1",se,se));
080        columnList.add(createColumn("col_2","val_2",se,se));
081        columnList.add(createColumn("col_3","val_3",se,se));
082        HSuperColumn<String, String, String> superColumn =
083            createSuperColumn("super_name", columnList, se, se, se);
084        MutationResult r = m.insert("sk1", "Super1", superColumn);
085        
086        SuperColumnQuery<String, String, String, String> scq = HFactory.createSuperColumnQuery(keyspace, se, se, se, se);
087        scq.setColumnFamily("Super1");
088        scq.setKey("sk1");
089        scq.setSuperName("super_name");
090        assertEquals(3,scq.execute().get().getColumns().size());
091        
092        m.discardPendingMutations();
093        columnList.remove(1);
094        columnList.remove(0);
095        superColumn.setSubcolumns(columnList);
096        m.addSubDelete("sk1", "Super1", superColumn);
097        m.execute();
098            
099        assertEquals(2,scq.execute().get().getColumns().size());
100      }

102      @Test
103      public void testBatchMutationManagement() {
104        String cf = "Standard1";
105    
106        Mutator<String> m = createMutator(keyspace, se);
107        for (int i = 0; i < 5; i++) {
108          m.addInsertion("k" + i, cf, createColumn("name", "value" + i, se, se));
109        }
110        MutationResult r = m.execute();
111        assertTrue("Execute time should be > 0", r.getExecutionTimeMicro() > 0);
112        assertTrue("Should have operated on a host", r.getHostUsed() != null);
113    
114        for (int i = 0; i < 5; i++) {
115          assertColumnExists("Keyspace1", "Standard1", "k"+i, "name");
116        }
117        // Execute an empty mutation
118        r = m.execute();
119        assertEquals("Execute time should be 0", 0, r.getExecutionTimeMicro());
120    
121        // Test discard and then exec an empty mutation
122        for (int i = 0; i < 5; i++) {
123          m.addInsertion("k" + i, cf, createColumn("name", "value" + i, se, se));
124        }
125        m.discardPendingMutations();
126        r = m.execute();
127        assertEquals("Execute time should be 0", 0, r.getExecutionTimeMicro());
128        assertTrue("Should have operated with a null host", r.getHostUsed() == null);
129    
130        // cleanup
131        for (int i = 0; i < 5; i++) {
132          m.addDeletion("k" + i, cf, "name", se);
133        }
134        m.execute();
135      }

137      @Test
138      public void testRowDeletion() {
139        String cf = "Standard1";
140        long initialTime = keyspace.createClock();
141    
142        Mutator<String> m = createMutator(keyspace, se);
143        for (int i = 0; i < 5; i++) {
144          m.addInsertion("key" + i, cf, createColumn("name", "value" + i, se, se));
145        }
146         m.execute();
147    
148        // Try to delete the row with key "k0" with a clock previous to the insertion.
149        // Cassandra will discard this operation.
150        m.addDeletion("key0", cf, null, se, (initialTime - 100));
151        m.execute();
152    
153        // Check that the delete was harmless
154        QueryResult<HColumn<String, String>> columnResult =
155            createColumnQuery(keyspace, se, se, se).setColumnFamily(cf).setKey("key0").
156                setName("name").execute();
157        assertEquals("value0", columnResult.get().getValue());
158    
159        for (int i = 0; i < 5; i++) {
160          m.addDeletion("key" + i, cf, null, se);
161        }
162        m.execute();
163    
164        // Check that the delete took place now
165        columnResult = createColumnQuery(keyspace, se, se, se).
166            setColumnFamily(cf).setKey("key0").setName("name").execute();
167        assertNull(columnResult.get());
168      }
