003    /**
004     * Specifies the "type" of operation - read or write.
005     * It's used for perf4j, so should be in sync with hectorLog4j.xml
006     * @author Ran Tavory (ran@outbain.com)
007     * 
008     */
009    public enum OperationType {

011      READ,
012      /** Write operations */
013      WRITE,
014      /** Meta read operations, such as describe*() */
015      META_READ,
016      /** Operation on one of the system_ methods */
017      META_WRITE;
018    }
