019    /**
020     * A cluster instance the client side representation of a cassandra server cluster.
021     *
022     * The cluster is usually the main entry point for programs using hector. To start operating on
023     * cassandra cluster you first get or create a cluster, then a keyspace operator for the keyspace
024     * you're interested in and then create mutations of queries
025     * <code>
026     * //get a cluster:
027     * Cluster cluster = getOrCreateCluster("MyCluster", new CassandraHostConfigurator("127.0.0.1:9170"));
028     * //get a keyspace from this cluster:
029     * Keyspace ko = createKeyspace("Keyspace1", cluster);
030     * //Create a mutator:
031     * Mutator m = createMutator(ko);
032     * // Make a mutation:
033     * MutationResult mr = m.insert("key", cf, createColumn("name", "value", serializer, serializer));
034     * </code>
035     *
036     * THREAD SAFETY: This class is thread safe.
037     *
038     * @author Ran Tavory
039     * @author zznate
040     */
041    public abstract class AbstractCluster implements Cluster {
042      private static final Map<String, String> EMPTY_CREDENTIALS = Collections.emptyMap();
043    
044      private final Logger log = LoggerFactory.getLogger(AbstractCluster.class);
045    
046      protected final HConnectionManager connectionManager;
047      private final String name;
048      private final CassandraHostConfigurator configurator;
049      private final ClockResolution clockResolution;
050      private final FailoverPolicy failoverPolicy;
051      private final CassandraClientMonitor cassandraClientMonitor;
052      private Set<String> knownClusterHosts;
053      private Set<CassandraHost> knownPoolHosts;
054      protected final ExceptionsTranslator xtrans;
055      private final Map<String, String> credentials;
056    
057      public AbstractCluster(String clusterName, CassandraHostConfigurator cassandraHostConfigurator) {
058        this(clusterName, cassandraHostConfigurator, EMPTY_CREDENTIALS);
059      }
060    
061      public AbstractCluster(String clusterName, CassandraHostConfigurator cassandraHostConfigurator, Map<String, String> credentials) {
062        connectionManager = new HConnectionManager(cassandraHostConfigurator);
063        name = clusterName;
064        configurator = cassandraHostConfigurator;
065        failoverPolicy = FailoverPolicy.ON_FAIL_TRY_ALL_AVAILABLE;
066        cassandraClientMonitor = JmxMonitor.getInstance(connectionManager).getCassandraMonitor();
067        xtrans = new ExceptionsTranslatorImpl();
068        clockResolution = cassandraHostConfigurator.getClockResolution();
069        this.credentials = Collections.unmodifiableMap(credentials);
070      }
071    
072      @Override
073      public HConnectionManager getConnectionManager() {
074        return connectionManager;
075      }
076    
077      /* (non-Javadoc)
078       * @see me.prettyprint.cassandra.service.Cluster#getKnownPoolHosts(boolean)
079       */
080      @Override
081      public Set<CassandraHost> getKnownPoolHosts(boolean refresh) {
082        if (refresh || knownPoolHosts == null) {
083          knownPoolHosts = connectionManager.getHosts();
084          if ( log.isInfoEnabled() ) {
085            log.info("found knownPoolHosts: {}", knownPoolHosts);
086          }
087        }
088        return knownPoolHosts;
089      }
090    
091    
092      /* (non-Javadoc)
093       * @see me.prettyprint.cassandra.service.Cluster#addHost(me.prettyprint.cassandra.service.CassandraHost, boolean)
094       */
095      @Override
096      public void addHost(CassandraHost cassandraHost, boolean skipApplyConfig) {
097        if (!skipApplyConfig && configurator != null) {
098          configurator.applyConfig(cassandraHost);
099        }
100        connectionManager.addCassandraHost(cassandraHost);
101      }

116      /* (non-Javadoc)
117       * @see me.prettyprint.cassandra.service.Cluster#describeKeyspaces()
118       */
119      @Override
120      public List<KeyspaceDefinition> describeKeyspaces() throws HectorException {
121        Operation<List<KeyspaceDefinition>> op = new Operation<List<KeyspaceDefinition>>(OperationType.META_READ, getCredentials()) {
122          @Override
123          public List<KeyspaceDefinition> execute(Cassandra.Client cassandra) throws HectorException {
124            try {
125              return ThriftKsDef.fromThriftList(cassandra.describe_keyspaces());
126            } catch (Exception e) {
127              throw xtrans.translate(e);
128            }
129          }
130        };
131        connectionManager.operateWithFailover(op);
132        return op.getResult();
133      }

173      /* (non-Javadoc)
174       * @see me.prettyprint.cassandra.service.Cluster#describeKeyspace(java.lang.String)
175       */
176      @Override
177      public KeyspaceDefinition describeKeyspace(final String keyspace)
178      throws HectorException {
179        Operation<KeyspaceDefinition> op = new Operation<KeyspaceDefinition>(
180            OperationType.META_READ, getCredentials()) {
181          @Override
182          public KeyspaceDefinition execute(Cassandra.Client cassandra)
183          throws HectorException {
184            try {
185              return new ThriftKsDef(cassandra.describe_keyspace(keyspace));
186            } catch (org.apache.cassandra.thrift.NotFoundException nfe) {
187              setException(xtrans.translate(nfe));
188              return null;
189            } catch (Exception e) {
190              throw xtrans.translate(e);
191            }
192          }
193        };
194        connectionManager.operateWithFailover(op);
195        return op.getResult();
196      }
197

199       * @see me.prettyprint.cassandra.service.Cluster#getClusterName()

201      @Override
202      public String getClusterName() throws HectorException {
203        log.info("in execute with client");
204        Operation<String> op = new Operation<String>(OperationType.META_READ, getCredentials()) {
205          @Override
206          public String execute(Cassandra.Client cassandra) throws HectorException {
207            try {
208              log.info("in execute with client {}", cassandra);
209              return cassandra.describe_cluster_name();
210            } catch (Exception e) {
211              throw xtrans.translate(e);
212            }
213          }
214        };
215        connectionManager.operateWithFailover(op);
216        return op.getResult();
217      }
