42      notNull(columnFamilyName, "columnFamilyName is null");
43      notNull(superName, "superName is null");
44      return new QueryResultImpl<HSuperColumn<SN, N, V>>(keyspace.doExecute(
45          new KeyspaceOperationCallback<HSuperColumn<SN, N, V>>() {
46            @Override
47            public HSuperColumn<SN, N, V> doInKeyspace(KeyspaceService ks) throws HectorException {
48              try {
49                ColumnPath cpath = ThriftFactory.createSuperColumnPath(columnFamilyName, superName, (N) null,
50                    sNameSerializer, columnNameSerializer);
51                SuperColumn thriftSuperColumn = ks.getSuperColumn(keySerializer.toByteBuffer(key), cpath);
52                if (thriftSuperColumn == null) {
53                  return null;
54                }
55                return new HSuperColumnImpl<SN, N, V>(thriftSuperColumn, sNameSerializer, columnNameSerializer,
56                    valueSerializer);
57              } catch (HNotFoundException e) {
58                return null;
59              }
60            }
61          }), this);
