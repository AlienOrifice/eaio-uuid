001    package me.prettyprint.cassandra.model;
002    
003    import static me.prettyprint.cassandra.utils.Assert.noneNull;
004    import me.prettyprint.hector.api.Keyspace;
005    import me.prettyprint.hector.api.Serializer;
006    import me.prettyprint.hector.api.beans.HSuperColumn;
007    import me.prettyprint.hector.api.query.SuperColumnQuery;
008    
009    /**
010     * Defines the commonalities b/w avro and thrift SuperColumnQuery implementations
011     *
012     * @author Ran Tavory
013     *
014     * @param <SN>
015     * @param <N>
016     * @param <V>
017     */
018    public abstract class AbstractSuperColumnQuery<K, SN, N, V> extends
019        AbstractQuery<K, N, V, HSuperColumn<SN, N, V>> implements SuperColumnQuery<K, SN, N, V>{
020    
021      protected final Serializer<SN> sNameSerializer;
022      protected K key;
023      protected SN superName;
024    
025      protected AbstractSuperColumnQuery(Keyspace k,
026          Serializer<K> keySerializer,
027          Serializer<SN> sNameSerializer,
028          Serializer<N> nameSerializer,
029          Serializer<V> valueSerializer) {
030        super(k, keySerializer, nameSerializer, valueSerializer);
031        noneNull(sNameSerializer, nameSerializer, valueSerializer);
032        this.sNameSerializer = sNameSerializer;
033      }
034    
035      @Override
036      public SuperColumnQuery<K, SN, N, V> setKey(K key) {
037        this.key = key;
038        return this;
039      }
040    
041      @Override
042      public SuperColumnQuery<K, SN, N, V> setSuperName(SN superName) {
043        this.superName = superName;
044        return this;
045      }
046    
047      @SuppressWarnings("unchecked")
048      @Override
049      public SuperColumnQuery<K, SN, N, V> setColumnFamily(String cf) {
050        return (SuperColumnQuery<K, SN, N, V>) super.setColumnFamily(cf);
051      }
052    
053      @Override
054      public String toString() {
055        return "SuperColumnQuery(" + key + "," + superName + ")";
056      }
057    }
