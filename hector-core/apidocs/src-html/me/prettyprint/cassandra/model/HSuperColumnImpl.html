001    package me.prettyprint.cassandra.model;
002    
003    import static me.prettyprint.cassandra.utils.Assert.noneNull;
004    import static me.prettyprint.cassandra.utils.Assert.notNull;
005    
006    import java.nio.ByteBuffer;
007    import java.util.ArrayList;
008    import java.util.Collections;
009    import java.util.List;
010    
011    import me.prettyprint.hector.api.Serializer;
012    import me.prettyprint.hector.api.beans.HColumn;
013    import me.prettyprint.hector.api.beans.HSuperColumn;
014    
015    import org.apache.cassandra.thrift.Column;
016    import org.apache.cassandra.thrift.SuperColumn;
017    
018    /**
019     * Models a SuperColumn in a protocol independant manner
020     *
021     * @param <SN>
022     *          SuperColumn name type
023     * @param <N>
024     *          Column name type
025     * @param <V>
026     *          Column value type
027     *
028     * @author zznate
029     */
030    public final class HSuperColumnImpl<SN,N,V> implements HSuperColumn<SN, N, V> {
031    
032      private SN superName;
033      private List<? extends HColumn<N,V>> columns;
034      private long clock;
035      private final Serializer<SN> superNameSerializer;
036      private final Serializer<N> nameSerializer;
037      private final Serializer<V> valueSerializer;
038    
039      /**
040       * @param <SN> SuperColumn name type
041       * @param List<HColumn<N,V>> Column values
042       * @param Serializer<SN> the serializer type
043       * @param clock
044       */
045      public HSuperColumnImpl(SN sName, List<HColumn<N, V>> columns, long clock,
046          Serializer<SN> sNameSerializer, Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
047        this(sNameSerializer, nameSerializer, valueSerializer);
048        notNull(sName, "Name is null");
049        notNull(columns, "Columns are null");
050        this.superName = sName;
051        this.columns = columns;
052        this.clock = clock;
053      }
054    
055      public HSuperColumnImpl(SuperColumn thriftSuperColumn, Serializer<SN> sNameSerializer,
056          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
057        this(sNameSerializer, nameSerializer, valueSerializer);
058        noneNull(thriftSuperColumn, sNameSerializer, nameSerializer, valueSerializer);
059        superName = sNameSerializer.fromByteBuffer(ByteBuffer.wrap(thriftSuperColumn.getName()));
060        columns = fromThriftColumns(thriftSuperColumn.getColumns());
061      }
062    
063      /*package*/ HSuperColumnImpl(Serializer<SN> sNameSerializer, Serializer<N> nameSerializer,
064          Serializer<V> valueSerializer) {
065        noneNull(sNameSerializer, nameSerializer, valueSerializer);
066        this.superNameSerializer = sNameSerializer;
067        this.nameSerializer = nameSerializer;
068        this.valueSerializer = valueSerializer;
069      }
070    
071      @Override
072      public HSuperColumn<SN, N, V> setName(SN name) {
073        notNull(name, "name is null");
074        this.superName = name;
075        return this;
076      }
077    
078      @Override
079      public HSuperColumn<SN, N, V> setSubcolumns(List<HColumn<N, V>> subcolumns) {
080        notNull(subcolumns, "subcolumns are null");
081        this.columns = subcolumns;
082        return this;
083      }
084    
085      @Override
086      public HSuperColumn<SN, N, V> setClock(long clock) {
087        this.clock = clock;
088        return this;
089      }
090    
091      @Override
092      public long getClock() {
093        return clock;
094      }
095    
096      @Override
097      public int getSize() {
098        return columns == null ? 0 : columns.size();
099      }
100    
101      @Override
102      public SN getName() {
103        return superName;
104      }
105    
106      /**
107       *
108       * @return an unmodifiable list of columns
109       */
110      @Override
111      public List<HColumn<N,V>> getColumns() {
112        return Collections.unmodifiableList(columns);
113      }
114    
115      @Override
116      public HColumn<N, V> get(int i) {
117        return columns.get(i);
118      }
119    
120      @Override
121      public Serializer<SN> getNameSerializer() {
122        return superNameSerializer;
123      }
124    
125      @Override
126      public byte[] getNameBytes() {
127        return superNameSerializer.toByteBuffer(getName()).array();
128      }
129      
130      public ByteBuffer getNameByteBuffer() {
131        return superNameSerializer.toByteBuffer(getName());
132      }
133    
134      public SuperColumn toThrift() {
135        if (superName == null || columns == null) {
136          return null;
137        }
138        return new SuperColumn(superNameSerializer.toByteBuffer(superName), toThriftColumn());
139      }
140    
141      private List<Column> toThriftColumn() {
142        List<Column> ret = new ArrayList<Column>(columns.size());
143        for (HColumn<N, V> c: columns) {
144          ret.add(((HColumnImpl<N, V>) c).toThrift());
145        }
146        return ret;
147      }
148    
149      private List<HColumn<N, V>> fromThriftColumns(List<Column> tcolumns) {
150        List<HColumn<N, V>> cs = new ArrayList<HColumn<N,V>>(tcolumns.size());
151        for (Column c: tcolumns) {
152          cs.add(new HColumnImpl<N, V>(c, nameSerializer, valueSerializer));
153        }
154        return cs;
155      }
156    
157      @Override
158      public Serializer<SN> getSuperNameSerializer() {
159        return superNameSerializer;
160      }
161    
162      @Override
163      public Serializer<V> getValueSerializer() {
164        return valueSerializer;
165      }
166    
167      @Override
168      public String toString() {
169        return String.format("HSuperColumn(%s,%s)", superName, columns);
170      }
171    }
