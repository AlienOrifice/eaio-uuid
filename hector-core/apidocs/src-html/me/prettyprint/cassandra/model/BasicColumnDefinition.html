001    package me.prettyprint.cassandra.model;
002    
003    import java.nio.ByteBuffer;
004    
005    import me.prettyprint.hector.api.ddl.ColumnDefinition;
006    import me.prettyprint.hector.api.ddl.ColumnIndexType;
007    
008    /**
009     * @author: peter
010     */
011    public class BasicColumnDefinition implements ColumnDefinition {
012    
013      private ByteBuffer name;
014      private String validationClass;
015      private ColumnIndexType indexType;
016      private String indexName;
017    
018      @Override
019      public ByteBuffer getName() {
020        return name;
021      }
022    
023      @Override
024      public String getValidationClass() {
025        return validationClass;
026      }
027    
028      @Override
029      public ColumnIndexType getIndexType() {
030        return indexType;
031      }
032    
033      @Override
034      public String getIndexName() {
035        return indexName;
036      }
037    
038      public void setName(ByteBuffer name) {
039        this.name = name;
040      }
041    
042      public void setValidationClass(String validationClass) {
043        this.validationClass = validationClass;
044      }
045    
046      public void setIndexType(ColumnIndexType indexType) {
047        this.indexType = indexType;
048      }
049    
050      public void setIndexName(String indexName) {
051        this.indexName = indexName;
052      }
053    
054    
055    }
