31    public AbstractThriftCountQuery(Keyspace k, Serializer<K> keySerializer,
32        Serializer<N> nameSerializer) {
33      Assert.notNull(k, "keyspaceOperator can't be null");
34      Assert.notNull(nameSerializer, "nameSerializer is null");
35      Assert.notNull(keySerializer, "keySerializer is null");
36      this.keyspace = (ExecutingKeyspace) k;
37      this.keySerializer = keySerializer;
38      this.slicePredicate = new HSlicePredicate<N>(nameSerializer);
39    }

42      this.key = key;

44    }

48      return this;
49    }

51    protected  QueryResult<Integer> countColumns() {
52      Assert.notNull(key, "key is null");
53      Assert.notNull(columnFamily, "columnFamily is null");
54      return new QueryResultImpl<Integer>(keyspace.doExecute(
55          new KeyspaceOperationCallback<Integer>() {
56            @Override
57            public Integer doInKeyspace(KeyspaceService ks) throws HectorException {
58              ColumnParent columnParent = new ColumnParent(columnFamily);
59              return ks.getCount(keySerializer.toByteBuffer(key), columnParent,
60                  slicePredicate.toThrift());
61            }
62          }), this);
63    }

68    }
