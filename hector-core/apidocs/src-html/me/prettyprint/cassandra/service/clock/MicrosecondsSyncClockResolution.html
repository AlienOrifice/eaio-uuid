001    package me.prettyprint.cassandra.service.clock;
002    
003    import me.prettyprint.hector.api.ClockResolution;
004    
005    
006    /**
007     * Synchronized Milliseconds Resolution used to create clocks.
008     *
009     * @author Patricio Echague (pechague@gmail.com)
010     */
011    public class MicrosecondsSyncClockResolution extends AbstractClockResolution implements ClockResolution {
012    
013      private static final long ONE_THOUSAND = 1000L;
014    
015        /**
016         * The last time value issued. Used to try to prevent duplicates.
017         */
018      private static long lastTime = -1;
019    
020      @Override
021      public long createClock() {
022        // The following simulates a microseconds resolution by advancing a static counter
023        // every time a client calls the createClock method, simulating a tick.
024        long us = getSystemMilliseconds() * ONE_THOUSAND;
025        // Synchronized to guarantee unique time within and across threads.
026        synchronized (MicrosecondsSyncClockResolution.class) {
027          if (us > lastTime) {
028            lastTime = us;
029          } else {
030            // the time i got from the system is equals or less
031            // (hope not - clock going backwards)
032            // One more "microsecond"
033            us = ++lastTime;
034          }
035        }
036        return us;
037      }
038    
039    }
