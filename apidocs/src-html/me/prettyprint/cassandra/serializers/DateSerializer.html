001    package me.prettyprint.cassandra.serializers;
002    
003    import java.nio.ByteBuffer;
004    import java.util.Date;
005    
006    /**
007     * Converts bytes to Date and vice versa, by first converting the Date to or
008     * from a long which represents the specified number of milliseconds since
009     * the standard base time known as "the Unix epoch", that is
010     * January 1, 1970, 00:00:00 UTC.
011     *
012     * @author Jim Ancona
013     * @see java.util.Date
014     */
015    public final class DateSerializer extends AbstractSerializer<Date> {
016      private static final LongSerializer LONG_SERIALIZER = LongSerializer.get();
017      private static final DateSerializer instance = new DateSerializer();
018    
019      public static DateSerializer get() {
020        return instance;
021      }
022    
023      @Override
024      public ByteBuffer toByteBuffer(Date obj) {
025        if (obj == null) {
026          return null;
027        }
028        return LONG_SERIALIZER.toByteBuffer(obj.getTime());
029      }
030    
031      @Override
032      public Date fromByteBuffer(ByteBuffer bytes) {
033        if (bytes == null) {
034          return null;
035        }
036        return new Date(LONG_SERIALIZER.fromByteBuffer(bytes));
037      }
038    }
