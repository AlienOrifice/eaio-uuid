001    package me.prettyprint.cassandra.model;
002    
003    import java.util.ArrayList;
004    import java.util.Collections;
005    import java.util.HashMap;
006    import java.util.List;
007    import java.util.Map;
008    
009    import me.prettyprint.cassandra.utils.Assert;
010    import me.prettyprint.hector.api.Serializer;
011    import me.prettyprint.hector.api.beans.HSuperColumn;
012    import me.prettyprint.hector.api.beans.SuperSlice;
013    
014    import org.apache.cassandra.thrift.SuperColumn;
015    
016    /**
017     * Represents a return of the get_slice query for supercolumns
018     *
019     * @param <SN> Super column name type
020     * @param <N> Column name type
021     * @param <V> Column value type
022     */
023    public final class SuperSliceImpl<SN,N,V> implements SuperSlice<SN, N, V> {
024    
025      private final Map<SN,HSuperColumn<SN,N,V>> columnsMap;
026    
027      private final List<HSuperColumn<SN,N,V>> columnsList;
028    
029      public SuperSliceImpl(List<SuperColumn> tSuperColumns, Serializer<SN> sNameSerializer,
030          Serializer<N> nameSerializer, Serializer<V> valueSerializer) {
031        Assert.noneNull(tSuperColumns, sNameSerializer, nameSerializer, valueSerializer);
032        columnsMap = new HashMap<SN,HSuperColumn<SN,N,V>>(tSuperColumns.size());
033        columnsList = new ArrayList<HSuperColumn<SN,N,V>>(tSuperColumns.size());
034        for (SuperColumn sc: tSuperColumns) {
035          HSuperColumn<SN,N,V> column = new HSuperColumnImpl<SN,N,V>(sc, sNameSerializer, nameSerializer,
036              valueSerializer);
037          columnsMap.put(column.getName(), column);
038          columnsList.add(column);
039        }
040      }
041    
042      /**
043       * @return an unmodifiable list of supercolumns
044       */
045      @Override
046      public List<HSuperColumn<SN,N,V>> getSuperColumns() {
047        return Collections.unmodifiableList(columnsList);
048    
049      }
050    
051      @Override
052      public HSuperColumn<SN, N, V> getColumnByName(SN columnName) {
053        return columnsMap.get(columnName);
054      }
055      @Override
056      public String toString() {
057        return "SuperSlice(" + columnsList.toString() + ")";
058      }
059    }
